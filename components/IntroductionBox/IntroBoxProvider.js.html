import React, { Component, createContext, createRef } from 'react'
import {
  View, StyleSheet, SafeAreaView, BackHandler
} from 'react-native'
import PropTypes from 'prop-types'
import { debounce } from 'utils/nbhHelper'
import IntroBoxShowCase from './IntroBoxShowCase'
import {
  layoutsEqual, createRegistry, measure, extractObjectProperty, ReferencesHandler
} from './utils'
import { injectContext } from './context-injector'
import IntroBoxRenderer from './IntroBoxRenderer'
 
export const IntroductionBoxContext = createContext({})
export const withIntroBoxCtx = injectContext(IntroductionBoxContext, 'ctx')
 
export const IntroBoxRef = new ReferencesHandler()
 
const styles = StyleSheet.create({
  flex1: {
    flex: 1
  },
  safeArea: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0
  }
})
 
const isLinearGradientRotate = (windowLayout = {}, targetLayout = {}) => {
  let isValid = false
  const halfHT = windowLayout.height / 2
  if (!(targetLayout.y <= halfHT)) { isValid = true }
  return isValid
}
 
class IntroBoxProvider extends Component {
  intoBoxCtx = {};
 
  ownLayout = null;
 
  safeAreaLayout = null;
 
  introBoxRegistry = createRegistry();
 
  placeholderRef = createRef();
 
  outsideActionHandler = {
    openBox: debounce(name => this.openBox(name), 800),
    closeBox: () => this.effectiveClose()
  }
 
  isIntroBoxClosing = false;
 
  constructor(props) {
    super(props)
    this.intoBoxCtx = {
      introBoxActions: {
        openBox: name => this.openBox(name),
        closeBox: () => this.closeBox(),
        isIntroBoxOpened: () => this.isBoxOpened(),
        getOpenedBox: () => this.getOpenedBox(),
        updateLayoutInfo: layoutInfo => this.effectiveUpdateInfo(layoutInfo),
        effectiveClose: () => this.effectiveClose()
      },
      introBoxRegistry: this.introBoxRegistry
    }
    IntroBoxRef.setReference(this.outsideActionHandler)
  }
 
  componentDidMount() {
    if (BackHandler) {
      this.backPressListener = BackHandler.addEventListener('hardwareBackPress', this.handleBackButton)
    }
  }
 
  componentWillUnmount() {
    this.backPressListener?.remove()
    IntroBoxRef.freeMemory()
  }
 
  onLayout = ({ nativeEvent: { layout } }) => {
    if (layoutsEqual(this.ownLayout, layout)) {
      return
    }
    this.ownLayout = layout
  }
 
  onSafeAreaLayout = ({ nativeEvent: { layout } }) => {
    if (layoutsEqual(this.safeAreaLayout, layout)) {
      return
    }
    this.safeAreaLayout = layout
  }
 
  getOpenedBox() {
    const name = this.placeholderRef.current && this.placeholderRef.current?.state?.openedBoxName
    const box = name ? this.introBoxRegistry.getBox(name) : undefined
    return box
  }
 
  handleBackButton = () => {
    if (this.isBoxOpened()) {
      this.effectiveClose()
      return true
    }
    return false
  }
 
  isBoxOpened = () => (!!this.getOpenedBox())
 
  closeBox = box => Promise.resolve().then(() => {
    this.introBoxRegistry.getAll().filter(b => b.instance.getOpened())
      .forEach(b => b.instance.setOpened(false))
  }).then(() => this?.backdropRef?.close?.().then(() => {
    if (box?.name || this?.openedBox?.name) {
      this.introBoxRegistry.updateLayoutInfo(box?.name || this?.openedBox?.name, {
        targetLayout: undefined,
        windowLayout: undefined,
        safeAreaLayout: undefined,
        boxViewLayout: undefined,
        pointerLayoutInfo: undefined
      })
    }
    if (!box) { this.openedBox = undefined }
    this.placeholderRef.current.setState({ openedBoxName: '' }, () => {
 
    })
  }))
 
  openBox = (name) => {
    const box = this.introBoxRegistry.getBox(name)
    if (!box) { return }
    box.instance.setOpened(true)
    this.notify()
  }
 
  effectiveClose = () => this.beforeClose()
 
  effectiveUpdateInfo = (layoutInfo) => {
    const boxName = this.getOpenedBox().instance.getName()
    return Promise.resolve().then(() => this.introBoxRegistry.updateLayoutInfo(boxName, layoutInfo))
  }
 
  onBackdropRef = (r) => {
    this.backdropRef = r
  }
 
  onRendererRef = (r) => {
    this.wrapperRendererRef = r
  }
 
  onCloseByClickingBackdrop = () => {
    this.closeBox()
  }
 
  makeIntroBox = () => {
    const { instance } = this.getOpenedBox()
    const elemView = instance.getIntroBox()
    const targetView = instance.getIntroBoxTarget()
    const targetViewProps = extractObjectProperty(targetView.props, ['children'])
    const targetViewStyle = (targetViewProps?.style || {})
    const boxElemProps = extractObjectProperty(elemView.props, ['children'])
    return React.createElement(IntroBoxRenderer, {
      boxName: instance.getName(),
      intoBoxCtx: this.intoBoxCtx,
      targetBoxViewStyle: targetViewStyle,
      ...targetViewProps,
      ...boxElemProps,
      ref: this.onRendererRef,
      onClose: this.onCloseByClickingBackdrop
    }, [targetView, elemView])
  }
 
  wrapperRendererRef;
 
  isInitialized() {
    return (!!this.ownLayout)
  }
 
  isBoxOpen() {
    return !!this.getOpenedBox()
  }
 
  beforeClose(box) {
    return (this?.wrapperRendererRef?.close()?.then?.(() => this.closeBox(box)))
  }
 
  notify() {
    const NULL = {}
    const prev = this.openedBox || NULL
    const next = this.introBoxRegistry.getAll().find((box) => {
      if (box.instance.isOpen() && prev?.name !== box?.name) {
        return true
      }
      return false
    }) || NULL
 
    this.openedBox = next === NULL ? undefined : next
    let beforeClose = () => Promise.resolve()
    let afterClose = () => Promise.resolve()
    if (prev?.name !== next?.name) {
      if (prev !== NULL && prev.instance.isOpen()) {
        beforeClose = () => this.beforeClose(prev)
      }
 
      if (next !== NULL) {
        afterClose = () => this.initOpen(next)
      }
    }
    return beforeClose().then(() => {
      if (!this.placeholderRef.current) {
        // eslint-disable-next-line no-console
        console.warn('setState ignored - maybe the context was unmounted')
        return
      }
      afterClose?.().then(() => {
        this.placeholderRef.current.setState({
          openedBoxName: this.openedBox?.name,
          isRotate: isLinearGradientRotate(this.ownLayout, this.openedBox.targetLayout)
        }, () => this.backdropRef?.open?.())
      })
    })
  }
 
  initOpen(box) {
    return measure(box.instance?.getTarget?.()).then((res) => {
      /**
       * Saving the measures into registry
       */
      this.introBoxRegistry.updateLayoutInfo(box?.name, {
        targetLayout: res,
        windowLayout: this.ownLayout,
        safeAreaLayout: this.safeAreaLayout
      })
    })
  }
 
  backdropRef;
 
  openedBox;
 
  render() {
    return (
      <IntroductionBoxContext.Provider value={this.intoBoxCtx}>
        <View
          style={styles.flex1}
          onLayout={this.onLayout}
        >
          <View style={styles.flex1}>
            {this.props.children}
          </View>
          <SafeAreaView
            style={styles.safeArea}
            pointerEvents="box-none"
          >
            <View
              style={styles.flex1}
              collapsable={false}
              pointerEvents="box-none"
              onLayout={this.onSafeAreaLayout}
            />
            <IntroBoxShowCase
              ctx={this}
              ref={this.placeholderRef}
            />
          </SafeAreaView>
        </View>
      </IntroductionBoxContext.Provider>
    )
  }
}
 
IntroBoxProvider.propTypes = {
  children: PropTypes.array.isRequired
}
IntroBoxProvider.defaultProps = {}
 
export default IntroBoxProvider
