function RateStar(props) {
  const { style, isSelected, onPress } = props
  const onRenderPath = () => (
    <Path
      fill={isSelected ? 'white' : 'none'}
      stroke="white"
      d={STAR_PATH}
    />
  )
  return (
    <TouchableOpacity
      onPress={onPress}
      style={style}
      activeOpacity={1}
    >
      <Svg
        width="100%"
        height="100%"
        viewBox="0 0 37 35"
        xmlns="http://www.w3.org/2000/svg"
      >
        {onRenderPath()}
      </Svg>
    </TouchableOpacity>
  )
}
RateStar.propTypes = {
  style: PropTypes.object,
  isSelected: PropTypes.bool,
  onPress: PropTypes.func
}
RateStar.defaultProps = {
  style: null,
  isSelected: false,
  onPress: () => {}
}
 
const NO_RATE_STARS = 5
const RATE_STARS = Array.from({ length: NO_RATE_STARS }, (_, index) => index + 1)
 
function RateStars(props) {
  const {
    startSize, onSelect, style, rateIndex, title
  } = props
  const styles = useStyles(createRateStartsStyles, { mainViewStyle: style, startSize })
  const checkRatingStar = rateValue => (rateIndex <= NO_RATE_STARS
    && rateIndex > -1 && rateValue <= rateIndex)
  const onRateStarClick = Rindex => () => {
    onSelect?.(Rindex)
  }
  return (
    <View style={styles.main}>
      <Text style={styles.titleStyle}>{title}</Text>
      <View style={styles.rateStarsRowStyle}>
        {RATE_STARS.map(rateValue => (
          <RateStar
            onPress={onRateStarClick(rateValue)}
            key={`${rateValue}`}
            style={styles.rateStarStyle}
            isSelected={checkRatingStar(rateValue)}
          />
        ))}
      </View>
    </View>
  )
}
 
RateStars.propTypes = {
  startSize: PropTypes.number,
  onSelect: PropTypes.func,
  style: PropTypes.object,
  rateIndex: PropTypes.number,
  title: PropTypes.string
}
RateStars.defaultProps = {
  startSize: 30,
  onSelect: () => {},
  style: null,
  rateIndex: -1,
  title: ''
}
 
export default RateStars
