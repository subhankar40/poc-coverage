import React, { useCallback, useEffect, memo } from 'react'
import { View, StyleSheet } from 'react-native'
import PropTypes from 'prop-types'
import Animated, {
  useSharedValue, withTiming, Easing, useAnimatedStyle, runOnJS
} from 'react-native-reanimated'
import colors from 'utils/colors'
 
const styles = StyleSheet.create({
  container: {
    flexDirection: 'row',
    alignItems: 'center',
    justifyContent: 'center',
    flex: 1
  }
})
 
let targetOpacity = 1
 
const AnimatedEllipsis = (props) => {
  const {
    numberOfDots,
    animationDelay,
    minOpacity,
    style
  } = props
 
  const initializeDots = () => {
    const opacities = []
    for (let i = 0; i < numberOfDots; i += 1) {
      const dot = useSharedValue(minOpacity)
      opacities.push(dot)
    }
    return opacities
  }
 
  const dotOpacities = initializeDots()
 
  const animateDots = useCallback((whichDot) => {
    let whichDotParam = whichDot
    if (whichDotParam >= dotOpacities.length) {
      whichDotParam = 0
      const min = minOpacity
      targetOpacity = targetOpacity === min ? 1 : min
    }
 
    const nextDot = whichDotParam + 1
    dotOpacities[whichDotParam].value = withTiming(targetOpacity, {
      duration: animationDelay,
      easing: Easing.linear
    }, isFinised => isFinised
    && runOnJS(animateDots)(nextDot))
  }, [])
 
  useEffect(() => {
    animateDots(0)
  }, [])
 
  const dots = dotOpacities?.map?.((o, i) => {
    const opacityAnimation = useAnimatedStyle(() => ({
      opacity: o.value
    }))
 
    return (
      <Animated.Text key={i?.toString?.()} style={[style, opacityAnimation]}>
        {' '}
        .
      </Animated.Text>
    )
  })
 
  return <View style={styles.container}>{dots}</View>
}
 
AnimatedEllipsis.propTypes = {
  numberOfDots: PropTypes.number,
  animationDelay: PropTypes.number,
  minOpacity: PropTypes.number,
  style: PropTypes.object
}
 
AnimatedEllipsis.defaultProps = {
  numberOfDots: 3,
  animationDelay: 300,
  minOpacity: 0,
  style: {
    color: colors.smooth_gray,
    fontSize: 32
  }
}
 
export default memo(AnimatedEllipsis)
