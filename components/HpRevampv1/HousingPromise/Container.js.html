import React, {
  useEffect, useState, memo, useCallback
} from 'react'
import { TouchableOpacity, View, Keyboard } from 'react-native'
import PropTypes from 'prop-types'
import { useDispatch, useSelector, shallowEqual } from 'react-redux'
 
import GenericText from 'common/components/react-native-generic-text'
import images from 'utils/images'
import { PHRASES, ACTION_FAILURE } from 'constants/StringConstants'
import { HP_PKG_PRODUCT_TYPE } from 'utils/constants'
import { emptyFunc } from 'common/utils/ValidationUtils'
import { createPromiseDispatch } from 'utils/CommonUtils'
import { getPromiseProducts, getNonNpoDeepLinkParams } from 'views/TenenatMonetisation/selectors'
import { fetchHPProducts, setCouponHP } from 'views/TenenatMonetisation/TMActions'
import HPAnimatedLoader from 'components/HpRevampv1/HPAnimatedLoader'
import HPNudgeTopTag from 'components/HpNudge/HPNudgeTopTag'
import { SimpleText, textTypes, fontType } from 'common/components/SimpleText'
import { RentProtectSelectors } from 'selectors/rentProtect'
import isEmpty from 'common/utils/isEmpty'
import { checkCouponExist } from 'views/TenenatMonetisation/utils'
import { showNewToast } from 'actions/toastActions'
import { getHeight } from 'common/utils/deviceDimensions'
import PromiseContainer from './PromiseContainer'
import { strings } from '../utils'
import { promisetStyles as styles } from './styles'
import ApplyCoupon from './ApplyCoupon'
 
function Container(props) {
  const {
    closeModal, onPress, serviceName,
    fromNonNpoNudge
  } = props
  const [state, setState] = useState({
    isContentShow: false,
    isContentError: false
  })
 
  const dispatch = useDispatch()
  const { routeParams } = useSelector(getNonNpoDeepLinkParams, shallowEqual)
 
  /**
   * couponDetail will be coming from backend on the bass of uses's applied coupon
   * and isCouponDetailLoading will be set to true if api in process
   */
  const [couponConfig, setCouponConfig] = useState({
    paramCoupon: '',
    couponDetail: null,
    isCouponDetailLoading: false,
    coupon: '',
    isCouponError: false
  })
  const setCouponDetails = (args = {}) => setCouponConfig({
    ...couponConfig,
    ...args
  })
  const onClearCoupon = () => {
    setCouponDetails({ couponDetail: null, paramCoupon: '' })
  }
  const onApplyCoupon = (c = '') => {
    if (c?.trim?.() !== '') {
      setCouponDetails({ isCouponDetailLoading: true })
      createPromiseDispatch(setCouponHP, dispatch, { couponCode: c }).then((res) => {
        const { paramCoupon } = couponConfig
        const isCExist = checkCouponExist(res?.data?.CouponDetails || null)
        const getCouponString = () => (`${res?.data?.CouponDetails?.[0]?.coupons?.[0]?.meta_data?.value}${PHRASES.PER_COUPON_APPLIED}`)
        const getParamValues = () => (paramCoupon || getCouponString())
        setCouponDetails({
          coupon: isCExist ? getCouponString()
            : getParamValues(),
          paramCoupon: paramCoupon || null,
          couponDetail: isCExist ? (res?.data?.CouponDetails || null) : null,
          isCouponDetailLoading: false,
          isCouponError: !isCExist
        })
        // coupon is success , this case can be used to show message
        if (isCExist && (res?.data?.CouponDetails || []).length > 0) {
          Keyboard.dismiss()
          const couponDetails = res?.data?.CouponDetails
          const toastMessage = `${PHRASES.COUPON_APPLIED_SUCCESS}\n${PHRASES.AND_U_GOT} ${couponDetails?.[0]?.coupons?.[0]?.meta_data?.value || ''}${PHRASES.PER_OFF}.`
          dispatch(showNewToast(toastMessage))
        }
      }).catch(() => {
        dispatch(showNewToast(ACTION_FAILURE.SOMETHING_WENT_WRONG.MESSAGE))
        setCouponDetails({ isCouponDetailLoading: false })
      })
    }
  }
  const onApply = () => {
    onApplyCoupon(couponConfig.coupon)
  }
 
  const onSetCoupon = (txt) => {
    setCouponDetails({
      coupon: txt,
      paramCoupon: '',
      couponDetail: null,
      isCouponError: false
    })
  }
 
  useEffect(() => {
    if (routeParams.coupon) {
      Promise.resolve(() => {
        onSetCoupon(routeParams.coupon)
      }).then(() => {
        onApplyCoupon(routeParams.coupon)
      })
    }
  }, [routeParams])
 
  const onRemoveCoupon = () => {
    setCouponDetails({
      paramCoupon: '',
      couponDetail: null,
      isCouponDetailLoading: false,
      coupon: '',
      isCouponError: false
    })
  }
 
  const products = useSelector(getPromiseProducts, shallowEqual)
  const insuranceWithHpExp = useSelector(RentProtectSelectors.getInsuranceWithHPExp)
  const insuranceWithHpProductTypeList = useSelector(
    RentProtectSelectors.getInsuranceWithHpTypeList, shallowEqual
  )
 
  const fetchSachetProducts = () => {
    const iscyberVisible = insuranceWithHpExp
    && insuranceWithHpProductTypeList?.includes?.(HP_PKG_PRODUCT_TYPE.PROMISE)
 
    const urlParams = { productType: HP_PKG_PRODUCT_TYPE.PROMISE, cyberInsurance: iscyberVisible }
    createPromiseDispatch(fetchHPProducts, dispatch,
      { urlParams }).then(() => {
      setState({
        isContentError: false,
        isContentShow: true
      })
    }).catch(() => {
      setState({
        isContentShow: false,
        isContentError: true
      })
    })
  }
 
  useEffect(() => {
    if (isEmpty(products)) {
      fetchSachetProducts()
    } else {
      setState({
        isContentError: false,
        isContentShow: true
      })
    }
  }, [])
 
  const getStyle = useCallback(() => ({
    height: !state.isContentShow || state.isContentError ? 0.9 * getHeight() : 0,
    ...styles.loaderContainer
  }), [state])
 
  const onFaqClick = () => {
    onPress()
  }
 
  return (
    <View style={styles.sachetContainer}>
      <View style={styles.main}>
        <HPNudgeTopTag
          image={images.housingPromise}
          isCustom
        />
        <View style={styles.main2}>
          <SimpleText
            text={strings.INTRODUSING}
            type={textTypes.large}
            style={styles.text1Style}
          />
          <SimpleText
            text={strings.HOUSING_PROMISE}
            type={textTypes.large}
            fontType={fontType.medium}
            style={styles.textStyle}
          />
          <GenericText style={styles.text2Style}>
            {strings.FIND_VERYFIED_AVAILABLE}
          </GenericText>
        </View>
      </View>
      <PromiseContainer
        closeModal={closeModal}
        serviceName={serviceName}
        fromNonNpoNudge={fromNonNpoNudge}
        couponDetail={couponConfig.couponDetail}
        coupon={couponConfig?.coupon}
        products={products}
      />
      {!isEmpty(products) ? (
        <ApplyCoupon
          coupon={couponConfig.coupon}
          couponDetail={couponConfig.couponDetail}
          onClearCoupon={onClearCoupon}
          onApply={onApply}
          isLoading={couponConfig.isCouponDetailLoading}
          onChangeCoupon={onSetCoupon}
          onRemoveCoupon={onRemoveCoupon}
          isCouponError={couponConfig.isCouponError}
        />
      ) : null}
      <View style={getStyle()}>
        {(!state.isContentShow && !state.isContentError) && (
          <HPAnimatedLoader
            size={20}
            isDark
          />
        )}
        {state.isContentError && (
          <SimpleText
            text={PHRASES.SOMETHING_WENT_WRONG}
            style={styles.errorColor}
            type={textTypes.large}
          />
        )}
      </View>
      <View style={styles.bottomContainer}>
        <TouchableOpacity
          onPress={onFaqClick}
        >
          <SimpleText
            text={strings.WHAT_IS_HOUSING_PROMISE}
            style={styles.faqsText}
          />
        </TouchableOpacity>
      </View>
    </View>
  )
}
Container.propTypes = {
  closeModal: PropTypes.func,
  onPress: PropTypes.func,
  serviceName: PropTypes.string,
  fromNonNpoNudge: PropTypes.bool
}
Container.defaultProps = {
  closeModal: emptyFunc,
  onPress: emptyFunc,
  serviceName: '',
  fromNonNpoNudge: false
}
 
export default memo(Container)
