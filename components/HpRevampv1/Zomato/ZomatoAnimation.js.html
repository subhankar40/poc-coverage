import { PanGestureHandler } from 'react-native-gesture-handler'
import React, { useEffect } from 'react'
import { shallowEqual, useSelector, useDispatch } from 'react-redux'
import Animated, {
  useAnimatedGestureHandler,
  useAnimatedStyle,
  useSharedValue,
  withTiming,
  withDelay,
  runOnJS
} from 'react-native-reanimated'
import { View } from 'react-native'
import PropTypes from 'prop-types'
import { EA, EC } from 'constants/GAEvents'
import { trackEvent } from 'utils/trackEvents'
import { getHpOffers, getSelectedCityName } from 'views/TenenatMonetisation/selectors'
import isEmpty from 'common/utils/isEmpty'
import { constValues } from 'utils/HpNewHookUtils'
import { saveZomatoSwiped } from 'views/TenenatMonetisation/TMActions'
import {
  strings, screenName, useHP, isOfferExpired
} from '../utils'
import ZomatoOfferLayer1 from './ZomatoOfferLayer1'
import ZomatoOfferLayer2 from './ZomatoOfferLayer2'
import { commonStyles } from './styles'
 
const ZomatoAnimation = (props) => {
  const { isBlocker } = props
  const dispatch = useDispatch()
  const data = useSelector(getHpOffers, shallowEqual)
  const offers = data?.placeholders || []
  const offerValidTo = data?.valid_to
  const selectedCity = useSelector(getSelectedCityName, shallowEqual) || null
  const panX = useSharedValue(0)
  const { state: { isZomatoSeen } } = useHP()
  const [wrapperData] = offers?.filter(item => item.key === strings.WRAPPER)
  const [offerData] = offers?.filter(item => item.key === strings.OFFER_TEXT)
  const [imageUrl] = offers?.filter(item => item.key === strings.IMAGE_URL)
 
  const slideRightAnimation = () => {
    panX.value = withTiming(60, {
      duration: constValues.duration
    })
  }
 
  const slideBeforeAnimation = () => {
    panX.value = withDelay(constValues.animationDelay, withTiming(0, {
      duration: constValues.duration
    }))
  }
 
  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: panX.value }]
  }))
 
  useEffect(() => {
    if (!isEmpty(data) && isZomatoSeen) {
      slideRightAnimation()
      slideBeforeAnimation()
      const payload = {
        offer_name: strings.ZOMATO,
        from_screen: screenName(isBlocker),
        offer_city: selectedCity
      }
      trackEvent(EC.HP_OFFER, EA.OFFER_SEEN, payload)
    }
  }, [isZomatoSeen])
 
  const dispatchAction = saveZomatoSwiped(true)
  const onAnimationFinish = () => {
    dispatch(dispatchAction)
    const payload = {
      offer_name: strings.ZOMATO,
      from_screen: screenName(isBlocker),
      offer_city: selectedCity
    }
    trackEvent(EC.HP_OFFER, EA.OFFER_UNLOCKED, payload)
  }
 
  const handleGestureEventHandler = useAnimatedGestureHandler({
    onStart: (_, ctx) => {
      ctx.startX = panX.value
    },
    onActive: (event, ctx) => {
      if (ctx.startX + event.translationX <= constValues.rightEndLimit
      && ctx.startX + event.translationX >= constValues.leftEndLimit) {
        panX.value = ctx.startX + event.translationX
      }
    },
    onEnd: (event, ctx) => {
      const dx = event.translationX + ctx.startX
      if (Math.abs(dx) < constValues.rightThresholdLimit) {
        panX.value = withTiming(0)
      } else if (dx > constValues.rightThresholdLimit) {
        panX.value = withTiming(constValues.rightEndLimit)
        runOnJS(onAnimationFinish)()
      } else if (dx < constValues.leftThresholdLimit) {
        panX.value = withTiming(constValues.leftEndLimit)
        runOnJS(onAnimationFinish)()
      }
    }
  }, [])
 
  if (isEmpty(data) || isOfferExpired(offerValidTo)) return null
  return (
    <View style={commonStyles.main}>
      <ZomatoOfferLayer2
        data={offerData?.value}
        imageUrl={imageUrl?.value}
        offerValidTo={offerValidTo}
      />
      <PanGestureHandler
        maxPointers={1}
        onGestureEvent={handleGestureEventHandler}
      >
        <Animated.View style={[animatedStyle, commonStyles.absolute]}>
          <ZomatoOfferLayer1
            data={wrapperData?.value}
            offerValidTo={offerValidTo}
          />
        </Animated.View>
      </PanGestureHandler>
    </View>
  )
}
 
ZomatoAnimation.propTypes = {
  isBlocker: PropTypes.bool
}
ZomatoAnimation.defaultProps = {
  isBlocker: false
}
 
export default ZomatoAnimation
