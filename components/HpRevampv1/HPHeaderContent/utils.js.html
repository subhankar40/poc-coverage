const FALLBACK_NAME = 'There'
const getStableUserName = (name) => {
  const trimedName = name?.trim() || ''
  if (!isEmpty(trimedName)) {
    const modifiedName = trimedName.split(' ')?.[0]
    if (validationMap('name')(modifiedName)) return capitalize(modifiedName)
  }
  return FALLBACK_NAME
}
 
const getValidatedSearchCount = (count = 0) => {
  if (count <= 10) {
    return 10
  } if (count > 999) {
    return (formatNumber(count, 1))
  }
  return count
}
 
const useRecentCityLocality = () => {
  const data = useSelector(getRecentSearchData, shallowEqual)
  const city = data?.selectedCity || ''
  const locality = data?.recentSearch?.base?.[0]?.label || ''
  const lowerCaseLocality = locality?.toLowerCase?.() || ''
  let localityCity
  if (isEmpty(locality) && !isEmpty(city)) {
    localityCity = city
  } else
  if (lowerCaseLocality?.includes(city?.toLowerCase?.())) {
    localityCity = locality
  } else {
    localityCity = `${locality}, ${city}`
  }
  const values = useMemo(() => ({ localityCity }), [localityCity])
  return values
}
 
const useRecentLocalityCityAndUserName = () => {
  const cityLocality = useRecentCityLocality() || {}
  const user = useSelector(getUserData, shallowEqual)
  const values = useMemo(() => ({
    ...cityLocality,
    name: user?.userData?.data?.name
  }), [cityLocality, user])
  return (values)
}

export const usePgServiceHeaderContent = () => {
  /**
     * This is specific to PG service header
     */
  const user = useSelector(getUserData, shallowEqual)
  const isLogin = useSelector(getIsLoggedIn)
  const { name } = useRecentLocalityCityAndUserName()
  const handledName = getStableUserName(name)
  const userTitle = userNameHeading(isLogin, user?.userData?.data?.name, true)
  const values = useMemo(() => ({ userTitle }), [userTitle])
  const { state: { couponMetaInfo } } = useHP()
  const getCouponValue = () => (couponMetaInfo?.[0]?.coupons?.[0]?.meta_data?.value)
  const updatedHeader = useMemo(() => ({ userTitle: replaceAll(HP_HEADER_HEADINGS.COUPON_HEADING, { '<name>': handledName, '<value>': getCouponValue() }) }), [couponMetaInfo])
 
  return couponMetaInfo ? updatedHeader : values
}
 
export const useHpBlockerHeaderContent = (searchCount) => {
/**
 * This is specific to Blocker header
 */
  const { localityCity, name } = useRecentLocalityCityAndUserName()
  const handledName = getStableUserName(name)
  const isDefault = Number(searchCount) === 0
  const { state: { couponMetaInfo } } = useHP()
  const getCouponValue = () => (couponMetaInfo?.[0]?.coupons?.[0]?.meta_data?.value)
 
  const getMainText = () => {
    if (couponMetaInfo) return replaceAll(HP_HEADER_HEADINGS.COUPON_HEADING, { '<name>': handledName, '<value>': getCouponValue() })
    return replaceAll(HP_HEADER_HEADINGS.MAIN_BLOCKER_TOP_TITLE, { '<name>': handledName })
  }
  const main = getMainText()
 
  const secondary = isDefault ? HP_HEADER_HEADINGS.SECONDARY_FALLBACK
    : replaceAll(HP_HEADER_HEADINGS.SECONDARY_BLOCKER_TITLE, {
      '<owner>': `${getValidatedSearchCount(searchCount)}`,
      '<locality-area>': localityCity
    })
  const values = useMemo(() => ({
    main,
    secondary,
    highlightWords: couponMetaInfo ? [replaceAll(HP_HEADER_HEADINGS.COUPON_HEADING, { '<name>': handledName, '<value>': getCouponValue() })] : [`Hey ${handledName}`]
  }), [main, secondary, searchCount])
  return (values)
}
 
export const useHPLandingHeaderContent = (searchCount) => {
  /**
     * This is specific to Landing page heder content
     */
  const { localityCity, name } = useRecentLocalityCityAndUserName()
  const handledName = getStableUserName(name)
  const isDefault = searchCount === 0
 
  const defaultText = HP_HEADER_HEADINGS.MAIN_LANDING_FALLBACK_TITLE
  const mainText = HP_HEADER_HEADINGS.MAIN_LANDING_TOP_TITLE
  const { state: { couponMetaInfo } } = useHP()
  const getCouponValue = () => (couponMetaInfo?.[0]?.coupons?.[0]?.meta_data?.value)
 
  const getMainText = () => {
    if (couponMetaInfo) return replaceAll(HP_HEADER_HEADINGS.COUPON_HEADING, { '<name>': handledName, '<value>': getCouponValue() })
    if (isDefault) return defaultText
    return replaceAll(mainText, { '<name>': handledName, '<owner>': getValidatedSearchCount(searchCount), '<locality-area>': localityCity })
  }
  const main = getMainText()
 
  const getHighlightedText = () => {
    if (couponMetaInfo) {
      return [replaceAll(HP_HEADER_HEADINGS.COUPON_HEADING, { '<name>': handledName, '<value>': getCouponValue() })]
    }
    if (!isDefault) return [`Hey ${handledName}`]
    return []
  }
  const values = useMemo(() => ({
    main,
    highlightWords: getHighlightedText()
  }), [main, searchCount])
  return (values)
}

const HEADER_PLACE_HOLDER_DEFAULT_VALUES = {
  '<name>': 'Hey',
  '<total_owner_properties>': 'many',
  '<locality>': 'your locality'
}
 
const HEADER_PLACE_HOLDERS = {
  '<name>': '',
  '<total_owner_properties>': '',
  '<locality>': ''
}
 
const useLandingPaywallHeaderPlaceHolders = (searchCount) => {
  const cityAndName = useRecentLocalityCityAndUserName()
  const config = useMemo(() => {
    const modifiedValues = {
      ...HEADER_PLACE_HOLDERS
    }
    if (searchCount) {
      modifiedValues['<total_owner_properties>'] = (searchCount
        || HEADER_PLACE_HOLDER_DEFAULT_VALUES['<total_owner_properties>'])
    }
    if (cityAndName) {
      modifiedValues['<name>'] = (cityAndName?.name || HEADER_PLACE_HOLDER_DEFAULT_VALUES['<name>'])
      modifiedValues['<locality>'] = (cityAndName?.localityCity || HEADER_PLACE_HOLDER_DEFAULT_VALUES['<locality>'])
    }
    return modifiedValues
  }, [searchCount, cityAndName])
  return config
}
 
export const useLandingPageSlidesData = (searchCount, serviceIndex) => {
  const lpHeroCopyConfig = useSelector(getHpLandingHeroCopy, shallowEqual)
  const headerPlaceholders = useLandingPaywallHeaderPlaceHolders(searchCount)
 
  const lpHeroCopy = useMemo(() => {
    const heroCopyList = lpHeroCopyConfig?.[serviceIndex === 1 ? 'rent' : 'buy'] || []
    return (heroCopyList.map(slide => ({
      ...slide,
      subheader: replaceAll(slide?.subheader || '', headerPlaceholders)
    })))
  }, [serviceIndex, lpHeroCopyConfig, headerPlaceholders])
  return lpHeroCopy
}

    return ({
      ...heroCopyData,

  return heroCopy
}
