const TimerSwipe = (props) => {
  const {
    seIsHighLightedView, newDiscountApplied, revampDiscount, setTopCardLayoutConfig
  } = props
  const [animationFinished, setAnimationState] = useState(false)
  const [animationStart, setAnimationStart] = useState(false)
  const panX = useSharedValue(0)
  const gamificationData = useSelector(isGamificationDiscount, shallowEqual)
  const { scrollYOffset, getStickyViewByTagName } = useScrollViewConfig()
  const scrollY = useDerivedValue(() => scrollYOffset.value, [scrollYOffset])
 
  const {
    state: {
      couponMetaInfo,
      paramCoupon,
      isMlDiscountApplied
    }
  } = useHP()
 
  const isExternalDiscountApplied = paramCoupon || couponMetaInfo || isMlDiscountApplied
 
  const slideRightAnimation = () => {
    panX.value = withRepeat(withTiming(15, {
      duration: 600
    }), -1, true)
  }
  const CONTAINER_WIDTH = getWindowWith()
  const constValues = {
    rightThresholdLimit: CONTAINER_WIDTH / 5,
    leftThresholdLimit: -1 * CONTAINER_WIDTH / 5,
    rightEndLimit: CONTAINER_WIDTH - 48 - 60 - 10,
    leftEndLimit: -1 * CONTAINER_WIDTH,
    panGestureHeartSlideAnimation: 400,
    duration: 600,
    animationDelay: 800
  }
  const onAnimationFinish = (val) => {
    setAnimationState(true)
    seIsHighLightedView(true)
    getStickyViewByTagName?.('n1').then((d) => {
      setTopCardLayoutConfig({
        scrollY: val,
        initialY: d?.y || 0
      })
    }).catch(() => {
      setTopCardLayoutConfig({
        scrollY: val,
        initialY: 0
      })
    })
  }
 
  const widthStyle = useAnimatedStyle(() => ({
    width: panX.value > 20 ? panX.value + 50 : 0
  }))
 
  const slideAnimation = () => {
    panX.value = withTiming(constValues.rightEndLimit, {
      duration: 300
    }, () => {
      runOnJS(onAnimationFinish)(scrollY?.value)
    })
  }
 
  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: panX.value }]
  }))
 
  useEffect(() => {
    slideRightAnimation()
  }, [])
 
  const stateSet = () => {
    setAnimationStart(true)
  }
 
  useEffect(() => {
    if (animationStart) {
      slideAnimation()
    }
  }, [animationStart])
 
  const handleGestureEventHandler = useAnimatedGestureHandler({
    onStart: (_, ctx) => {
      ctx.startX = panX.value
    },
    onActive: (event, ctx) => {
      if (ctx.startX + event.translationX <= constValues.rightEndLimit
        && event.translationX > 0) {
        panX.value = ctx.startX + event.translationX
      }
    },
    onEnd: (event, ctx) => {
      const dx = event.translationX + ctx.startX
      if (dx > constValues.rightThresholdLimit) {
        runOnJS(stateSet)()
      } else {
        panX.value = withTiming(0)
      }
    }
  }, [])
 
  if (newDiscountApplied || isExternalDiscountApplied
    || gamificationData?.offerType === DISCOUNT_TYPE.PERCENT_DISCOUNT) {
    return <View style={styles.emptyView} />
  }
 
  return (
    <View style={styles.main}>
      <LinearGradientBorderView
        gradientProps={gradConfig}
        style={styles.borderGrad}
      >
        <LinearGradient
          style={styles.linearGrad1}
          colors={GRADIENT_COLORS.NEW_TIMER4}
          start={GRADIENT_CONST.XY_13_1}
          end={GRADIENT_CONST.XY_00}
        >
          <View style={styles.flexDirection}>
            <View style={styles.width} />
            <View style={styles.flex1}>
              <HpFomoTextView
                title={revampDiscount?.text || ''}
                style={styles.text1}
                maskStyle={styles.maskView}
                colors={GRADIENT_COLORS.NEW_TIMER2}
                angle={0}
                shimmerWidth={250}
                duration={2000}
              />
            </View>
          </View>
          <Animated.View style={[styles.animatedView1, widthStyle]}>
            <LinearGradient
              style={styles.linearGrad2}
              colors={GRADIENT_COLORS.NEW_TIMER5}
              start={GRADIENT_CONST.XY_00}
              end={GRADIENT_CONST.XY_11}
            >
              <Text
                style={styles.text4}
                numberOfLines={1}
              >
                {(revampDiscount?.sub_text) || ''}
              </Text>
            </LinearGradient>
          </Animated.View>
          <PanGestureHandler
            maxPointers={1}
            onGestureEvent={handleGestureEventHandler}
          >
            <Animated.View style={[styles.animated2, animatedStyle]}>
              {!animationFinished && (
                <LinearGradient
                  style={styles.linearGrad3}
                  colors={GRADIENT_COLORS.NEW_TIMER6}
                  start={GRADIENT_CONST.XY_01}
                  end={GRADIENT_CONST.XY_00}
                >
                  <View>
                    <SimpleText
                      text={`${revampDiscount?.value || ''}%`}
                      style={styles.text2}
                      fontType={fontType.medium}
                      type={textTypes.large}
                    />
                    <View style={styles.flexDirection}>
                      <View style={styles.dot} />
                      <SimpleText
                        text={strings.OFF}
                        style={styles.text3}
                        fontType={fontType.medium}
                        type={textTypes.smaller}
                      />
                      <View style={styles.dot} />
                    </View>
                  </View>
                  <View style={styles.arrowContainer}>
                    <ImageComponent
                      source={images.rightArrowIcon}
                      style={styles.image1}
                      tintColor={colors.neon_pink}
                    />
                    <ImageComponent
                      source={images.rightArrowIcon}
                      style={styles.image2}
                      tintColor={colors.neon_pink}
                    />
                  </View>
                </LinearGradient>
              )}
            </Animated.View>
          </PanGestureHandler>
        </LinearGradient>
      </LinearGradientBorderView>
    </View>
  )
}
