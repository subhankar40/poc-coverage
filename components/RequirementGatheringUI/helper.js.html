export const BuyAdvanceFilter = (props) => {
  const { isDark } = props
  const {
    state, setState, onFilterHandler, appliedFilters, validationState
  } = useRequirementGath()
  const [isAdvanceFilterClicked, setAdvanceFilterClicked] = useState(false)
  const brands = useSelector(homepageBrands, shallowEqual) || []
 
  const sliderFilterChangeCallbacks = (filterConfig, appliedFiltersState) => {
    setState(filterConfig, appliedFiltersState)
  }
  const toggleImageFilter = (val) => {
    val ? trackEvent(EC.HP_PSP, EA.WHATSAPP_TOGGLE, { TOGGLE_STATUS: EL.ON })
      : trackEvent(EC.HP_PSP, EA.WHATSAPP_TOGGLE, { TOGGLE_STATUS: EL.OFF })
    setState({ whatsApp: val })
  }
 
  useEffect(() => {
    state?.whatsApp ? trackEvent(EC.HP_PSP, EA.WHATSAPP_TOGGLE, { TOGGLE_STATUS: EL.ON })
      : trackEvent(EC.HP_PSP, EA.WHATSAPP_TOGGLE, { TOGGLE_STATUS: EL.OFF })
  }, [])
 
  const renderBuyFilters = () => (
    <BuyFilters
      filterConfig={state}
      onChange={onFilterHandler}
      sliderFilterChangeCallbacks={sliderFilterChangeCallbacks}
      appliedFilters={appliedFilters}
      edge
      showAdvanceEdge={isAdvanceFilterClicked}
      validationState={validationState}
      isDark={isDark}
    />
  )
  const renderRentFilters = () => (
    <RentFilters
      filterConfig={state}
      onChange={onFilterHandler}
      sliderFilterChangeCallbacks={sliderFilterChangeCallbacks}
      appliedFilters={appliedFilters}
      edge
      showAdvanceEdge={isAdvanceFilterClicked}
      validationState={validationState}
      isDark={isDark}
    />
  )
  const renderPgFilters = () => (
    <PGFilters
      filterConfig={state}
      onChange={onFilterHandler}
      sliderFilterChangeCallbacks={sliderFilterChangeCallbacks}
      appliedFilters={appliedFilters}
      validationState={validationState}
      brands={brands}
      isDark={isDark}
    />
  )
 
  const onClickAdvanceFilter = (click) => {
    setAdvanceFilterClicked(!click)
  }
 
  const filters = (serviceName) => {
    switch (serviceName) {
      case SERVICE_TYPE.BUY_LC:
        return renderBuyFilters()
      case SERVICE_TYPE.RENT_LC:
        return renderRentFilters()
      case SERVICE_TYPE.PG_LC:
        return renderPgFilters()
      default:
        return null
    }
  }
  const renderContent = () => (
    <View>
      {filters(state?.serviceName)}
      <View style={style.margin}>
        {
          showAdvanceFilter(state?.serviceName) && (
            <RenderFooter
              isDark
              collapsText={HIDE_ADVANCED_FILTERS}
              expandText={SHOW_ADVANCED_FILTERS}
              onChange={onClickAdvanceFilter}
              isAdvanceFilterClicked={isAdvanceFilterClicked}
            />
          )
        }
      </View>
      {whatsAppFilter(state.whatsApp, toggleImageFilter, isDark)}
    </View>
  )
  return (
    <View>
      {renderContent()}
    </View>
  )
}
 
export const CustomTime = (props) => {
  const { time, setTime, isDark } = props
  const [modalVisible, setModalVisible] = useState(false)
  const widthValue = boxWidth(2)
  let sublabel = null
  const parseStartData = parseInt(time.startTime, 10)
  const parseEndData = parseInt(time.endTime, 10)
  const timeString = () => {
    let string = null
    if (time.startTime !== null && time.endTime !== null) {
      string = `${timeFilter(parseStartData) }-${ timeFilter(parseEndData)}`
    }
    return string
  }
  sublabel = timeString()
  const openDatePicker = () => (
    <View>
      <RequirementDatePikker
        onDateSelect={setModalVisible}
        timeSelect={setTime}
        time={time}
      />
    </View>
  )
  const checkISselected = () => {
    if (parseStartData || parseEndData) {
      return true
    }
    return false
  }
  const onPress = () => {
    setModalVisible(!modalVisible)
  }
  const isSelected = checkISselected()
 
  const mainStyle = useMemo(() => {
    let rst = { ...styles.boxCommonStyle, width: widthValue }
    if (isDark) {
      rst = isSelected ? { ...rst, ...styles.darkSelected } : { ...rst, ...styles.darkNotSelected }
    } else {
      rst = isSelected ? { ...rst, ...styles.selected } : { ...rst, ...styles.notSelected }
    }
    return rst
  }, [isSelected, isDark, widthValue])
 
  const textStyle = useMemo(() => {
    if (isDark) {
      return (isSelected ? styles.darkSelectedText : styles.darkUnSelectdText)
    }
    return isSelected ? styles.purpleText : styles.text
  }, [isDark, isSelected])
 
  return (
    <Pressable onPress={onPress} style={mainStyle}>
      <SimpleText
        text={CUSTOM_TIME}
        type={textTypes.regular}
        style={textStyle}
      />
      {modalVisible ? openDatePicker() : null}
      {sublabel
        ? (
          <SimpleText
            text={sublabel}
            type={textTypes.regular}
            style={textStyle}
          />
        ) : null
      }
    </Pressable>
  )
}

export const languageModal = (onChange, selectedId, languageData) => {
  const renderListItem = (data) => {
    const { item } = data
    const hide = () => {
      hideModal()
      onChange(item.id)
    }
    const textStyle = selectedId === item.id ? style.LanguageItemTextPurple : style.LanguageItemText
    return (
      <TouchableOpacity onPress={hide} style={style.LanguageItemheight}>
        <SimpleText
          text={item.displayName}
          type={textTypes.large}
          style={textStyle}
        />
      </TouchableOpacity>
    )
  }
  const seperator = () => (
    <View style={[style.seperator, style.listSeperatorMargin]} />
  )
  const errorMessage = () => (
    <View>
      <SimpleText
        text={SOMETHING_WENT_WRONG}
        style={style.languageErrorMessage}
      />
    </View>
  )
  const renderList = () => (
    <View style={style.languageModalHeight}>
      <View style={style.modalHeader}>
        <SimpleText
          text={SELECTED_LANGUAGE}
          type={textTypes.larger}
          style={style.title}
        />
      </View>
      <View style={style.seperator} />
      <View style={style.flex1}>
        {
          !isEmpty(languageData)
            ? (
              <FlatList
                data={languageData}
                showsVerticalScrollIndicator={false}
                renderItem={renderListItem}
                ItemSeparatorComponent={seperator}
                ListFooterComponent={seperator}
              />
            )
            : errorMessage()
        }
      </View>
    </View>
  )
  return (
    <View>
      {showModal({
        renderedComponent: () => (
          renderList()
        ),
        hideModalOnTouchOutside: true
      })}
    </View>
  )
}
 
export const OtherLanguage = (props) => {
  const {
    setLanguage, selectedLanguage, languageData, isDark
  } = props
 
  const topTitleStyle = useMemo(() => {
    let rst = {
      ...fontUtils.getFontStyle('400'),
      fontSize: 11,
      color: undefined,
      opacity: 1
    }
    if (isDark) {
      rst = { ...rst, color: colors.violet45 }
    } else {
      rst = { ...rst, opacity: 0.6 }
    }
    return rst
  }, [isDark])
  const titleStyle = useMemo(() => {
    let rst = {
      ...fontUtils.getFontStyle('400'),
      fontSize: 14,
      color: undefined
    }
    if (isDark) {
      rst = {
        ...rst,
        color: colors.violet45
      }
    }
    return rst
  }, [isDark])
 
  const separatorStyle = useMemo(() => {
    let rst = { ...style.seperator }
    if (isDark) {
      rst = {
        backgroundColor: colors.violet45,
        height: 1
      }
    }
    return rst
  }, [isDark])
 
  const languageText = () => {
    let displayName = null
    if (!isEmpty(languageData)) {
      const otherLang = languageData.filter(item => selectedLanguage === item.id)
      displayName = otherLang[0]?.displayName
    }
    return (
      <View>
        <SimpleText
          text={displayName}
          type={textTypes.large}
          style={titleStyle}
        />
        <View style={separatorStyle} />
      </View>
    )
  }
 
  return (
    <View style={style.otherLanguageContainer}>
      <Pressable onPress={() => {
        languageModal(setLanguage, selectedLanguage, languageData)
      }}
      >
        <SimpleText
          text={OTHER_LANGUAGE}
          type={textTypes.small}
          style={topTitleStyle}
        />
        {languageText()}
      </Pressable>
    </View>
  )
}
