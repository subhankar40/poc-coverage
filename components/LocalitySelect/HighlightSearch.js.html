  <ImageComponent testID="searchIcon" source={images.search} style={style} />

const renderPricePerUnit = (price, renderNewUI, unitOfArea = null) => {
  let unit = ''
  if (price !== PRICE_ON_REQUEST && price !== PRICE_NOT_AVAILABLE) {
    unit = price.unit === 'bedroom' ? 'bedroom' : unitOfArea || SQ_FT
  }
  let priceValue = null
  if (price && price.value) {
    const priceWithoutSymbol = removeRupeeSymbolFromAmount(price.value)
    const formatedPrice = formatPrice(priceWithoutSymbol, {
      showUnit: true,
      noCurrency: false,
      useUnitCharacter: true
    })
    priceValue = formatedPrice && formatedPrice.toLocaleLowerCase().replace(' ', '')
  } else priceValue = unit ? `${RUPEE_SYMBOL}${price}` : price
 
  return (
    <GenericText
      font={renderNewUI ? fontUtils.appFont : fontUtils.defaultFont}
      style={styles.priceText}
      size={12}
      lineHeight={14}
    >
      {`${priceValue} ${unit ? '/' : ''}${unit}`}
    </GenericText>
  )
}
 
const renderLocalityName = (locality, renderNewUI) => {
  const localityName = locality.join(', ')
  const lineHeight = renderNewUI ? 16 : 18
  return (
    <GenericText
      lineHeight={lineHeight}
      style={styles.localityName}
      font={renderNewUI ? fontUtils.appFont : fontUtils.defaultFont}
    >
      {localityName}
    </GenericText>
  )
}
 
export default class HighlightSearch extends PureComponent {
  handleOnPress = () => {
    const {
      dispatch, saveLocalityAndCity, isAnimated, onDidSelect, name, uuid,
      citySelectUUID, type, superType, index, search, showSerp, fromCitySelectScreen,
      fireDidSelectMethod
    } = this.props
    const payload = {
      type, superType, name, uuid, citySelectUUID, position: index, search, showSerp
    }
    if (fromCitySelectScreen || fireDidSelectMethod) {
      onDidSelect(payload)
    } else if (!isMethod(saveLocalityAndCity) || isAnimated) {
      !isAnimated && dispatch(showRootLoader())
      onDidSelect(payload)
    } else {
      saveLocalityAndCity(payload)
    }
  }
 
  getHeight = (params) => {
    const { isLocalityPresent, showVariantAStyle, showUpdatedUI } = params
    let height = 47
    if (isLocalityPresent) height = 60
    else if (showUpdatedUI) height = showVariantAStyle ? 50 : 42
    return height
  }
 
  getPadding = (fromCitySelectScreen, showUpdatedUI) => {
    let paddingH = 25
    if (fromCitySelectScreen) paddingH = 0
    else if (showUpdatedUI) paddingH = 24
    return paddingH
  }
 
  render() {
    const {
      text,
      search,
      type,
      superType,
      showSearchIcon,
      isAnimated,
      animationDelay,
      price,
      shouldShowPricePerSqFt = false,
      locality,
      showUpdatedUI,
      unitOfArea,
      showAddButton,
      showVariantAStyle,
      index,
      fromCitySelectScreen
    } = this.props
    const isLocalityPresent = !isEmpty(locality)
    const fontStyle = showVariantAStyle ? fontUtils.appFontMedium : fontUtils.appFont
    const height = this.getHeight({ isLocalityPresent, showVariantAStyle, showUpdatedUI })
    const testID = `filter_localitySelect_${index}`
    const paddingHorizontal = this.getPadding(fromCitySelectScreen, showUpdatedUI)
    return (
      <AnimatedView
        isAnimated={showVariantAStyle ? true : isAnimated}
        animationDelay={animationDelay}
      >
        <TouchableOpacity
          activeOpacity={1}
          {...{ [resourceID]: testID }}
          onPress={this.handleOnPress}
          hitSlop={styles.hitSlop}
        >
          <View style={{ paddingHorizontal }}>
            <View style={[styles.resultWrapper, { height }]}>
              <View style={styles.resultRow}>
                {showSearchIcon && renderSearchIcon(styles.searchIcon)}
                {showAddButton && (
                  <ImageComponent source={images.localityAddIcon} style={styles.localityAddIcon} />
                )}
                {!isEmpty(text) && (
                  <GenericText
                    size={showUpdatedUI ? 14 : 16}
                    lineHeight={showUpdatedUI ? 16 : 18}
                    color={showUpdatedUI ? colors.grey20 : colors.black54}
                    numberOfLines={1}
                    ellipsizeMode="tail"
                    font={showUpdatedUI ? fontStyle : fontUtils.defaultFont}
                    marginLeft={showSearchIcon ? 2 : 0}
                    style={styles.flex_3}
                  >
                    {Children.toArray(transformSearchResults(trim(head(split(',', text))), search))}
                  </GenericText>
                )}
                {shouldShowPricePerSqFt && price
                  ? renderPricePerUnit(price, showUpdatedUI, unitOfArea)
                  : (
                    <GenericText
                      size={showUpdatedUI ? 12 : 11}
                      color={showUpdatedUI ? colors.grey5 : colors.black26}
                      align="right"
                      font={showUpdatedUI ? fontUtils.appFont : fontUtils.defaultFont}
                      style={styles.flexContainer}
                    >
                      {toUpper(getType(type, superType))}
                    </GenericText>
                  )}
              </View>
              {isLocalityPresent && renderLocalityName(locality, showUpdatedUI)}
              {!isEmpty(text) && notEmpty(trim(join(',', tail(split(',', text)))))
                  && (
                    <View style={styles.resultRow}>
                      <GenericText
                        size={showUpdatedUI ? 12 : 14}
                        lineHeight={showUpdatedUI ? 14 : 20}
                        color={colors.black54}
                        numberOfLines={1}
                        ellipsizeMode="tail"
                        marginLeft={showSearchIcon ? 18 : 0}
                        style={styles.resultRowText}
                        font={showUpdatedUI ? fontUtils.appFont : fontUtils.defaultFont}
                      >
                        {Children.toArray(
                          transformSearchResults(trim(join(',', tail(split(',', text)))), search)
                        )}
                      </GenericText>
                    </View>
                  )
              }
            </View>
          </View>
        </TouchableOpacity>
      </AnimatedView>
    )
  }
}
