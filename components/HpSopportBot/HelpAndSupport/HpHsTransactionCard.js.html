import React from 'react'
import { StyleSheet, View, TouchableOpacity } from 'react-native'
import isEmpty from 'common/utils/isEmpty'
import PropTypes from 'prop-types'
import { SYMBOLS, TRANSACTION_INFO_STRINGS } from 'constants/StringConstants'
import { payRentStrings } from 'constants/strings'
import { fontType, SimpleText } from 'common/components/SimpleText'
import { getDayMonthDateYearTime } from 'common/utils/date'
import colors from 'utils/colors'
import fontUtils from 'common/utils/fontUtils'
import { numberWithCommas } from 'common/utils/formatePrice'
import GradientWrapper from 'common/components/commonComponents/GradientWrapper'
import { COLUMN_GRADIENT } from 'components/HpRevampv1/utils'
import { emptyFunc } from 'common/utils/ValidationUtils'
import { STRINGS } from 'constants/HpNewHookConstant'
import { SERVICE_TYPE } from 'common/constants/commonTypes'
 
const transactionColor = (status) => {
  switch (status) {
    case TRANSACTION_INFO_STRINGS.PENDING:
      return colors.orange4
    case TRANSACTION_INFO_STRINGS.ACTIVE:
      return colors.green_36
    case TRANSACTION_INFO_STRINGS.EXPIRED:
      return colors.redEA
    default:
      return colors.green_36
  }
}
 
const getProductName = (serviceType) => {
  switch (serviceType) {
    case SERVICE_TYPE.RENT_LC:
      return SERVICE_TYPE.RENT_PC
    case SERVICE_TYPE.BUY_LC:
      return SERVICE_TYPE.BUY_PC
    case SERVICE_TYPE.PG_LC:
      return SERVICE_TYPE.PG_UC
    default:
      return SERVICE_TYPE.RENT_PC
  }
}
 
const conditionalStyles = ({ chatBot, statusColor }) => StyleSheet.create({
  container: {
    padding: 8,
    marginHorizontal: 10,
    marginTop: 8,
    borderRadius: 12
  },
  innerContainer: {
    flexDirection: 'row',
    justifyContent: 'space-between'
  },
  paidRentTo: {
    color: colors.white60,
    opacity: chatBot ? 1 : 0.8
  },
  name: {
    color: colors.white,
    marginTop: chatBot ? 0 : 4
  },
  whiteColor: {
    color: colors.white
  },
  dateText: {
    color: colors.white60,
    marginTop: chatBot ? 8 : 4
  },
  transactionText: {
    color: colors.white60,
    marginTop: 2
  },
  paymentStatusText: {
    color: chatBot ? statusColor : (statusColor || colors.black),
    letterSpacing: 0.7,
    marginTop: 4,
    fontFamily: chatBot ? fontUtils.appFontMedium : fontUtils.appFont
  }
})
 
const color = [colors.dark_grey06, colors.dark_grey3, colors.periwinkle_blue01]
 
const HpHsTransactionCard = (props) => {
  const {
    item, onPress, containerStyle, chatBot
  } = props
 
  const productName = isEmpty(item?.productName) ? null : `${item?.productName} ${STRINGS.PLAN} [ ${getProductName(item?.serviceType || '')} ]`
  const date = item?.paymentDate ? getDayMonthDateYearTime(item?.paymentDate) : null
  const transactionId = isEmpty(item?.orderId) ? '' : item?.orderId
  const totalAmt = isEmpty(item?.amount) ? '' : `${SYMBOLS.RUPEE_SYMBOL} ${numberWithCommas(item?.amount || '')}`
 
  const styles = conditionalStyles({ chatBot, statusColor: transactionColor(item?.status) })
  return (
    <TouchableOpacity
      onPress={onPress}
      activeOpacity={1}
    >
      <GradientWrapper
        style={[styles.container, containerStyle]}
        colors={chatBot ? COLUMN_GRADIENT.MODAL_COLORS : color}
        vertical={false}
      >
        <View style={styles.innerContainer}>
          <View>
            {!chatBot && <SimpleText text={STRINGS.AMOUNT_PAID} style={styles.paidRentTo} />}
            {productName && (
              <SimpleText
                text={productName}
                style={styles.name}
                fontType={fontType.medium}
              />
            )}
          </View>
          <SimpleText text={totalAmt} fontType={fontType.medium} style={styles.whiteColor} />
        </View>
        <SimpleText text={date} style={styles.dateText} />
        <SimpleText text={`${payRentStrings.transactionId} - ${transactionId}`} style={styles.transactionText} />
        <SimpleText text={item?.status || ''} style={styles.paymentStatusText} />
      </GradientWrapper>
    </TouchableOpacity>
  )
}
 
HpHsTransactionCard.defaultProps = {
  onPress: emptyFunc,
  containerStyle: {},
  chatBot: false
}
 
HpHsTransactionCard.propTypes = {
  item: PropTypes.object.isRequired,
  onPress: PropTypes.func,
  containerStyle: PropTypes.oneOfType([PropTypes.object, PropTypes.array]),
  chatBot: PropTypes.bool
}
 
export default HpHsTransactionCard
