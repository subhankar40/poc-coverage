import React from 'react'
import {
  View, Animated, PanResponder, BackHandler, Keyboard, KeyboardAvoidingView,
  StyleSheet
} from 'react-native'
import { SafeAreaView, SafeAreaProvider } from 'react-native-safe-area-context'
import { getHeight } from 'common/utils/deviceDimensions'
import { isAndroid, isIos } from 'utils/TypeChecker'
import { popModalRef } from 'utils/AllRefs'
import isEmpty from 'common/utils/isEmpty'
 
import isNil from 'ramda/src/isNil'
import { hideShowTab, hideShowTabOnModalOpen } from 'utils/hideShowTab'
import styles from './styles'
 
const DefaultComponent = () => <React.Fragment />
const panReleaseOffset = isAndroid ? 0 : 151
 
export class ModalBox extends React.Component {
  constructor(props) {
    super(props)
    this.headerComponent = null
    this.backgroundTopComponent = null
    this.panGestureHeaderStyles = {}
    this.modalHeaderComponent = null
    this.state = {
      showModalBox: false,
      sealModalBox: false,
      modalContainerStyles: {}
    }
    this.init()
    this.onClose = null
    this.showFullScreen = false
    this.hideBottomTab = null
  }
 
  componentWillUnmount() {
    popModalRef()
    this.backPressListener?.remove()
  }
 
  isModalBoxOpen = () => this.state.showModalBox
 
  init = () => {
    this.animatedValue = new Animated.Value(getHeight())
    this.backgroundOpacity = this.animatedValue.interpolate({
      inputRange: [0, getHeight()],
      outputRange: [0.9, 0]
    })
    this.gestureResponser = this.createPanGeatureResponder()
    this.renderedComponent = DefaultComponent
    this.hideModalOnTouchOutside = true
    this.showDarkTheme = false
  }
 
  createPanGeatureResponder = () => (
    PanResponder.create({
      onStartShouldSetPanResponder: () => true,
      onPanResponderMove: (_, gestureState) => {
        if (gestureState.dy > 0) {
          this.animatedValue.setValue(gestureState.dy)
        }
      },
      onPanResponderRelease: (_, gestureState) => {
        this.onRelease(gestureState.dy)
      },
      onPanResponderTerminate: () => {
        this.onRelease(panReleaseOffset)
      },
      onShouldBlockNativeResponder: () => true
    })
  )
 
  onRelease = (y) => {
    if (y > 150) {
      this.closeModal(true)
    } else {
      this.animate(0)
    }
  }
 
  animate = (toValue, callback = () => { }) => {
    Animated.spring(this.animatedValue, {
      toValue,
      useNativeDriver: true,
      bounciness: 2
    }).start(() => { callback() })
  }
 
  openModal = ({
    renderedComponent = DefaultComponent, hideModalOnTouchOutside = true,
    onModalClose, headerComponent = null, showDarkTheme, showFullScreen,
    modalHeaderComponent = null, backgroundTopComponent = null, customBackgroundStyles = {},
    disableOutsideTouch = false, disableOutsideTouchAfterAnimationComplete = false,
    isHeaderOwerFlow = false, disableSafeArea = false, hideBottomTab, hideNativeTabBar = false,
    panGestureHeaderStyles = {}
  }) => {
    this.disableSafeArea = disableSafeArea
    this.showFullScreen = showFullScreen
    this.renderedComponent = renderedComponent
    this.hideModalOnTouchOutside = hideModalOnTouchOutside
    this.headerComponent = headerComponent
    this.showDarkTheme = showDarkTheme
    this.backgroundTopComponent = backgroundTopComponent
    this.customBackgroundStyles = customBackgroundStyles
    this.disableOutsideTouch = disableOutsideTouch
    this.onClose = onModalClose
    this.modalHeaderComponent = modalHeaderComponent
    this.hideBottomTab = hideBottomTab
    this.hideNativeTabBar = hideNativeTabBar
    this.panGestureHeaderStyles = panGestureHeaderStyles
    this.isHeaderOwerFlow = isHeaderOwerFlow
 
    this.backPressListener = BackHandler.addEventListener('hardwareBackPress', this.handleHardwareBackPress)
    if (hideNativeTabBar && isAndroid) {
      hideShowTab(true)
    } else {
      hideShowTabOnModalOpen(true)
    }
    this.setState({ showModalBox: true })
    const callback = () => {
      if (disableOutsideTouchAfterAnimationComplete) {
        this.disableOutsideTouch = false
      }
    }
    this.animate(0, callback)
  }
 
  closeModal = (isModalClosedExt = false) => {
    if (this.hideNativeTabBar) {
      hideShowTab(false)
    }
    this.hideBottomTab()
    Keyboard.dismiss()
    hideShowTabOnModalOpen(false)
    this.animate(getHeight(), () => {
      this.forceCloseModal(isModalClosedExt)
      if (this.state.sealModalBox) this.unsealedModalBox()
    })
  }
 
  forceCloseModal = (isModalClosedExt = false) => {
    this.backPressListener?.remove()
    this.setState({ showModalBox: false }, () => {
      if (this.onClose) {
        this.onClose(isModalClosedExt)
      }
    })
  }
 
  setModalContainerStyles = (styles) => {
    this.setState({ modalContainerStyles: styles })
  }
 
  sealedModalBox = () => {
    this.setState({ sealModalBox: true })
  }
 
  unsealedModalBox = () => {
    this.setState({ sealModalBox: false })
  }
 
  handleHardwareBackPress = () => {
    const { sealModalBox } = this.state
    if (!sealModalBox) {
      this.animate(getHeight(), () => this.forceCloseModal(true))
    }
    return true
  }
 
  renderBackgroundContainer = () => (
    <Animated.View
      style={[styles.backgroundContainer,
        { opacity: this.backgroundOpacity }, this.customBackgroundStyles]}
      onTouchStart={this.handlePressTouchOutside}
    />
  )
 
  handlePressTouchOutside = () => {
    const { sealModalBox } = this.state
    if (this.disableOutsideTouch) {
      return
    }
    if (this.hideModalOnTouchOutside && !sealModalBox) {
      this.closeModal(true)
    }
  }
 
  renderModalBoxContainer = () => {
    const transform = [{ translateY: this.animatedValue }]
    let WrapperComponent = View
    let wrapperComponentProps = this.state.modalContainerStyles
    if (isIos) {
      WrapperComponent = KeyboardAvoidingView
      wrapperComponentProps = { behavior: 'padding' }
    }
    const modalHeaderComponent = this.modalHeaderComponent && this.modalHeaderComponent() ? (
      this.modalHeaderComponent()
    ) : null
    return (
      <WrapperComponent {...wrapperComponentProps}>
        <Animated.View
          style={[
            styles.modalContainer,
            styles.modalOverflowStyle(isNil(modalHeaderComponent)),
            this.showFullScreen && styles.fullHeight,
            this.showDarkTheme && styles.darkModalContainer,
            { transform, ...this.state.modalContainerStyles },
            this.isHeaderOwerFlow && styles.overFlow
          ]}
        >
          <View>
            {modalHeaderComponent}
            <View>
              {this.renderHeader()}
              {this.renderContent()}
            </View>
          </View>
        </Animated.View>
      </WrapperComponent>
    )
  }
 
  renderHeader = () => {
    const { sealModalBox } = this.state
    const panGestureHeaderStyles = this.panGestureHeaderStyles
    const header = !isEmpty(this.headerComponent) ? (
      this.headerComponent()
    ) : (
      <View
        style={[
          !sealModalBox && styles.gestureIndicator,
          this.showDarkTheme && styles.darkGestureIndicator
        ]}
      />
    )
 
    return (
      <View
        style={[!sealModalBox && styles.gestureContainer, panGestureHeaderStyles]}
        {...!sealModalBox && this.gestureResponser.panHandlers}
      >
        {header}
      </View>
    )
  }
 
  renderContent = () => (
    <View>
      {this.renderedComponent()}
    </View>
  )
 
  renderModal = () => {
    const backgroundTopComponent = this.backgroundTopComponent && this.backgroundTopComponent() ? (
      this.backgroundTopComponent()
    )
      : null
    const SView = (this.disableSafeArea || isIos) ? View : SafeAreaView
    return (
      <SafeAreaProvider style={StyleSheet.absoluteFillObject}>
        <SView style={styles.container}>
          { isNil(backgroundTopComponent) ? (
            this.renderBackgroundContainer()
          ) : (
            <View
              style={[styles.backgroundContainer, { opacity: 0.9 }]}
              onTouchStart={this.handlePressTouchOutside}
            />
          )}
          {backgroundTopComponent}
          {this.renderModalBoxContainer()}
        </SView>
      </SafeAreaProvider>
    )
  }
 
  render() {
    const { showModalBox } = this.state
    return showModalBox ? this.renderModal() : null
  }
}
