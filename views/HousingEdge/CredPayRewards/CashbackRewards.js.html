import React, {
  memo, useCallback, useEffect, useMemo, useState
} from 'react'
import {
  View, Pressable, ActivityIndicator, TouchableOpacity
} from 'react-native'
import { FlatList } from 'react-native-gesture-handler'
import { useNavigation } from '@react-navigation/native'
import { useDispatch, useSelector } from 'react-redux'
 
import ImageComponent from 'common/components/ImageComponent'
import { SimpleText, textTypes, fontType } from 'common/components/SimpleText'
import { otherServicesStrings } from 'constants/strings'
import { SCREENS } from 'constants/screens'
import { ScratchCardSelectors } from 'selectors/scratchGiftCardSelectors'
import colors from 'utils/colors'
import isEmpty from 'common/utils/isEmpty'
import {
  isClaimed, isExpired, isValidAmountToClaim
} from 'utils/scratchGiftCard'
import { useBackHandler } from 'utils/hooks/useBackHandler'
import images from 'utils/images'
import { ScratchCardActions } from 'views/RentPay/GiftCard/actions'
import GiftCard from 'views/RentPay/GiftCard'
import styles from 'views/HousingEdge/CredPayRewards/styles/CashbackRewardsStyles'
import GiftCardDetailsModal from 'views/HousingEdge/CredPayRewards/GiftCardDetailsModal'
import { CredPayRewardsUtils } from 'views/HousingEdge/CredPayRewards/rewardsHelper'
import CashbackExpiredSectionContainer from 'views/HousingEdge/CredPayRewards/CashbackExpiredSectionContainer'
import { useStyles } from 'utils/hooks/useStyles'
import { useSafeAreaInsets } from 'react-native-safe-area-context'
import useInAppNotification from 'utils/hooks/useInAppNotification'
import { inAppNotifScreens } from 'constants/inAppNotif'
 
const CashbackRewards = () => {
  useInAppNotification({ screenName: inAppNotifScreens.CASHBACK_SCREEN })
  const state = useSelector(state => state.scratchGiftCard)
  const allRewards = useSelector(ScratchCardSelectors.getAllRewards)
  const allRewardsStart = useSelector(ScratchCardSelectors.getAllRewardsStart)
  const allRewardsLoader = useSelector(ScratchCardSelectors.getAllRewardsLoader)
  const expiredSectionAmount = useSelector(ScratchCardSelectors.getExpiredSectionAmount)
  const navigation = useNavigation()
  const [openGiftCardDetails, setOpenGiftCard] = useState(false)
  const [selectedGiftCardOrderId, setSelectedGiftCardOrderId] = useState('')
  const [isExpanded, setIsExpanded] = useState(false)
  const dispatch = useDispatch()
 
  const openGiftCardModals = () => {
    setOpenGiftCard(true)
  }
 
  const closeGiftCardModals = () => {
    setOpenGiftCard(false)
  }
 
  const onBackPress = useCallback(() => {
    navigation.pop()
  }, [navigation])
 
  const onPress = orderId => () => {
    setSelectedGiftCardOrderId(orderId)
    openGiftCardModals()
  }
 
  const dispatchFetchMoreAllRewards = (payload) => {
    dispatch(ScratchCardActions.fetchAllRewards(payload))
  }
 
  const onPressExUlRewards = useCallback(orderId => () => {
    const giftCardData = state?.[orderId]?.data || {}
    const newClaimedAfterExpiration = giftCardData.claimedAfterExpiration
    /* For Unclaimed + Expired Cards (claimedAfterExpiration - True) with 0 amount
     (Better Luck Next Time) - Perform No action on pressing the card
    */
    if (newClaimedAfterExpiration && !isValidAmountToClaim(giftCardData.amount)) {
      return
    }
    setSelectedGiftCardOrderId(orderId)
    if (!newClaimedAfterExpiration) {
      /* Claim Scratch Card if it is Uncalimed + Expired (claimAfterExpiration - False)
       Once claimed newClaimedAfterExpiration will be true but in the giftCardData it will be false
      */
      dispatch(ScratchCardActions.claimScratchCard(orderId))
    }
    openGiftCardModals()
  }, [state])
 
  const fetchMoreAllRewards = useCallback(() => {
    if (allRewardsLoader) {
      // Prevents API Call if one API Call is already in Progress
      return
    }
    // Updates start for All Rewards
    dispatch(ScratchCardActions.updateAllRewardsStart(allRewardsStart + 1))
    const payload = {
      start: allRewardsStart + 1,
      rows: CredPayRewardsUtils.allRewardsRows
    }
    // API Call to fetch allRewards with updated start
    dispatchFetchMoreAllRewards(payload)
  }, [allRewardsLoader, allRewardsStart])
 
  const renderPaginationLoader = useMemo(() => (
    <ActivityIndicator color={colors.white} size="small" />
  ))
 
  useBackHandler(() => {
    if (isExpanded) {
      setIsExpanded(false)
      return true
    }
    navigation?.goBack?.()
    return true
  }, [isExpanded])
 
  useEffect(() => {
    if (isEmpty(allRewards)) {
      const payload = {
        start: allRewardsStart,
        rows: CredPayRewardsUtils.allRewardsRows
      }
      dispatchFetchMoreAllRewards(payload)
    }
    // Fetch Expired Section Amount if it is 0
    if (isEmpty(expiredSectionAmount)) {
      dispatch(ScratchCardActions.fetchExpiredSectionAmount())
    }
  }, [])
 
  useEffect(() => () => {
    /* Clears allRewards and expiredLateClaimedRewards
     on UnMount and Filtered  */
    dispatch(ScratchCardActions.clearCashbackRewards())
  }, [])
 
  const renderItemSeparator = useCallback(() => <View style={styles.itemSeparatorStyles} />, [])
 
  const renderReward = useCallback(({ item }) => {
    const {
      entityId, discount, orderId, rewardStatus, claimedAfterExpiration, powerUserCard = false
    } = item || {}
    const keyForCard = powerUserCard ? `${entityId}_p` : entityId
    if ((isClaimed(rewardStatus) || (isExpired(rewardStatus) && claimedAfterExpiration))
     && isValidAmountToClaim(discount)) {
      return (
        <Pressable
          onPress={onPress(keyForCard)}
        >
          <View>
            <GiftCard
              delay={0}
              orderId={keyForCard}
              mainOrderId={orderId}
              giftAmount={discount}
              isSquare
              fromScreen={SCREENS.CASHBACK_REWARDS}
              customOnPress={onPress(keyForCard)}
            />
          </View>
        </Pressable>
      )
    }
    if (isExpired(rewardStatus) && !claimedAfterExpiration) {
      return (
        <Pressable
          onPress={onPressExUlRewards(entityId)}
        >
          <View>
            <GiftCard
              delay={0}
              orderId={entityId}
              mainOrderId={orderId}
              giftAmount={discount}
              isSquare
              fromScreen={SCREENS.CASHBACK_REWARDS}
              customOnPress={onPressExUlRewards(entityId)}
            />
          </View>
        </Pressable>
      )
    }
    return (
      <View>
        <GiftCard
          delay={0}
          orderId={keyForCard}
          mainOrderId={orderId}
          giftAmount={discount}
          isSquare
          fromScreen={SCREENS.CASHBACK_REWARDS}
          customOnPress={onPress(entityId)}
        />
      </View>
    )
  }, [onPress, state])
 
  const inset = useSafeAreaInsets()
  const container = useStyles(styles.bgStyles, {inset})
 
  return (
    <View style={container}>
      <View style={styles.headerContainer}>
        <TouchableOpacity onPress={onBackPress} style={styles.backIconStyles}>
          <ImageComponent
            style={styles.backIcon}
            source={images.backButtonWhite}
          />
        </TouchableOpacity>
        <SimpleText
          text={otherServicesStrings.cashbackRewards}
          type={textTypes.large}
          fontType={fontType.medium}
          style={styles.headerTextStyles}
        />
      </View>
      <FlatList
        contentContainerStyle={styles.listContainerStyles}
        style={styles.listStyles}
        data={allRewards}
        showsVerticalScrollIndicator={false}
        renderItem={renderReward}
        numColumns={2}
        ItemSeparatorComponent={renderItemSeparator}
        columnWrapperStyle={styles.columnWrapperStyles}
        ListFooterComponent={allRewardsLoader ? renderPaginationLoader : null}
        onEndReached={fetchMoreAllRewards}
        bounces={false}
        onEndReachedThreshold={0.7}
        maxToRenderPerBatch={8}
        initialNumToRender={10}
        windowSize={17}
      />
      {!isEmpty(expiredSectionAmount) && expiredSectionAmount > 0
          && (
            <CashbackExpiredSectionContainer
              onGiftCardPress={onPress}
              isExpanded={isExpanded}
              setIsExpanded={setIsExpanded}
            />
          )
      }
      <GiftCardDetailsModal
        onClose={closeGiftCardModals}
        openGiftCardDetailsModal={openGiftCardDetails}
        selectedGiftCardOrderId={selectedGiftCardOrderId}
      />
    </View>
  )
}
 
export default memo(CashbackRewards)
