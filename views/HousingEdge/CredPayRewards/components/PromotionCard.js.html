import React, { useEffect } from 'react'
import { View, StyleSheet } from 'react-native'
import FastImage from 'react-native-fast-image'
import PropTypes from 'prop-types'
import ImageComponent from 'common/components/ImageComponent'
import { SimpleText, fontType } from 'common/components/SimpleText'
import usePOCLandingContext from 'modules/POC_Revamp_v2/hooks/usePOCLandingContext'
import { otherServicesStrings } from 'constants/strings'
import { scale, verticalScale, moderateScale } from 'common/utils/scalingFunctions'
import ButtonWrapper from 'common/components/ButtonWrapperV2/ButtonWrapperV2'
import colors from 'utils/colors'
import { CredPayUtils } from 'utils/credPayUtils'
import { POCSelectors } from 'modules/POC_Revamp_v2/selectors/pocSelector'
import { useDispatch, useSelector } from 'react-redux'
import { POCActions } from 'modules/POC_Revamp_v2/actions/pocActions'
import { trackEvent } from 'utils/trackEvents'
import { EA, EC } from 'constants/GAEvents'
import NewPaymentModal from 'modules/POC_Revamp_v2/components/returningUser/NewPaymentModal'
import { setPaymentModalRef } from 'utils/AllRefs'
import POCPaymentProvider from 'modules/POC_Revamp_v2/contexts/POCPaymentProvider'
 
const styles = StyleSheet.create({
  backgroundImage: {
    width: scale(310),
    height: verticalScale(48),
    marginHorizontal: scale(24),
    justifyContent: 'center',
    alignItems: 'center',
    borderRadius: scale(6),
    backgroundColor: colors.black2
  },
  container: {
    flex: 1,
    flexDirection: 'row',
    paddingLeft: 16,
    alignItems: 'center'
  },
  iconStyle: {
    width: 54,
    height: 44,
    marginRight: 10
  },
  titleStyle: {
    color: '#FFFFFF',
    fontSize: moderateScale(10),
    paddingTop: scale(5),
    lineHeight: moderateScale(11)
  },
  cashbackTextStyle: {
    color: '#FFFFFFCC',
    marginBottom: 4,
    fontSize: moderateScale(10),
    lineHeight: moderateScale(11)
  },
  textView: {
    flex: 1,
    marginLeft: 3,
    justifyContent: 'center'
  },
  buttonTitleStyle: { fontSize: moderateScale(12) },
  buttonContainerStyle: { marginRight: scale(20), height: scale(24), width: scale(100) }
})
 
const PromotionCard = ({
  pocPillId, imageIcon, headerText, bgImage, iconStyle
}) => {
  const { openPaymentModal } = usePOCLandingContext()
  const credPaySectionsData = useSelector(POCSelectors.getCredPaySectionsData)
  const selectedOption = CredPayUtils.getCredPayOptionDetailsById(pocPillId, credPaySectionsData)
 
  const openPaymentFlow = () => {
    trackEvent(EC.TRANSACTION_SUCCESS_PAGE, EA.PROM_CTA_CLICKED)
    openPaymentModal(undefined, null, selectedOption)
  }
 
  const {
    populateDataToReducer
  } = usePOCLandingContext()
 
  const dispatch = useDispatch()
 
  useEffect(() => {
    trackEvent(EC.TRANSACTION_SUCCESS_PAGE, EA.PROM_SEEN)
  })
 
  useEffect(() => () => {
    dispatch(POCActions.promotionCardSeen())
  }, [])
 
  return (
    <FastImage
      source={bgImage}
      style={styles.backgroundImage}
    >
      <View style={styles.container}>
        <ImageComponent
          source={imageIcon}
          style={iconStyle}
        />
        <View style={styles.textView}>
          <SimpleText
            text={headerText}
            style={styles.titleStyle}
            fontType={fontType.medium}
          />
          <SimpleText
            text={otherServicesStrings.withNoBankSurcharge}
            style={styles.cashbackTextStyle}
            fontType={fontType.normal}
          />
        </View>
        <ButtonWrapper
          title={otherServicesStrings.payNow}
          backgroundColor={colors.black3a3d}
          onPress={openPaymentFlow}
          titleStyle={styles.buttonTitleStyle}
          containerStyle={styles.buttonContainerStyle}
        />
        <POCPaymentProvider
          onPayPressed={populateDataToReducer}
          openPaymentModal={openPaymentModal}
        >
          <NewPaymentModal ref={setPaymentModalRef} />
        </POCPaymentProvider>
      </View>
    </FastImage>
  )
}
 
PromotionCard.propTypes = {
  pocPillId: PropTypes.number.isRequired,
  headerText: PropTypes.string.isRequired,
  imageIcon: PropTypes.func.isRequired,
  bgImage: PropTypes.func.isRequired,
  iconStyle: PropTypes.object.isRequired
}
 
export default PromotionCard
