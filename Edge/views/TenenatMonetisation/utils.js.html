export const useHpFaqs = () => {
  const [faqs, setFaqs] = useState([])
  const { faqs: hpFaqs = [] } = useSelector(getHPFaqStates, shallowEqual)
  const dispatch = useDispatch()
 
  const getFaqs = () => {
    if (hpFaqs.length > 0) {
      setFaqs(hpFaqs)
    } else {
      createPromiseDispatch(fetchFaqs, dispatch).then((res) => {
        setFaqs(res)
      }).catch(() => { setFaqs([]) })
    }
  }
 
  useEffect(() => {
    getFaqs()
  }, [])
 
  return [faqs]
}

export const userNameHeading = (isLoggedIn, name, isPgText) => {
  const trimedName = name?.trim() || ''
  if (isLoggedIn && !isEmpty(trimedName)) {
    const modifiedName = trimedName.split(' ')?.[0]
    return `${isPgText ? DEFAULT_STRING.HEY : DEFAULT_STRING.HI} ${capitalize(modifiedName)}!`
  }
  return `${isPgText ? DEFAULT_STRING.HEY : DEFAULT_STRING.HI} ${DEFAULT_STRING.THERE}`
}
