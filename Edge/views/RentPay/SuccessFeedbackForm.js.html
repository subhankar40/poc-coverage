import React, { PureComponent } from 'react'
import {
  View, FlatList, TouchableOpacity, Modal, Keyboard
} from 'react-native'
import { bindActionCreators } from 'redux'
import { connect } from 'react-redux'
import PropTypes from 'prop-types'
import colors from 'utils/colors'
import ImageComponent from 'common/components/ImageComponent'
import images from 'utils/images'
import GenericText from 'common/components/react-native-generic-text'
import fontUtils from 'common/utils/fontUtils'
import isEmpty from 'common/utils/isEmpty'
import { otherServicesStrings } from 'common/constants/strings'
import CollapsibleScroll from 'components/CollapsibleScroll'
import { getFeedbackOptions, clearFeedbackOptions } from 'utils/rentpayFeedbackUtil'
import { trackEvent } from 'utils/trackEvents'
import { PHRASES } from 'constants/StringConstants'
import KeyboardSpacer from 'common/components/KeyboardSpacer'
import { EA, EC, EL } from 'constants/GAEvents'
import { orderTypes } from 'constants/EdgeConstants'
import { showNewToast } from 'actions/toastActions'
import analyticsConstants from './analyticsConstants'
import FeedbackOptionComponent from './FeedbackOptionComponent.js'
import styles, { getThemesStyle } from './styles'
 
const DATA = [{}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}]
class SuccessFeedbackForm extends PureComponent {
  constructor(props) {
    super(props)
    this.state = {
      rating: -1,
      isFeedbackSelected: false,
      isVisible: false,
      keyboardVisible: false
    }
    this.scrollRef = null
    this.feedbackRef = null
    this.themeStyle = getThemesStyle(props.dark)
  }
 
  componentDidMount() {
    this.keyboardDidHideListener = Keyboard.addListener(
      'keyboardDidHide',
      this.keyboardDidHide
    )
    this.keyboardDidShowListener = Keyboard.addListener(
      'keyboardDidShow',
      this.keyboardDidShow
    )
  }
 
  componentDidUpdate(prevProps, prevState) {
    const { keyboardVisible: prevKeyboardVisible } = prevState
    const { keyboardVisible } = this.state
    if (prevKeyboardVisible !== keyboardVisible) {
      this.scrollRef && this.scrollRef.scrollToEnd()
    }
  }
 
  componentWillUnmount() {
    this.keyboardDidHideListener && this.keyboardDidHideListener.remove()
    this.keyboardDidShowListener && this.keyboardDidShowListener.remove()
    clearFeedbackOptions()
  }
 
  keyboardDidHide = () => {
    this.setState({
      keyboardVisible: false
    })
  }
 
  keyboardDidShow = () => {
    this.setState({ keyboardVisible: true })
  }
 
  setFeedbackRef = (ref) => {
    this.feedbackRef = ref
  }
 
  setScrollRef = (ref) => {
    this.scrollRef = ref
  }
 
  openModal = () => {
    const { source, isHP } = this.props
    if (!isEmpty(getFeedbackOptions())) {
      this.setState({
        isVisible: true,
        isFeedbackSelected: false
      })
      if (isHP) {
        trackEvent(EC.HP_NPS, EA.SECTION_SEEN, { source })
      } else {
        trackEvent(analyticsConstants.RP_FEEDBACK_SCREEN, EA.OPENED)
      }
    }
  }
 
  onClose = (e, sendEvent = true) => {
    const {
      isHP, source, submitFeedback, dispatch
    } = this.props
 
    this.setState({
      isVisible: false,
      rating: -1
    })
    if (sendEvent) {
      if (isHP) {
        trackEvent(EC.HP_NPS, EA.CLOSED, { source })
        const body = {
          rating: -1,
          feedback_option_ids: [],
          feedback_type_id: 1,
          service_object_type: orderTypes.TENANT_MONETIZATION
        }
        const payload = {
          body,
          submitClicked: false
        }
        dispatch(submitFeedback(payload))
      } else {
        trackEvent(EC.POC_NPS, EA.NPS_SHEET_CLOSED, { [EL.NPS_VALUE]: this.state.rating })
        trackEvent(analyticsConstants.RP_FEEDBACK_SCREEN, EA.CLOSED)
        dispatch(submitFeedback({
          rating: null,
          feedback_option_ids: [],
          feedback_type_id: 1
        }))
      }
    }
  }
 
  renderText = ({
    text, size, lineHeight, color, font, ...restProps
  }) => (
    <GenericText
      size={size || 14}
      lineHeight={lineHeight || 17}
      color={color || colors.grey20}
      font={font || fontUtils.appFont}
      {...restProps}
    >
      {text}
    </GenericText>
  )
 
  onRatingPress = (index) => {
    trackEvent(EC.POC_NPS, EA.NPS_VALUE_SELECTED, { [EL.NPS_VALUE]: index })
    this.setState({ rating: index }, () => {
      this.scrollRef.scrollToEnd()
    })
  }
 
  ratingBgColor = (rating) => {
    if (rating >= 0 && rating <= 6) {
      return colors.red11
    } if (rating >= 7 && rating <= 8) {
      return colors.paleYellow1
    } if (rating >= 9 && rating <= 10) {
      return colors.brightGreen
    }
    return colors.grey9
  }
 
  renderRatingItem = ({ index }) => {
    const { rating } = this.state
    const bgColor = this.props.dark ? colors.black37 : colors.paleGrey
    return (
      <TouchableOpacity
        onPress={() => this.onRatingPress(index)}
        activeOpacity={1}
        style={[styles.ratingStyle, {
          backgroundColor: index <= rating
            ? this.ratingBgColor(rating) : bgColor
        }, index === 0 && styles.firstRatingStyle,
        index === 10 && styles.lastRatingStyle]}
      >
        {this.renderText({
          text: index,
          size: 12,
          ...this.themeStyle.heading
        })}
      </TouchableOpacity>
    )
  }
 
  separator = () => (<View style={styles.sepertorStyle} />)
 
  renderRating = () => {
    const { rating } = this.state
    return (
      <FlatList
        horizontal
        data={DATA}
        renderItem={this.renderRatingItem}
        extraData={rating}
        style={styles.ratingList}
        ItemSeparatorComponent={this.separator}
      />
    )
  }
 
  getSubtitle = () => {
    const { rating } = this.state
    if (rating === -1) {
      return PHRASES.WE_PROMISE
    }
    // this will be used when BE adds support for different feedback options when rating is 5 start
    // if (rating === 5) {
    //   return 'Tell us what you loved  '
    // }
    return PHRASES.CHOOSE_ONE_MORE
  }
 
  renderRatingContainer = () => {
    const { isHP } = this.props
 
    return (
      <View>
        <View style={[styles.ratingContainer, this.themeStyle.headingContainer]}>
          <View style={[styles.scrollHandler, this.themeStyle.scrollHandler]} />
          {this.renderText({
            text: isHP ? PHRASES.hpString : PHRASES.rentPayString,
            size: 17,
            lineHeight: 22,
            letterSpacing: 0.2,
            style: styles.headerTextStyle,
            ...this.themeStyle.heading
          })}
        </View>
        <View style={styles.subTextContainer}>
          {this.renderText({
            text: otherServicesStrings.notAtAll,
            size: 12,
            lineHeight: 16,
            style: styles.ratingEmojiLeft,
            ...this.themeStyle.ratingDescription
          })}
          {this.renderText({
            text: otherServicesStrings.ofCourse,
            size: 12,
            lineHeight: 16,
            style: styles.emojiRight,
            ...this.themeStyle.ratingDescription
          })}
        </View>
        <View style={styles.alignItem}>
          {this.renderRating()}
        </View>
      </View>
    )
  }
 
  onOptionChange = (value) => {
    const { isFeedbackSelected } = this.state
    value !== isFeedbackSelected && this.setState({
      isFeedbackSelected: value
    })
  }
 
  renderFeedbackContainer = () => (
    <View>
      <ImageComponent
        style={styles.curveLine}
        source={images.curveLine}
      />
      <FeedbackOptionComponent
        ref={this.setFeedbackRef}
        data={getFeedbackOptions()}
        onChange={this.onOptionChange}
        showCheckbox={false}
        dark={this.props.dark}
      />
    </View>
  )
 
  isButtonActive = () => this.state.rating !== -1
 
  getFeedBackText = () => (
    /* For some reason the key inside selected ids for text is always 5 according to
    the existing code .TODO - Please debug this and fix this whenever possible
    */
    this.feedbackRef?.state?.selectedIds[5]?.text
  )
 
  onSubmit = () => {
    const { rating, isFeedbackSelected } = this.state
    const {
      submitFeedback, isHP, source, dispatch
    } = this.props
 
    if (rating !== -1) {
      const feedbackOptionIds = this.feedbackRef && this.feedbackRef.getParsedSelectedIds()
      if (isHP) {
        const body = {
          rating,
          feedback_option_ids: feedbackOptionIds,
          feedback_type_id: 1,
          service_object_type: orderTypes.TENANT_MONETIZATION
        }
        const payload = {
          body,
          submitClicked: true
        }
        dispatch(submitFeedback(payload))
        trackEvent(EC.HP_NPS, EA.FILLED, { source })
        this.onClose({}, false)
      } else {
        trackEvent(EC.POC_NPS, EA.NPS_SUBMITTED,
          { [EL.NPS_VALUE]: rating, [EL.SUGGESTION]: this.getFeedBackText() })
        const payload = { rating, feedback_option_ids: feedbackOptionIds, feedback_type_id: 1 }
        dispatch(submitFeedback(payload))
        const action = isFeedbackSelected
          ? otherServicesStrings.feedbackSubmit : otherServicesStrings.ratingSubmitted
        trackEvent(analyticsConstants.RP_FEEDBACK_SCREEN, action)
        this.onClose({}, false)
      }
    } else {
      showNewToast(otherServicesStrings.pleaseSelectRating)
    }
  }
 
  renderSubmitButton = () => (
    <TouchableOpacity
      style={[styles.submitButton, this.isButtonActive() && styles.buttonStyle]}
      activeOpacity={1}
      disabled={!this.isButtonActive()}
      onPress={this.onSubmit}
    >
      {this.renderText({
        text: otherServicesStrings.submit, color: colors.white, font: fontUtils.appFontMedium
      })}
    </TouchableOpacity>
  )
 
  render() {
    const { rating, isVisible } = this.state
    if (!getFeedbackOptions()) {
      return null
    }
 
    return (
      <Modal
        animationType="fade"
        transparent
        visible={isVisible}
        onRequestClose={this.onClose}
      >
        <CollapsibleScroll
          useThresholdToClose
          closeModal={this.onClose}
          ref={this.setScrollRef}
        >
          <View style={[styles.container, this.themeStyle.container]}>
            {this.renderRatingContainer()}
            {rating !== -1 && this.renderFeedbackContainer()}
            {this.renderSubmitButton()}
            <KeyboardSpacer />
          </View>
        </CollapsibleScroll>
      </Modal>
    )
  }
}
 
SuccessFeedbackForm.propTypes = {
  submitFeedback: PropTypes.func.isRequired,
  showNewToast: PropTypes.func.isRequired,
  dark: PropTypes.bool,
  isHP: PropTypes.bool,
  source: PropTypes.string,
  dispatch: PropTypes.func.isRequired
}
 
SuccessFeedbackForm.defaultProps = {
  dark: false,
  isHP: false,
  source: ''
}
 
const mapDispatchToProps = dispatch => ({
  ...bindActionCreators({
    showNewToast
  }, dispatch),
  dispatch
})
 
export default connect(
  null,
  mapDispatchToProps,
  null,
  { forwardRef: true }
)(SuccessFeedbackForm)
