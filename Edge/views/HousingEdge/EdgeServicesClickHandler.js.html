const navigateToPayRent = (navigation, useNativeNavigation = false) => {
  // by doing we fixed one edge case. when user has not selected any city from onboarding
  // and directly lands in rentpay home {returning user} from edge landing page
  // and come back to edge landing page and after that when user is trying to press
  // hard ware back press from edge. home page is loaded with empty city.
  navigation.navigate(SCREENS.PAY_ON_CREDIT_V2)
}
 
const navigateToService = (page, navigation, params) => {
  if (page) { navigation?.navigate?.(page, params) }
}
 
/**
 * This is nothing but a interface that help in order to have handler for
 * given id, coming from backend.
 * We can add handler for onClick event.
 */
const CALLBACK_HANDLERS = {
  [edgeHooksConfig.PAY_RENT]:
    ({ navigation }) => navigateToPayRent(
      navigation, false
    ),
  [edgeHooksConfig.RENT_AGREEMENT]:
    ({ navigation }) => navigateToService(SCREENS.RENT_AGREEMENT, navigation),
  [edgeHooksConfig.HOME_INTERIORS]:
    ({ navigation }) => navigateToService(SCREENS.HI_LANDING_PAGE, navigation),
  [edgeHooksConfig.SOLAR_ROOFTOP]:
    ({ navigation }) => navigateToService(SCREENS.SOLAR_ROOFTOP_LANDING_SCREEN, navigation),
  [edgeHooksConfig.HOME_LOANS]:
    ({ navigation }) => navigateToService(SCREENS.HOME_LOAN_LANDING_SCREEN, navigation),
  [edgeHooksConfig.PROPERTY_MANAGEMENT]:
    ({ navigation }) => navigateToService(SCREENS.PROPERTY_MANAGEMENT_LANDING_SCREEN, navigation),
  [edgeHooksConfig.FIXED_DEPOSIT]:
    ({ navigation }) => navigateToService(SCREENS.FD_LANDING_SCREEN, navigation),
  [edgeHooksConfig.HOUSING_PREMIUM]:
    ({ navigation, fromProfile, source }) => {
      /**
       * Applying dispatch before navigation to make sure that
       * whenever HP-Landing-page has been navigated, it should be fresh
       * for transcation status, so that if in case transcation status is success
       * and screen should not be navigated back or seems like self going back
       * action.
       */
      Promise.resolve().then(() => {
        store?.dispatch?.(updateTMOrderStatus({ status: TMstatus.INIT }))
      }).then(() => {
        const fromPage = {
          fromPage: fromProfile ? SOURCES.PROFILE_PAGE_HOUSING_EDGE
            : source
        }
        navigateToService(SCREENS.HOUSING_PREMIUM, navigation, fromPage)
      })
    },
  [edgeHooksConfig.QUICK_CASH]: ({
    navigation, fromProfile, isLoggedIn, source
  }) => {
    let fromPg = SOURCES.EDGE_SERVICES_BOTTOM_BAR
    if (fromProfile) { fromPg = SOURCES.PROFILE_PAGE_HOUSING_EDGE }
    if (source === SOURCES.EDGE_LANDING_SPOTLIGHT) {
      fromPg = SOURCES.EDGE_SERVICES_SPOTLIGHT
    }
    trackEvent(EC.QUICK_CASH_HOOK, EA.REDIRECT, {
      hook_name: fromPg,
      login_status: isLoggedIn
    })
    navigation.navigate(SCREENS.NIRO_PL_WEB_VIEW)
  },
  [edgeHooksConfig.CREDIT_LINE]: ({ navigation, platformSpecificServiceUrl }) => {
    trackEvent(EC.EDGE_SERVICE_PAGE, EA.CL_CTA_CLICKED)
    navigation.navigate(SCREENS.CREDIT_LINE_WEBVIEW, { url: platformSpecificServiceUrl })
  },
  [edgeHooksConfig.PERSONAL_LOAN]:
    ({ navigation }) => navigateToService(SCREENS.PERSONAL_LOAN_LANDING, navigation, {
      eventSource: EC.EDGE_SERVICE_PAGE
    }),
  [edgeHooksConfig.PACKERS_MOVERS]:
    ({ navigation }) => navigateToService(SCREENS.PACKERS_MOVERS_LANDING, navigation),
  [edgeHooksConfig.RENT_NOW_PAY_LATER]:
    ({ navigation }) => navigateToService(SCREENS.BNPL_SCREEN, navigation)
}
 
export const onEdgeServiceClick = (id, serviceUrl, args = {}) => {
  // eslint-disable-next-line no-prototype-builtins
  if (CALLBACK_HANDLERS.hasOwnProperty(id)) {
    CALLBACK_HANDLERS[id]?.(args)
  } else if (serviceUrl) {
    // default case
    openInBrowser(serviceUrl)
  }
}
