const HOME_CTA = {
  title: '',
  id: 'home',
  ctaText: 'Home'
}
const ACTIVE_CTA = {
  title: Constants.Policies,
  id: 'active',
  ctaText: 'Active'
}
const EXPIRED_CTA = {
  title: Constants.Policies,
  id: 'expired',
  ctaText: 'Expired'
}
 
const getCtaList = (isActivePolicies, isExpiredPolicies) => {
  if (isActivePolicies && isExpiredPolicies) {
    return [HOME_CTA, ACTIVE_CTA, EXPIRED_CTA]
  }
  if (isActivePolicies && !isExpiredPolicies) {
    return [HOME_CTA, ACTIVE_CTA]
  }
  if (!isActivePolicies && isExpiredPolicies) {
    return [HOME_CTA, EXPIRED_CTA]
  }
  return [HOME_CTA, ACTIVE_CTA]
}
 
const getIndex = (tab, activePolicies, expiredPolicies) => {
  if (tab === ACTIVE_CTA.id) {
    return !isEmpty(activePolicies) ? tab : HOME_CTA.id
  }
  if (tab === EXPIRED_CTA.id) {
    return !isEmpty(expiredPolicies) ? tab : HOME_CTA.id
  }
  return tab
}
 
export default function InsuranceReturningUser(props) {
  const {
    handleRenewInsurance,
    openInsuranceDetailsModal,
    handleBuyInsurance,
    setDeeplinkState,
    fromDeepLinkDetail
  } = props
  const {
    params: {
      // @ts-ignore
      tab, orderId
    } = {}
  } = useRoute() || {}
 
  const listRef = useRef(null)
  const scrollX = useRef(new Animated.Value(0)).current
  const [isDetailModal, setDatailModal] = useState(false)
  const [isRaiseClaimModal, setRaiseClaimModal] = useState(false)
  const [selectedItem, setData] = useState({})
  const {
    title
  } = useSelector(getHousingProtectConfigV2, shallowEqual)
  const userName = useSelector(getUserName)
  const activePolicies = useSelector(geActivePoliciesData, shallowEqual)
  const expiredPolicies = useSelector(geExpiredPoliciesData, shallowEqual)
  const allpolicies = useSelector(gePoliciesData, shallowEqual)
 
  const navigation = useNavigation()
 
  const CTAs = useMemo(
    () => getCtaList(!isEmpty(activePolicies), !isEmpty(expiredPolicies)),
    [activePolicies, expiredPolicies]
  )
 
  const [activeCta, setActiveCta] = useState(getIndex(tab,
    activePolicies, expiredPolicies) || CTAs[1].id)
  const transactionData = useMemo(
    () => activeCta === ACTIVE_CTA.id ? activePolicies : expiredPolicies,
    [activeCta, activePolicies]
  )
  const firstName = useMemo(() => userName?.split(' ')?.[0] || 'There', [userName])
 
  const onDetailPress = (item) => {
    setData(item)
    setDatailModal(true)
  }
 
  const onCloseDetailModal = () => {
    setDatailModal(false)
  }
 
  const onToggleRaiseModal = () => {
    onCloseDetailModal()
    setRaiseClaimModal(!isRaiseClaimModal)
  }
 
  const handleCtaPress = item => () => {
    setActiveCta(item.id)
    trackEvent(EC.INSURANCE_LANDING_PAGE, EA.TAB_CLICKED, {
      tab_name: item.id
    })
    listRef?.current?.scrollToOffset?.({ offset: 0, animated: false })
  }
 
  const handleRenewPressFromDetailsModal = (...args) => {
    onCloseDetailModal()
    handleRenewInsurance(...args)
  }
 
  const handleBuyNowPressFromDetailsModal = (...args) => {
    onCloseDetailModal()
    handleBuyInsurance(...args)
  }
 
  useEffect(() => {
    if (fromDeepLinkDetail) {
      const data = getOrderById(allpolicies, orderId)
      if (!isEmpty(data)) {
        navigation.navigate(
          // @ts-ignore
          SCREENS.INSURANCE_QUICK_LINKS,
          { selectedItem: data }
        )
        setDeeplinkState?.(false)
      }
    }
  }, [orderId, allpolicies, fromDeepLinkDetail])
 
  const renderProductCards = () => (
    <GradientWrapper
      colors={[colors.white_009, colors.grey_56_026]}
      start={{ x: -1, y: 0.56 }}
      end={{ x: 0.908, y: -0.162 }}
      style={styles.productsContainer}
    >
      <SimpleText
        // @ts-ignore
        type={textTypes.small}
        text="OUR PRODUCTS"
        style={styles.productHeading}
      />
      <InsuranceProducts
        openInsuranceDetailsModal={openInsuranceDetailsModal}
        handleBuyInsurance={handleBuyInsurance}
      />
    </GradientWrapper>
  )
 
  // eslint-disable-next-line react/prop-types
  const renderPolicyCard = ({ item }) => (
    <PolicyCard
      data={item}
      onSelect={onDetailPress}
      onRenew={handleRenewInsurance}
      handleBuyInsurance={handleBuyInsurance}
    />
  )
 
  return (
    <>
      <View style={styles.topContainer}>
        <View style={styles.repeatUserTopIconWrapper}>
          <ImageComponent
            source={images.HousingIcon}
            style={styles.repUserTopImgStyle}
          />
          <Text style={styles.repeatUserTitleStyle}>{title}</Text>
        </View>
        <Text style={styles.repeatUserHey}>
          {strings.HEY}
          <Text style={styles.repeatUserName}>
            {` ${firstName}!`}
          </Text>
        </Text>
        <View style={styles.policiesContainer}>
          {activeCta !== CTAs[0].id ? (
            <Text style={styles.policiesTitleStyle}>
              {Constants.Policies}
            </Text>
          ) : null }
          <View style={styles.policiesCtaContainer}>
            {
              CTAs.map(item => (
                <PoliciesCta
                  key={item.id}
                  title={item.ctaText}
                  isActive={activeCta === item.id}
                  onPress={handleCtaPress(item)}
                />
              ))
            }
          </View>
        </View>
      </View>
      {activeCta !== CTAs[0].id ? (
        <View style={styles.listDataContainer}>
          <FlatList
            horizontal
            ref={listRef}
            data={transactionData}
            onScroll={Animated.event(
              [{ nativeEvent: { contentOffset: { x: scrollX } } }],
              { useNativeDriver: false }
            )}
            keyExtractor={item => item?.order?.id?.toString?.()}
            renderItem={renderPolicyCard}
            contentContainerStyle={styles.policiesContainer}
            ItemSeparatorComponent={() => <View style={styles.separator} />}
            showsHorizontalScrollIndicator={false}
          />
          <ExpandingDot
            data={transactionData}
            expandingDotWidth={24}
            scrollX={scrollX}
            inActiveDotOpacity={0.6}
            activeDotColor={colors.seafoamBlue}
            inActiveDotColor={colors.lightGrey12}
            dotStyle={styles.dotStyle}
            containerStyle={styles.dotsContainer}
          />
        </View>
      ) : null}
      {activeCta === CTAs[0].id ? renderProductCards() : null}
      <DetailModal
        visible={isDetailModal}
        data={selectedItem}
        onClose={onCloseDetailModal}
        onToggleRaiseModal={onToggleRaiseModal}
        onRenew={handleRenewPressFromDetailsModal}
        handleBuyInsurance={handleBuyNowPressFromDetailsModal}
      />
      <RaiseClaimModal
        visible={isRaiseClaimModal}
        onClose={onToggleRaiseModal}
        selectedItem={selectedItem}
      />
    </>
  )
}
