const deviceHeight = getHeight()
const ScrollViewRef = React.createRef()
const BLANK_HEIGHT = 2
const edgeLandingScrollRef = React.createRef()
const SectionRefs = {
  S1: React.createRef(),
  S2: React.createRef(),
  S3: React.createRef(),
  S4: React.createRef(),
  S5: React.createRef(),
  S6: React.createRef()
}
 
const triggerOffset = getHeight() / 3
 
class EdgeLanding extends React.PureComponent {
  scrollYEnd = 0
 
  constructor(props) {
    super(props)
    this.sectionOffset = {}
    this.sections = []
    this.timer = null
    this.state = { showContent: false }
    this.eLandingAnimatedVal = new Animated.Value(0)
    this.edgeDetailsAnimatedVal = new Animated.Value(-deviceHeight)
    this.isAnimationStarted = false
    this.bottomSheetAnimatedValue = new Animated.Value(deviceHeight)
    this.scrollY = new Animated.Value(0)
  }
 
  componentDidMount() {
    const {
      navigation, getEdgeTestimonials, getEdgeArticles, getEdgeFaq, getEdgeStats, route,
      fetchNiroPlAcessTokens, togglePlTag
    } = this.props
    if (route?.params?.fromNiroPlDeeplink) {
      fetchNiroPlAcessTokens()
      navigation.navigate(SCREENS.NIRO_PL_WEB_VIEW)
    }
    togglePlTag(false)
    getEdgeTestimonials()
    getEdgeArticles()
    getEdgeFaq()
    getEdgeStats()
 
    if (!edgeStartAction.ishousingEdgeCalled) edgeStartAction.sendHousingEdgeDispatches()
    this.timer = this.setTimer()
    navigation.addListener('focus', this.onDidFocus)
    navigation.addListener('blur', this.onDidBlur)
    trackEvent(analyticsConstants.EDGE_LANDING_PAGE_CATEGORY, EA.LANDING_PAGE_OPENED)
    if (shouldStopTrace(TraceKeys.HOME_DEEPLINK_TRACE)) {
      stopTrace(TraceKeys.HOME_DEEPLINK_TRACE)
    }
  }
 
  componentWillUnmount() {
    const { navigation, route } = this.props
    this.timer && clearTimeout(this.timer)
    navigation.removeListener('focus', this.onDidFocus)
    navigation.removeListener('blur', this.onDidBlur)
    if (route?.params?.edgeDripMoeNotif && isAndroid) {
      UtilNativeModule.unSetDeeplinkUrl()
    }
  }
 
  setTimer = () => setTimeout(() => {
    this.scrollToServices()
  }, 100)
 
  onDidFocus = () => {
    this.timer && clearTimeout(this.timer)
    this.eLandingAnimatedVal = new Animated.Value(0)
    this.edgeDetailsAnimatedVal = new Animated.Value(-deviceHeight)
    this.startLandingAnimation()
    // reassigning animated value bcs when user comes from
    // deeplinks of rent-agreement and comeback in this screen after 4-5 sec.
    // UI was distorted for couple of seconds. which happened
    // bcs of we put setTimeout for scrollToServices animation
    // which was running in background(means behind of rent agreement screen).
    this.timer = this.setTimer()
    this.setState({ showContent: true })
  }
 
  onDidBlur = () => {
    this.timer && clearTimeout(this.timer)
    this.setState({ showContent: false })
  }
 
  startAnimation = (edgeDetailsFVal, edgeLandingFVal, doScrolling = false) => {
    Animated.parallel([Animated.timing(this.edgeDetailsAnimatedVal, {
      duration: 700,
      toValue: edgeDetailsFVal,
      useNativeDriver: true
    }), Animated.timing(this.eLandingAnimatedVal, {
      duration: 700,
      toValue: edgeLandingFVal,
      useNativeDriver: true
    })]).start(() => { this.isAnimationStarted = false })
    this.isAnimationStarted = true
    !doScrolling && ScrollViewRef.current?.scrollTo({ y: BLANK_HEIGHT, animated: true })
    doScrolling && edgeLandingScrollRef.current?.scrollTo({ y: 0, animated: true })
  }
 
  scrollToServices = () => {
    !this.isAnimationStarted && this.startAnimation(0, -deviceHeight)
    if (!isEmpty(this.sectionOffset)
      && !isEmpty(this.sectionOffset[1]) && !this.sectionOffset[1].visible) {
      SectionRefs.S1.current?.showContent?.()
      this.sectionOffset[1].visible = true
    }
  }
 
  onLayout = key => (e) => {
    this.sectionOffset[key] = {}
    this.sectionOffset[key].y = e?.nativeEvent?.layout?.y
    this.sectionOffset[key].visible = false
    this.sections[key] = key
  }
 
  onScroll = (e) => {
    const { nativeEvent: { contentOffset: { y } } } = e
    const offset = (y + getHeight() - triggerOffset)
    this.sections.forEach((key) => {
      const sectionOffset = this.sectionOffset[`${key}`].y
      const sectionVisible = this.sectionOffset[`${key}`].visible
      if (!sectionVisible && offset > sectionOffset) {
        SectionRefs[`S${key}`].current?.showContent?.()
        this.sectionOffset[`${key}`].visible = true
      }
    })
    if (y <= 0) {
      !this.isAnimationStarted && this.startAnimation(deviceHeight, 0, true)
    }
  }
 
  onClickServicehandle = (id, serviceUrl, fromPage, platformSpecificServiceUrl = '') => {
    const {
      isLoggedIn
    } = this.props
    // Navigation handling when user click on edge service or spotlight as well
    onEdgeServiceClick(id, serviceUrl, {
      navigation: this.props.navigation,
      credPayDarkMode: this.props.credPayDarkMode,
      source: fromPage,
      isLoggedIn,
      platformSpecificServiceUrl
    })
  }
 
  renderScrollView = () => {
    const {
      S1, S2, S3, S4, S5, S6
    } = SectionRefs
    const { navigation, spotlights = [] } = this.props
    return (
      <View>
        {spotlights?.length > 0 ? (
          <Animated.View
            style={{
              marginTop: 120,
              opacity: this.scrollY.interpolate({
                inputRange: [0, 40],
                outputRange: [1, 0],
                extrapolate: 'clamp'
              })
            }}
          >
            <SpotlightsRow
              data={spotlights}
              onClickHandler={this.onClickServicehandle}
            />
          </Animated.View>
        ) : null}
        <Animated.View style={[styles.scrollServicesContainer, {
          transform: [{
            translateY: this.bottomSheetAnimatedValue
          }],
          marginTop: spotlights?.length > 0 ? 0 : 120
        }]}
        >
          <TenantServices
            ref={S1}
            onLayout={this.onLayout(1)}
            navigation={navigation}
            onClickHandler={this.onClickServicehandle}
          />
          <HomeLoanCalculators navigation={navigation} />
          <OtherServices ref={S2} onLayout={this.onLayout(2)} navigation={navigation} />
          <BenefitsView ref={S3} onLayout={this.onLayout(3)} />
          <Testimonials ref={S4} onLayout={this.onLayout(4)} />
          <ArticlesView ref={S5} onLayout={this.onLayout(5)} navigation={navigation} />
          <FAQView ref={S6} onLayout={this.onLayout(6)} />
        </Animated.View>
      </View>
    )
  }
 
  onScrollEndDrag = (e) => {
    const { contentOffset } = e.nativeEvent
    const isUp = contentOffset.y > this.scrollYEnd
    const isScroll = this.props.spotlights.length > 0
    if (isScroll && isUp && (contentOffset.y > 5 && contentOffset.y <= 170)) {
      ScrollViewRef.current?.scrollTo({
        y: checkIndicatorExist(this.props.spotlights.length)
          ? 170 : 160,
        animated: true
      })
    } else if (isScroll && !isUp && (contentOffset.y > 0 && contentOffset.y <= 170)) {
      ScrollViewRef.current?.scrollTo({ y: 0, animated: true })
    }
    this.scrollYEnd = contentOffset.y
  }
 
  onScrollBeginDrag = (e) => {
    const { contentOffset } = e.nativeEvent
    this.scrollYEnd = contentOffset.y
  }
 
  onBackArrow = () => {
    backPress(this.props.navigation)
  }
 
  startLandingAnimation() {
    this.bottomSheetAnimatedValue.setValue(deviceHeight)
    Animated.timing(this.bottomSheetAnimatedValue, {
      duration: 500,
      toValue: 0,
      useNativeDriver: true,
      easing: Easing.inOut(Easing.quad)
    }).start()
  }
 
  render() {
    const { showContent } = this.state
    if (!showContent) return null
    return (
      <SafeAreaView style={{ flex: 1 }} edges={['top']}>
        <View style={styles.bgCoverStyle}>
          <EdgeBg />
        </View>
        <Animated.ScrollView
          ref={ScrollViewRef}
          showsVerticalScrollIndicator={false}
          onScroll={Animated.event([{
            nativeEvent: {
              contentOffset: {
                y: this.scrollY
              }
            }
          }], { listener: this.onScroll })}
          onScrollBeginDrag={this.onScrollBeginDrag}
          onScrollEndDrag={this.onScrollEndDrag}
          contentContainerStyle={styles.contentContainer}
        >
          <TouchableOpacity
            onPress={this.onBackArrow}
            style={styles.leftBtnStyle}
            activeOpacity={0.9}
          >
            <ImageComponent
              style={styles.imgStyle}
              source={images.getHpBackIcon}
            />
          </TouchableOpacity>
          {this.renderScrollView()}
        </Animated.ScrollView>
      </SafeAreaView>
    )
  }
}
 
EdgeLanding.propTypes = {
  route: PropTypes.object.isRequired,
  navigation: PropTypes.object.isRequired,
  getEdgeTestimonials: PropTypes.func.isRequired,
  getEdgeArticles: PropTypes.func.isRequired,
  getEdgeFaq: PropTypes.func.isRequired,
  getEdgeStats: PropTypes.func.isRequired,
  spotlights: PropTypes.array.isRequired,

}
