import React, {
  useEffect, useState, useRef, createRef, memo, useCallback
} from 'react'
import { useNavigation } from '@react-navigation/native'
import { useDispatch, useSelector } from 'react-redux'
import PropTypes from 'prop-types'
import {
  View, Modal, Pressable
} from 'react-native'
import ButtonWrapper from 'common/components/ButtonWrapper'
import CollapsibleScroll from 'components/CollapsibleScroll'
import { SimpleText, textTypes, fontType } from 'common/components/SimpleText'
import { otherServicesStrings } from 'common/constants/strings'
import { SCREENS } from 'constants/screens'
import ImageComponent from 'common/components/ImageComponent'
import { UserSelectors } from 'selectors/user'
import isEmpty from 'common/utils/isEmpty'
import images from 'utils/images'
import { emptyFunc } from 'common/utils/ValidationUtils'
import LottieViewWrapper from 'common/components/LottieViewWrapper'
import { DigioActions } from 'views/HousingEdge/Aadhaar/DigioActions'
import { trackEvent } from 'utils/trackEvents'
import { EA, EC } from 'constants/GAEvents'
import { delayedPromise } from 'utils/CommonUtils'
import styles, { barStyles } from './styles/PaymentKycLaunchScreenStyles'
 
const PaymentKycLaunchScreen = (props) => {
  const [isLoading, setIsLoading] = useState(false)
  const [showFirstLottie, setShowFirstLottie] = useState(true)
  const scrollRef = createRef()
  const { customCallback, closeModal } = props
  const dispatch = useDispatch()
  const navigation = useNavigation()
  const userName = useSelector(UserSelectors.getUserName)
  const userPhone = useSelector(UserSelectors.getUserPhoneNos)
  const userEmail = useSelector(UserSelectors.getEmail)
  const secondLottieRef = useRef(null)
 
  useEffect(() => {
    if (!showFirstLottie) {
      delayedPromise(20).then(() => secondLottieRef?.current?.play?.())
    }
  }, [showFirstLottie])
 
  const onClose = useCallback(() => {
    dispatch(DigioActions.setIsKycModalFlowActive(false))
    trackEvent(EC.POC_PRE_AADHAR_FLOW, EA.CANCEL_CTA)
    closeModal?.()
  }, [])
 
  const hideFirstLottie = useCallback(() => {
    setShowFirstLottie(false)
  }, [])
 
  const getCustomerIdentifier = () => {
    if (!isEmpty(userPhone)) return userPhone
    if (!isEmpty(userEmail)) return userEmail
    return null
  }
 
  const startKycForPoc = () => {
    if (isLoading) {
      return
    }
    const customerIdentifier = getCustomerIdentifier()
 
    setIsLoading(true)
    // Callback to naviate to Digio Web View after fetching url & requestId
    const callback = (err) => {
      setIsLoading(false)
      closeModal?.()
      if (!err) {
        navigation.navigate(SCREENS.DIGIO_WEB_VIEW, { customCallback, isPrePayment: true })
      }
    }
 
    /* TODO - Handle case when we don't have user name or email.
  Fill info from user in such cases */
    if (!isEmpty(customerIdentifier)) {
      const payload = {
        customer_identifier: customerIdentifier,
        customer_name: !isEmpty(userName) ? userName : '',
        notify_customer: false,
        request_details: {},
        generate_access_token: true,
        isPrePayment: true,
        callback
      }
      // Fetch Digio Web View Url & RequestId
      dispatch(DigioActions.fetchDigioUrlReq(payload))
    }
  }
 
  const renderDescription = () => (
    <>
      <SimpleText
        text={otherServicesStrings.oops}
        type={textTypes.larger}
        fontType={fontType.medium}
        style={styles.titleStyles}
      />
      <SimpleText
        text={otherServicesStrings.yourAccountHasBeenTemporarilyBlocked}
        numberOfLines={2}
        type={textTypes.regular}
        fontType={fontType.normal}
        style={styles.subTitleStyles}
      />
      <SimpleText
        text={otherServicesStrings.toUnblockContinueKyc}
        numberOfLines={2}
        type={textTypes.regular}
        fontType={fontType.normal}
        style={styles.unblockTextStyles}
      />
    </>
  )
 
  const renderCurveImage = () => (
    <ImageComponent
      source={images.curveImage}
      style={styles.curveImageStyles}
      resizeMode="contain"
    />
  )
 
  const renderIconImage = icon => (
    <ImageComponent
      source={icon}
      style={styles.iconImageStyles}
      resizeMode="contain"
    />
  )
 
  const renderListView = () => (
    <View style={styles.listViewStyles}>
      <View style={styles.listViewItemStyles}>
        {renderIconImage(images.oneTimeLoopImage)}
        <SimpleText
          text={otherServicesStrings.oneTime}
          type={textTypes.small}
          fontType={fontType.normal}
          style={styles.listItemTextStyles}
        />
      </View>
      <View style={styles.listViewItemStyles}>
        {renderIconImage(images.clockTimer)}
        <SimpleText
          text={otherServicesStrings.takesLess}
          type={textTypes.small}
          fontType={fontType.normal}
          style={styles.listItemTextStyles}
        />
      </View>
      <View style={styles.listViewItemStyles}>
        {renderIconImage(images.instantImage)}
        <SimpleText
          text={otherServicesStrings.instantTransfer}
          type={textTypes.small}
          fontType={fontType.normal}
          style={styles.listItemTextStyles}
        />
      </View>
    </View>
  )
 
  const renderBottomSection = () => (
    <>
      <SimpleText
        text={otherServicesStrings.completeKyc}
        type={textTypes.large}
        fontType={fontType.normal}
        style={styles.bottomTitleStyles}
      />
      <SimpleText
        text={otherServicesStrings.tipKeepYourAadhar}
        type={textTypes.small}
        fontType={fontType.normal}
        style={styles.bottomSubTitleStyles}
      />
      {renderListView()}
    </>
  )
 
  const startKyc = useCallback(() => {
    dispatch(DigioActions.setIsKycModalFlowActive(true))
    trackEvent(EC.POC_PRE_AADHAR_FLOW, EA.START_KYC_CTA)
    startKycForPoc()
  }, [])
 
  const renderBottomCtas = () => (
    <>
      <View style={styles.ctaContainerStyles}>
        <Pressable onPress={onClose}>
          <SimpleText
            text={otherServicesStrings.cancel}
            type={textTypes.regular}
            fontType={fontType.medium}
            style={styles.cancelCtaStyles}
          />
        </Pressable>
        <ButtonWrapper
          onPress={startKyc}
          title={otherServicesStrings.startKyc}
          containerStyle={styles.startKycCtaStyles}
          titleStyle={styles.startKycTextStyles}
          isFetching={isLoading}
        />
      </View>
    </>
  )
 
  return (
    <Modal
      animationType="fade"
      transparent
      visible
      onRequestClose={onClose}
    >
      <View style={styles.bgModalStyles} />
      <CollapsibleScroll
        ref={scrollRef}
        useThresholdToClose
        closeModal={onClose}
      >
        <View style={styles.modalContentStyles}>
          <ImageComponent
            source={images.aadhaarModalBg}
            style={styles.bgImageStyles}
            resizeMode="contain"
          />
          <View style={styles.contentStyles}>
            <View
              style={barStyles}
            />
            <View style={styles.lottieContainerStyle}>
              {showFirstLottie && (
                <LottieViewWrapper
                  loop={false}
                  autoPlay
                  source={images.prePaymentDigioLaunchAnimation}
                  style={styles.lottieAnimationStyles}
                  onAnimationFinish={hideFirstLottie}
                />
              )
              }
              {!showFirstLottie && (
                <LottieViewWrapper
                  source={images.prePaymentDigioLaunchAnimationV2}
                  loop
                  ref={secondLottieRef}
                  style={styles.lottieStyles}
                />
              )}
            </View>
            {renderDescription()}
            {renderCurveImage()}
            {renderBottomSection()}
            {renderBottomCtas()}
          </View>
        </View>
      </CollapsibleScroll>
    </Modal>
  )
}
 
PaymentKycLaunchScreen.propTypes = {
  customCallback: PropTypes.func,
  closeModal: PropTypes.func.isRequired
}
 
PaymentKycLaunchScreen.defaultProps = {
  customCallback: emptyFunc
}
 
export default memo(PaymentKycLaunchScreen)
