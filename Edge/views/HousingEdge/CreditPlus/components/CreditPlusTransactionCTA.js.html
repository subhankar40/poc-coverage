import {
  View, Pressable
} from 'react-native'
import React, { useEffect } from 'react'
import { useDispatch, useSelector } from 'react-redux'
import ImageComponent from 'common/components/ImageComponent'
import { fontType, SimpleText, textTypes } from 'common/components/SimpleText'
import images from 'utils/images'
import { trackEvent } from 'utils/trackEvents'
import {
  knowMore, youSaved, totalSaving, amountWithRupee, saveMore, amountWhenDisplayFOMO,
  TYPES as CreditPlusModalTypes
} from 'constants/creditPlus'
import { EA, EC } from 'constants/GAEvents'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import PropTypes from 'prop-types'
import styles from './styles/CreditPlusTransactionCTA'
import { CreditPlusActions } from '../actions'
import { CreditPlusSelectors } from '../selectors'
 
const CreditPlusTransactionCTA = ({ container }) => {
  const dispatch = useDispatch()
 
  const {
    subscribed, discount, totalSavings
  } = useSelector(CreditPlusSelectors.getPocSubscriptionDetails) || {}
  const convChargeData = useSelector(CredPaySelectors.getConvChargeData) || {}
  const { subscriptionDiscountInRupees = 0, chargeInRupees = 0 } = convChargeData
  const lastOrder = useSelector(CredPaySelectors.lastSuccessfulOrderDetails)
  const transactionInProgress = useSelector(CreditPlusSelectors.getTransactionInProgress)
 
  // display fomo when amount is larger
  const getLastOrderRentAmount = () => {
    if (lastOrder?.reqBody?.rentInformation?.amount) {
      return Number(lastOrder?.reqBody?.rentInformation?.amount)
    }
    return 0
  }
 
  useEffect(() => {
    if (subscribed) {
      trackEvent(EC.TRANSACTION_SUCCESS_PAGE, EA.SUBS_YAY_SEEN)
    }
    if (!subscribed && getLastOrderRentAmount() >= amountWhenDisplayFOMO) {
      trackEvent(EC.TRANSACTION_SUCCESS_PAGE, EA.SUBS_CARD_SEEN)
    }
  }, [subscribed])
 
  const onPressCTA = () => {
    trackEvent(EC.TRANSACTION_SUCCESS_PAGE, EA.SUBS_KM_CLICKED)
    dispatch(CreditPlusActions.openSubscriptionModal({
      type: CreditPlusModalTypes.onlySubscription
    }))
  }
 
  if (transactionInProgress) {
    return null
  }
 
  const renderSubscribedText = () => (
    <>
      <SimpleText
        text={youSaved(subscriptionDiscountInRupees)}
        type={textTypes.regular}
        fontType={fontType.regular}
        style={styles.descText}
      />
      <View style={styles.rowContainer}>
        <SimpleText
          text={totalSaving}
          type={textTypes.small}
          fontType={fontType.regular}
          style={styles.descText}
        />
        <SimpleText
          text={amountWithRupee(totalSavings + subscriptionDiscountInRupees)}
          type={textTypes.small}
          fontType={fontType.regular}
          style={styles.greenText}
        />
      </View>
    </>
  )
  const renderNonSubscribedText = () => {
    const couldSaved = Math.ceil(chargeInRupees * discount / 100)
    return (
      <SimpleText
        text={saveMore(couldSaved)}
        type={textTypes.regular}
        fontType={fontType.regular}
        style={styles.descText}
      />
    )
  }
 
  if (getLastOrderRentAmount() < amountWhenDisplayFOMO && !subscribed) {
    return null
  }
 
  return (
    <View style={[styles.container, container]}>
      <ImageComponent source={images.creditPlusCTABG3} style={styles.rightBottomBG} />
      <View style={styles.iconContainer}>
        <ImageComponent source={images.creditPlusCTALogo} style={styles.plusLogo} />
      </View>
      <View style={styles.textContainer}>
        {subscribed ? renderSubscribedText() : renderNonSubscribedText()}
      </View>
      {!subscribed && (
        <Pressable style={styles.knowMoreButton} onPress={onPressCTA}>
          <SimpleText
            text={knowMore}
            type={textTypes.regular}
            fontType={fontType.medium}
            style={styles.knowMoreText}
          />
        </Pressable>
      )}
    </View>
  )
}
 
CreditPlusTransactionCTA.propTypes = {
  container: PropTypes.oneOf([PropTypes.object, PropTypes.array])
}
 
CreditPlusTransactionCTA.defaultProps = {
  container: {}
}
 
export default CreditPlusTransactionCTA
