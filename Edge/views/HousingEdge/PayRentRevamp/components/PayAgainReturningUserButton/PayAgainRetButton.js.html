import React, { memo, useCallback } from 'react'
import { otherServicesStrings } from 'common/constants/strings'
import { CredPayUtils } from 'utils/credPayUtils'
import { batch, useDispatch, useSelector } from 'react-redux'
import { bool, func, object } from 'prop-types'
 
import { hideModal } from 'components/ModalBox'
import { EA, EC, EL } from 'constants/GAEvents'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import { CreditPlusSelectors } from 'views/HousingEdge/CreditPlus/selectors'
import { CredPayV2Actions } from 'views/HousingEdge/CredPay/actions'
import { getSubscriptionDuration, getSubscriptionPercentage } from 'constants/creditPlus'
import { WalletSelectors } from 'views/HousingEdge/PayRentRevamp/components/Wallet/selectors'
import useCreditPlusCheckBoxLogic from 'views/HousingEdge/CreditPlus/hooks/useCreditPlusCheckBoxLogic'
import POCButton from 'views/HousingEdge/PayRentRevamp/components/POCPaymentButtons/components/POCButton'
import { trackEvent } from 'utils/trackEvents'
import { epochToDateFormat } from 'common/utils/date'
import isEmpty from 'utils/isEmpty'
import { CyberInsuranceSelectors } from 'selectors/cyberInsurance'
import { getServiceMapper } from './useServiceMapper'
 
const PayAgainRetButton = (props) => {
  const {
    onPayPress, transactionCardData: item, showDetailedText,
    showAnimatedCounter, fromDiscountTimerScreen
  } = props
  const { payButton } = CredPayUtils.returningUserButtons
 
  const dispatch = useDispatch()
 
  /** Selectors */
  const { getIsWalletAmtAdded, getWalletBal } = WalletSelectors
  const { getConvChargeLoader, getCredPayOptions, getPayAgainServices } = CredPaySelectors
  const isConvFeeLoading = useSelector(getConvChargeLoader)
  const credPayOptionList = useSelector(getCredPayOptions)
  const walletBal = useSelector(getWalletBal)
  const isWalletAmountAdded = useSelector(getIsWalletAmtAdded)
  const payAgainServices = useSelector(getPayAgainServices)
  const creditPlusCardCheckboxSelected = useSelector(CreditPlusSelectors.getCardCheckboxSelected)
  const isCyberInsuranceActiveToBeShown = useSelector(
    CyberInsuranceSelectors.checkCyberInsuranceActiveToBeShown
  )
 
  /* From Rent protect logic hook */
  const activeInsuranceData = getServiceMapper(isCyberInsuranceActiveToBeShown)({ item })
  const totalPayable = activeInsuranceData.totalPayableAmount
 
  /** Credit Plus logic */
  const creditPlusData = useCreditPlusCheckBoxLogic()
  const { displayPrice, cardCheckboxSelected } = creditPlusData
 
  /* Visible Service hook */
  const visibleService = payAgainServices?.[item?.orderId]
 
  const {
    discount, validityInMonths
  } = useSelector(CreditPlusSelectors.getPocSubscriptionDetails)
 
  const getCustomLabelForCreditPlus = useCallback((item, convFee = 0) => ({
    amount: item?.amount,
    subscription_fee: convFee || item?.recentConvFees?.chargeInRupees,
    subscription_duration: getSubscriptionDuration(validityInMonths),
    subscription_discount_pct: getSubscriptionPercentage(discount)
  }), [discount, validityInMonths])
 
  const getEventCustomDataForSelectedItem = (item) => {
    const payOption = credPayOptionList?.find(e => e.id === item?.rentTypeId)
    const rentType = (payOption?.rentType || payOption?.type)?.toLowerCase?.()?.split(' ')?.join('_')
    return ({
      amount: item?.amount || '',
      rent_type: rentType || '',
      beneficiary_name: item?.landlordName || '',
      payment_date: epochToDateFormat(new Date())
    })
  }
 
  const onPayButtonPressed = ({
    item, visibleService, openCreditPlusModal = false, finalPayableData
  }) => {
    const isOpt = item?.isRentProtectApplied
    const isCyberSelected = item?.isCyberInsuranceApplied
    if (activeInsuranceData.shouldShowRentProtectBanner) {
      trackEvent(
        EC.POC_HOME_PAGE,
        EA.PAY_AGAIN_CLICKED_RP,
        { label: isOpt ? EL.WITH_RENT_PROTECT_COVER : EL.WITHOUT_RENT_PROTECT_COVER }
      )
    }
    if (activeInsuranceData.shouldShowCyberInsuranceBanner) {
      trackEvent(
        EC.POC_HOME_PAGE,
        EA.PAY_AGAIN_CLICKED_INSURANCE,
        { label: isCyberSelected ? EL.WITH_CYBER_PROTECT_COVER : EL.WITHOUT_CYBER_PROTECT_COVER }
      )
    }
    trackEvent(
      EC.POC_HOME_PAGE,
      EA.PAY_AGAIN_CLICKED,
      getEventCustomDataForSelectedItem(item)
    )
    batch(() => {
      dispatch(CredPayV2Actions.updateFinalPayableData(finalPayableData))
      dispatch(CredPayV2Actions.setRentTypeId(item?.rentTypeId))
    })
    if (!openCreditPlusModal) {
      const label = getCustomLabelForCreditPlus(item)
      if (visibleService === CredPayUtils.cardServices.creditPlus) {
        label.cp_checkbox = creditPlusCardCheckboxSelected ? 'selected' : 'unselected'
        trackEvent(EC.POC_SUBSCRIPTION, EA.TOTAL_PAY_CLICKED, label)
      }
    }
    if (!showDetailedText) {
      hideModal()
    }
    onPayPress(item, payButton, visibleService, openCreditPlusModal)
 
    // Insurance new events
    if (isOpt || isCyberSelected) {
      trackEvent(EC.INSURANCE_CTA, EA.CLICK, {
        insurance_type:
        isCyberSelected ? EL.CYBER_SAFE : EL.RENT_PROTECT,
        poc_order_id: item?.orderId
      })
    }
  }
 
  const renderPayButton = () => {
    const {
      conveniencefees, recentConvFees, rentProtectFee, amount
    } = item || {}
    const convFee = !isEmpty(recentConvFees) ? recentConvFees?.chargeInRupees : conveniencefees
 
    /*
    1. Amount without discount includes - > Amount + convFee + rentProtectFee +creditPlusFee
    2. Total Payable includes all amount with discount except rentProtectFee & creditPlusFee
    */
    const {
      shouldShowRentProtectBanner,
      discountedConvFee,
      cyberInsurancePrice,
      isCyberInsuranceSelected
    } = activeInsuranceData
    let amountWithoutDiscount = amount + convFee
    // Rent Protect case
 
    if (shouldShowRentProtectBanner) {
      amountWithoutDiscount += rentProtectFee?.totalAmount || 0
    }
 
    // Cyber Insurance case
    if (isCyberInsuranceSelected) {
      amountWithoutDiscount += cyberInsurancePrice || 0
    }
 
    if (visibleService === CredPayUtils.cardServices.creditPlus && cardCheckboxSelected) {
      amountWithoutDiscount += displayPrice
    }
 
    const walletDeductions = CredPayUtils.calcWalletBalToSpend(
      isWalletAmountAdded, walletBal, discountedConvFee
    )
    const amountToPay = totalPayable - walletDeductions
 
    /** Total Saved is Subscription discount + wallet amount + invite code */
    const savedAmount = (recentConvFees?.subscriptionDiscountInRupees || 0) + walletDeductions
 
    const buttonTextCard = {
      totalPayAmount: otherServicesStrings.totalPayAmount(amountToPay),
      amountWithoutDiscount,
      amountToPay,
      savedAmount,
      walletDeductions
    }
 
    const onPress = () => {
      onPayButtonPressed({
        item,
        visibleService,
        openCreditPlusModal: false,
        finalPayableData: buttonTextCard
      })
    }
    /** TODO: remove totalPayble */
    return (
      <POCButton
        showAnimatedCounter={showAnimatedCounter}
        showDetailedText={showDetailedText}
        fromRetScreen
        buttonTextCard={buttonTextCard}
        onPress={onPress}
        isFetching={isConvFeeLoading}
        convFee={convFee}
        fromDiscountTimerScreen={fromDiscountTimerScreen}
      />
    )
  }
 
  return (
    renderPayButton()
  )
}
 
PayAgainRetButton.propTypes = {
  onPayPress: func.isRequired,
  transactionCardData: object.isRequired,
  showDetailedText: bool,
  showAnimatedCounter: bool,
  fromDiscountTimerScreen: bool
}
 
PayAgainRetButton.defaultProps = {
  showDetailedText: false,
  showAnimatedCounter: false,
  fromDiscountTimerScreen: false
}
 
export default memo(PayAgainRetButton)
