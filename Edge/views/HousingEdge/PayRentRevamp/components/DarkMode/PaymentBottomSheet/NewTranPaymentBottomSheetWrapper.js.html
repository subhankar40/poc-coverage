import React, { memo, useEffect } from 'react'
import { func } from 'prop-types'
// eslint-disable-next-line max-len
import PaymentDetailsBottomSheet from 'views/HousingEdge/PayRentRevamp/components/DarkMode/PaymentBottomSheet/PaymentDetailsBottomSheet'
import { batch, useDispatch, useSelector } from 'react-redux'
import { RentProtectSelectors } from 'selectors/rentProtect'
import { CreditPlusSelectors } from 'views/HousingEdge/CreditPlus/selectors'
import { WalletSelectors } from 'views/HousingEdge/PayRentRevamp/components/Wallet/selectors'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import { CredPayUtils } from 'utils/credPayUtils'
import { WalletActions } from 'views/HousingEdge/PayRentRevamp/components/Wallet/actions'
import NewTransactionButton from 'views/HousingEdge/PayRentRevamp/components/NewTransactionButton/index.'
import { paymentSectionKeys } from 'constants/CredPayConstants'
import { CreditPlusActions } from 'views/HousingEdge/CreditPlus/actions'
import { CredPayActions } from 'views/HousingEdge/PayRentRevamp/actions'
import useAddRemoveLogics from 'views/HousingEdge/PayRentRevamp/components/DarkMode/PaymentBottomSheet/useAddRemoveLogics'
import { hideModal } from 'components/ModalBox'
import { ReferralV2Actions } from 'views/HousingEdge/Referral/actions'
import useConvFeeService from 'views/HousingEdge/PayRentRevamp/hooks/useConvFeeService'
import { CyberInsuranceSelectors } from 'selectors/cyberInsurance'
import { CyberInsuranceAction } from 'actions/cyberInsuranceAction'
 
const { setCyberInsuranceCheckBox } = CyberInsuranceAction
 
const NewTranPaymentBottomSheetWrapper = (props) => {
  const { onPayRentPressed, disabled } = props
  const { calcWalletBalToSpend, getDiscountedConvFee } = CredPayUtils
  const dispatch = useDispatch()
  const convFeeService = useConvFeeService()
 
  const { walletPayload, inviteCodeDiscountPayload } = useAddRemoveLogics()
  /* Selectors */
  const {
    getAmount, getConvChargeData,
    getRentTypeId, getIsReturningUser
  } = CredPaySelectors
  const {
    getSubscriptionChecked, getTransactionInProgress, getPocSubscriptionDetails
  } = CreditPlusSelectors
  const { getPaymentSectionsData, getWalletBal, getIsWalletAmtAdded } = WalletSelectors
 
  const rentTypeId = useSelector(getRentTypeId)
  const paymentSectionsData = useSelector(getPaymentSectionsData)
  const amount = useSelector(getAmount)
  const getConvFeeData = useSelector(getConvChargeData)
  // const rentProtectData = useSelector(getRentProtectChargeData)
 
  const walletBal = useSelector(getWalletBal)
  const isWalletAmountAdded = useSelector(getIsWalletAmtAdded)
 
  const transactionInProgress = useSelector(getTransactionInProgress)
  const subscriptionDetails = useSelector(getPocSubscriptionDetails)
  const subscriptionSelected = useSelector(getSubscriptionChecked)
  const isReturningUser = useSelector(getIsReturningUser)
 
  // Cyber Insurance data
  const cyberInsuranceData = useSelector(CyberInsuranceSelectors.getCyberInsuranceNewFlowConfigData)
  const isCyberInsuranceActiveToBeShown = useSelector(
    CyberInsuranceSelectors.checkCyberInsuranceActiveToBeShown
  )
 
  // Actions to toggle
  const { updateReturningBotttomSheetAll, updateReturningBotttomSheetIndi } = WalletActions
  const { toggleSubscriptionSelection } = CreditPlusActions
 
  const isRentProtectApplied = useSelector(RentProtectSelectors.getRentProtectApplied)
  const rpEligible = useSelector(RentProtectSelectors.getRPEligiblityAndABTest)
  const rpMonthlyPrice = useSelector(RentProtectSelectors.getMonthlyPrice)
 
  const convFeeAmt = getConvFeeData?.chargeInRupees
  // let totalAmountAfterDiscounts = getConvFeeData?.totalAmount
  // totalAmountAfterDiscounts += rentProtectData?.totalAmount || 0
  const referralDiscount = getConvFeeData?.referralDiscount
 
  const walletBalToDeduct = calcWalletBalToSpend(isWalletAmountAdded,
    walletBal, getDiscountedConvFee(getConvFeeData))
  const subscriptionDiscountAmt = getConvFeeData?.subscriptionDiscountInRupees
 
  const performDispatchesPostOpeningModal = () => {
    const requiredService = rentTypeId === 1 || rentTypeId === 2
    const showRP = rpEligible && requiredService
 
    const payload = {
      convFeeAmt,
      walletBalAmt: isReturningUser ? walletBalToDeduct : null,
      subscriptionDiscountAmt
    }
 
    if (!subscriptionDetails?.subscribed && !transactionInProgress) {
      payload.creditPlusAmt = subscriptionSelected ? subscriptionDetails?.displayPrice : 0
      payload.iscreditPlusFeeAdded = subscriptionSelected
    } else {
      payload.creditPlusAmt = null
    }
    if (showRP && !isCyberInsuranceActiveToBeShown) {
      payload.rentProtectAmt = isRentProtectApplied ? rpMonthlyPrice : 0
      payload.isRPFeeAdded = isRentProtectApplied
    } else {
      payload.rentProtectAmt = null
    }
    if (referralDiscount > 0) {
      payload.referralDiscountAmt = referralDiscount
    } else {
      payload.referralDiscountAmt = null
    }
    /**
     * Cyber insurance flow wil be set
     */
    const {
      isCyberInsuranceEligible,
      isCyberInsuranceSelected,
      price: cyberPrice
    } = cyberInsuranceData
    if (cyberInsuranceData?.price > 0
      && isCyberInsuranceEligible && isCyberInsuranceActiveToBeShown) {
      payload.cyberInsuranceAmt = isCyberInsuranceSelected ? cyberPrice : 0
      payload.isCyberInsuranceAdded = isCyberInsuranceSelected
    } else {
      payload.cyberInsuranceAmt = null
    }
    dispatch(updateReturningBotttomSheetAll(payload))
  }
 
  const onAddRemovePressed = (id, isAmountAdded) => () => {
    const {
      wallet, rentProtect, subscription, inviteCodeDiscount,
      cyberInsurance
    } = paymentSectionKeys
 
    if (id === wallet) {
      const payload = walletPayload({ isAmountAdded })
      dispatch(updateReturningBotttomSheetIndi(payload))
    }
    if (id === rentProtect) {
      /* Rent protect amount will be updated only if conv fees succeeds and we get new data */
      dispatch(CredPayActions.toggleRentProtect(!isRentProtectApplied))
    }
    if (id === subscription) {
      dispatch(toggleSubscriptionSelection())
    }
    if (id === inviteCodeDiscount) {
      const payload = inviteCodeDiscountPayload(isAmountAdded)
      convFeeService.callConvFeeApi()
      batch(() => {
        dispatch(ReferralV2Actions.removeInviteCode())
        dispatch(updateReturningBotttomSheetIndi(payload))
      })
    }
 
    if (id === cyberInsurance) {
      /**
       * Cyber insurance amount will be addedd without any conv. fee api call
       */
      dispatch(setCyberInsuranceCheckBox(!cyberInsuranceData?.isCyberInsuranceSelected))
    }
  }
 
  useEffect(() => {
    performDispatchesPostOpeningModal()
  }, [])
 
  const onButtonPress = () => {
    hideModal()
    onPayRentPressed()
  }
 
  return (
    <PaymentDetailsBottomSheet
      paymentSectionsData={paymentSectionsData}
      onAddRemovePressed={onAddRemovePressed}
      rentTypeId={rentTypeId}
      amount={amount}
      footerComp={(
        <NewTransactionButton fromRetScreen onPayRentPressed={onButtonPress} disabled={disabled} />
      )}
    />
 
  )
}
 
NewTranPaymentBottomSheetWrapper.propTypes = {
  onPayRentPressed: func.isRequired,
  disabled: func.isRequired
}
 
export default memo(NewTranPaymentBottomSheetWrapper)
