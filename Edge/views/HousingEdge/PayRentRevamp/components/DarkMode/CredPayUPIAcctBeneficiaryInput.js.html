import { StyleSheet } from 'react-native'
import React, {
  forwardRef, useImperativeHandle, useRef, useState
} from 'react'
import { useDispatch, useSelector } from 'react-redux'
import PropTypes from 'prop-types'
 
import { otherServicesStrings } from 'common/constants/strings'
import { CredPayV2Actions } from 'views/HousingEdge/CredPay/actions'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import GenericTextInput, { validationMap } from 'components/GenericTextInputV2'
import colors from 'utils/colors'
import fontUtils from 'common/utils/fontUtils'
import { emptyFunc } from 'common/utils/ValidationUtils'
import { removeComaFromAmount } from 'common/utils/formatePrice'
import isEmpty from 'common/utils/isEmpty'
import { CredPayUtils } from 'utils/credPayUtils'
 
const styles = StyleSheet.create({
  inputStyle: {
    color: colors.white,
    fontSize: 18,
    fontFamily: fontUtils.appFontMedium
  }
})
 
const UPI_LIMIT = 100000
 
const propTypes = {
  onReAccountSubmitEditing: PropTypes.func,
  onUPIAcctChangeSubmitEditing: PropTypes.func,
  renderInsideHorizontalFlatlist: PropTypes.bool
}
 
const defaultProps = {
  onReAccountSubmitEditing: emptyFunc,
  onUPIAcctChangeSubmitEditing: emptyFunc,
  renderInsideHorizontalFlatlist: false
 
}
 
const numericRegex = new RegExp('^[0-9]*$')
 
const CredPayUPIAcctBeneficiaryInput = (props, ref) => {
  const {
    onReAccountSubmitEditing, onUPIAcctChangeSubmitEditing,
    renderInsideHorizontalFlatlist
  } = props
 
  const upiAcctNosRef = useRef(null)
  const reEnterAcctNosRef = useRef(null)
  const [suggestion, setSuggestion] = useState(false)
 
  useImperativeHandle(ref, () => ({
    /*  Focus on the field and pass the ref to parent for scrolling
     to the exact place in scrollview */
    focusOnMainText: () => {
      upiAcctNosRef?.current?.focus?.()
      return upiAcctNosRef
    },
    focusOnReEnterAcctText: () => {
      reEnterAcctNosRef?.current?.focus?.()
      return reEnterAcctNosRef
    }
  }))
 
  const dispatch = useDispatch()
 
  const {
    changeAccountType, verifyUpiReq, clearUPIData, updateAccountNos,
    updateReEnterAccountNos, resetIfscAndReEnterAcctNos
  } = CredPayV2Actions
 
  const upiData = useSelector(CredPaySelectors.getUpiData)
  const isUpiValid = useSelector(CredPaySelectors.getIsUpiValid)
  const acctNos = useSelector(CredPaySelectors.getAcctNumber)
  const reEnterAcctNos = useSelector(CredPaySelectors.getReEnterAcctNumber)
  const amount = useSelector(CredPaySelectors.getAmount)
 
  const { upiId, upiName, status } = upiData || {}
 
  const isUpiEligible = () => removeComaFromAmount(amount) <= UPI_LIMIT
 
  const onUPIAcctChangeText = (value) => {
    const isAccountValid = validationMap('accountNumber')(value, '+91') || isEmpty(value)
    const isUpiValueValid = validationMap('upi')(value)
 
    const numeric = numericRegex.test(value)
 
    if (value?.includes?.('@')) {
      setSuggestion(true)
    } else {
      setSuggestion(false)
    }
 
    if (isUpiEligible() && !numeric && !isEmpty(value)) {
      dispatch(changeAccountType(true))
    }
    if ((!isUpiEligible() && isUpiValid) || (!isUpiValueValid && (numeric || isEmpty(value)))) {
      dispatch(changeAccountType(false))
    }
    if (!isEmpty(upiData?.upiName) || upiData?.status === CredPayUtils.reqStatus.error) {
      dispatch(clearUPIData())
    }
    if (
      (isUpiValid && upiAcctNosRef?.current?.isValid)
      || (!isAccountValid && isUpiValueValid && isUpiEligible())
    ) {
      /**
       * verifyUpiReq is not called when changeAccountType action is
       * changing isUpiValid value to true in same iteration hence 2nd condition
       */
      dispatch(verifyUpiReq({ upiId: upiAcctNosRef?.current?.value }))
    }
 
    if (!isAccountValid && isUpiValueValid) {
      /* If account number changes to UPI reset renetered nos and Ifsc Data  */
      dispatch(resetIfscAndReEnterAcctNos())
    }
 
    if (
      !isUpiValid
      && reEnterAcctNosRef?.current?.value
      && (reEnterAcctNosRef?.current?.value !== upiAcctNosRef?.current?.value)
    ) {
      reEnterAcctNosRef?.current?.showError?.(otherServicesStrings.accNumDoNotMatch)
    } else if (
      !isUpiValid
      && reEnterAcctNosRef?.current?.value
      && (reEnterAcctNosRef?.current?.value === upiAcctNosRef?.current?.value)
    ) {
      reEnterAcctNosRef?.current?.hideError?.()
    }
 
    dispatch(updateAccountNos(value))
  }
 
  const onUpiRetry = () => {
    const upiId = upiAcctNosRef?.current?.value
    dispatch(verifyUpiReq({ upiId }))
  }
 
  const getPlaceholder = () => {
    if (isUpiValid) { return otherServicesStrings.beneficiaryUPIid }
    if (!isUpiEligible()) { return otherServicesStrings.bankAccountNumber }
    return otherServicesStrings.enterBenficiaryBankOrUPI
  }
 
  const getErrorText = () => {
    if (isUpiValid) { return 'Please enter a Valid UPI Id' }
    if (!isUpiEligible()) { return 'Please enter a valid account number' }
    return 'Please enter a valid account number or UPI Id'
  }
 
  const onUPIAcctBlur = () => {
    if (!isUpiValid) {
      dispatch(updateAccountNos(upiAcctNosRef?.current?.value))
    }
  }
 
  const renderUPIAccountInputBox = () => (
    <GenericTextInput
      ref={upiAcctNosRef}
      autoCapitalize="none"
      type={isUpiValid ? 'upi' : 'accountNumber'}
      value={isUpiValid ? upiId : acctNos}
      status={isUpiValid ? status : undefined}
      onBlur={onUPIAcctBlur}
      blurOnSubmit={false}
      resourceId="rp_main_input"
      onChangeText={onUPIAcctChangeText}
      keyboardType={!isUpiEligible() ? 'numeric' : 'default'}
      placeholder={getPlaceholder()}
      fontFamily={fontUtils.appFont}
      errorText={getErrorText()}
      inputStyle={styles.inputStyle}
      address={isUpiValid ? upiName : undefined}
      onRetry={isUpiValid ? onUpiRetry : undefined}
      showSuggestion={suggestion}
      onSubmitEditing={onUPIAcctChangeSubmitEditing}
      handleValuePropChange={false}
      darkMode
      renderInsideHorizontalFlatlist={renderInsideHorizontalFlatlist}
    />
  )
 
  const onChangeReEnterAcctNos = (val) => {
    dispatch(updateReEnterAccountNos(val))
  }
 
  const onReAcEndEditing = () => {
    if (reEnterAcctNosRef?.current?.value !== acctNos) {
      reEnterAcctNosRef?.current?.showError?.(otherServicesStrings.accNumDoNotMatch)
    }
  }
 
  const renderReEnterAcctNosInputBox = () => (
    <GenericTextInput
      ref={reEnterAcctNosRef}
      value={reEnterAcctNos}
      onEndEditing={onReAcEndEditing}
      blurOnSubmit={false}
      onChangeText={onChangeReEnterAcctNos}
      keyboardType={!isUpiEligible() ? 'numeric' : 'default'}
      resourceId="rp_reaccount_input"
      autoCapitalize="none"
      secureTextEntry
      inputStyle={styles.inputStyle}
      placeholder={otherServicesStrings.reenterBankAccountNos}
      fontFamily={fontUtils.appFont}
      errorText={otherServicesStrings.accNumDoNotMatch}
      onSubmitEditing={onReAccountSubmitEditing}
      handleValuePropChange={false}
      darkMode
    />
  )
 
  const renderBankAcctBoxes = () => {
    if (isUpiValid) {
      return null
    }
    return renderReEnterAcctNosInputBox()
  }
 
  return (
    <>
      {renderUPIAccountInputBox()}
      {renderBankAcctBoxes()}
    </>
  )
}
 
CredPayUPIAcctBeneficiaryInput.propTypes = propTypes
CredPayUPIAcctBeneficiaryInput.defaultProps = defaultProps
 
const CredPayUPIAcctBeneficiaryInputForwardRefComp = forwardRef(CredPayUPIAcctBeneficiaryInput)
 
export default React.memo(CredPayUPIAcctBeneficiaryInputForwardRefComp)
