import { StatusBar } from 'react-native'
import React, {
  useCallback, useEffect, useMemo, useState
} from 'react'
import { useDispatch, useSelector } from 'react-redux'
import { useNavigation, useIsFocused, useRoute } from '@react-navigation/native'
 
import { EA, EC, EL } from 'constants/GAEvents'
import { CredPayV2Actions } from 'views/HousingEdge/CredPay/actions'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import { UserSelectors } from 'selectors/user'
import CredPayLandingDark from 'views/HousingEdge/PayRentRevamp/screens/CredPayLandingDark'
import { getReferralEnabled } from 'views/Referral/ReferralHelpers'
import { ReferralSelectors } from 'selectors/referral'
import { RewardsSelectors } from 'selectors/rewards'
import { SCREENS } from 'constants/screens'
import { CredPayRewardsUtils } from 'views/HousingEdge/CredPayRewards/rewardsHelper'
import { trackEvent } from 'utils/trackEvents'
import { CredPayUtils } from 'utils/credPayUtils'
import isEmpty from 'common/utils/isEmpty'
import colors from 'utils/colors'
import { CredPayActions } from 'views/HousingEdge/PayRentRevamp/actions'
import { CreditPlusActions } from 'views/HousingEdge/CreditPlus/actions'
import { clearRentProtect, fetchTransactions } from 'views/RentPay/Actions'
import { ReferralV2Actions } from 'views/HousingEdge/Referral/actions'
import { CreditPlusSelectors } from 'views/HousingEdge/CreditPlus/selectors'
import { EMPTY_STATE } from 'common/constants/AppConstants'
import { CyberInsuranceAction } from 'actions/cyberInsuranceAction'
import { PocConfigActions } from 'actions/POCConfigActions'
import { RewardActions } from 'views/HousingEdge/CredPayRewards/actions'
import { createPromiseDispatch } from 'common/utils/CommonUtils'
import useNativeBackHandler from 'common/hooks/useNativeBackHandler'
import { sendReferralViaOptions } from 'views/HousingEdge/Referral/helper'
 
const CredPayDarkModeWrapper = () => {
  const [usedSelectedOptionParam, setUsedSelectedOptionParam] = useState(false)
  const dispatch = useDispatch()
  const navigation = useNavigation()
  const isFocused = useIsFocused()
  const { params = {} } = useRoute()
  const {
    fromRewardsDeeplink = false, selectedOption = {},
    fromReferralLink = false
  } = params || {}
  const isJackpotElgible = useSelector(RewardsSelectors.getIsJackpotEligible)
 
  useNativeBackHandler(navigation, true)
 
  const selectedOptionFromParam = useMemo(() => {
    if (!usedSelectedOptionParam) {
      return selectedOption
    }
    return EMPTY_STATE.OBJECT
  }, [usedSelectedOptionParam])
 
  const clearSelectedOptionParam = useCallback(() => {
    setUsedSelectedOptionParam(true)
  }, [])
 
  /* Actions */
  const {
    updateSessionTime, setRentPaid, fetchRentTypeOptions,
    fetchPaymentModesReq
  } = CredPayV2Actions
 
  /* Selectors */
  const {
    getIsReturningUser, getSessionStartTime, getPaymentModes
  } = CredPaySelectors
  const isReturningUser = useSelector(getIsReturningUser)
  const sessionStartTime = useSelector(getSessionStartTime)
  const paymentModes = useSelector(getPaymentModes)
  const isLoggedIn = useSelector(UserSelectors.getIsLoggedIn)
  const referralCode = useSelector(ReferralSelectors.getReferralCode)
  const expVariant = useSelector(CreditPlusSelectors.getExpVariant)
 
  const fetchReferralDetails = (payload = {}) => {
    /* Will only trigger if user is returning by default */
    const isReferralEnabled = getReferralEnabled()
 
    if (isReferralEnabled && isEmpty(referralCode)) {
      dispatch(ReferralV2Actions.fetchReferralCode(payload))
    }
  }
 
  const triggerRewardsDeeplinkActions = () => {
    if (isJackpotElgible) {
      /* If Logged In & Returing User & JACKPOT_USER fetchTwoSuccessfulpayments
        for Jackpot Card Type & Navigate to REWARDS_PAGE */
      const payload = {
        type: CredPayRewardsUtils.fetchLastTwoSuccPayments,
        cb: () => {
          dispatch(CredPayV2Actions.toggleCredPayLoader(false))
          navigation.navigate(SCREENS.CRED_PAY_REWARDS)
        }
      }
      dispatch(CredPayV2Actions.fetchSuccessfulPaymentsReq(payload))
    } else {
      dispatch(CredPayV2Actions.toggleCredPayLoader(false))
      // If Logged In & Returing User & NON_JACKPOT_USER Navigate to REWARDS_PAGE
      navigation.navigate(SCREENS.CRED_PAY_REWARDS)
    }
  }
 
  const fetchOldTransactions = () => {
    const referralPayload = {
      cb: (err, data) => {
        if (!err && !isEmpty(data)) {
          if (fromReferralLink) {
            const { code, referralMessage } = data || {}
            sendReferralViaOptions(code, referralMessage)
          }
        }
      }
    }
    const payload = {
      callback: (err, data) => {
        if (!err && !isEmpty(data)) {
          fetchReferralDetails(referralPayload)
          if (fromRewardsDeeplink) {
            /* For Returning User from Rewards Deeplink */
            triggerRewardsDeeplinkActions()
            return
          }
        } else {
          /* No order data available.Reroute to new flow and ask user to login */
          /* For New user from Rewards Deep link. Stay Here. */
          createPromiseDispatch(RewardActions.fetchFirstPaymentCashback, dispatch)
            .then((res) => {
              trackEvent(EC.POC_WRAPPER, EA.NEW_USER_CB_EXP, { [EL.VARIANT]: res })
            })
          dispatch(setRentPaid(false))
        }
        dispatch(CredPayV2Actions.toggleCredPayLoader(false))
      }
    }
    dispatch(CredPayV2Actions.fetchTransactionsForUser(payload))
  }
 
  const fetchRentPayTransactions = () => {
    const payload = {
      start: 0,
      rows: 10
    }
    dispatch(fetchTransactions(payload))
  }
 
  const initialActions = () => {
    dispatch(fetchRentTypeOptions())
    isEmpty(paymentModes) && dispatch(fetchPaymentModesReq())
    dispatch(ReferralV2Actions.fetchReferralConfig())
    dispatch(PocConfigActions.fetchDiscountsConfig())
    trackEvent(EC.POC_WRAPPER, EA.CP_RP_EXP, CredPayUtils.getCreditPlusExpLabel(expVariant))
  }
 
  useEffect(() => {
    initialActions()
  }, [])
 
  const fetchPocSubscriptionDetails = () => {
    dispatch(CreditPlusActions.getCredPaySubscriptionData())
  }
 
  useEffect(() => {
    if (isLoggedIn && isFocused) {
      dispatch(CredPayV2Actions.toggleCredPayLoader(true))
      fetchOldTransactions()
      fetchRentPayTransactions()
      dispatch(CredPayActions.fetchRPActiveTransactions())
      fetchPocSubscriptionDetails()
      // clear rent protect order id on load
      dispatch(clearRentProtect())
      dispatch(CredPayActions.fetchRentProtectDetails())
      // calling cyber insurance api in order to have data
      dispatch(CyberInsuranceAction.fetchCyberInsurance())
    }
 
    isEmpty(sessionStartTime) && dispatch(updateSessionTime(new Date().toISOString()))
  }, [isFocused, isLoggedIn])
 
  return (
    <>
      <StatusBar backgroundColor={colors.black37} animated />
      <CredPayLandingDark
        navigation={navigation}
        isReturningUser={isReturningUser}
        selectedOption={selectedOptionFromParam}
        clearSelectedOptionParam={clearSelectedOptionParam}
      />
    </>
  )
}
 
export default CredPayDarkModeWrapper
