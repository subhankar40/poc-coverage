import {
  Pressable, SafeAreaView, View
} from 'react-native'
import React, {
  useCallback, useEffect, useMemo, useRef, useState
} from 'react'
import PropTypes from 'prop-types'
import BottomSheet from '@gorhom/bottom-sheet'
import { useSelector, useDispatch } from 'react-redux'
import { ScrollView } from 'react-native-gesture-handler'
 
import { SimpleText } from 'common/components/SimpleText'
import ImageComponent from 'common/components/ImageComponent'
import { EA, EC } from 'constants/GAEvents'
import { UserSelectors } from 'selectors/user'
import { capitalizeFirstLetter } from 'utils/helpers'
import colors from 'utils/colors'
import { trackEvent } from 'utils/trackEvents'
import images from 'utils/images'
import CredPayOptionsList from 'views/HousingEdge/PayRentRevamp/components/CredPayOptionsList'
import CredPayUserTransactionCardList from 'views/HousingEdge/PayRentRevamp/components/CredPayUserTransactionCardList'
import ReferCTA from 'views/HousingEdge/Referral/components/ReferCTA'
import PocHeader from 'views/HousingEdge/PayRentRevamp/components/PocHeader'
import CredPayDarkImageBackground from 'views/HousingEdge/PayRentRevamp/components/CredPayDarkImageBackground'
import OffersStrip from 'views/HousingEdge/PayRentRevamp/components/OffersStrip/OffersStrip'
import { CredPaySelectors } from 'views/HousingEdge/PayRentRevamp/selectors'
import DevMenuToggler from 'views/HousingEdge/PayRentRevamp/components/DevMenuToggler'
import CustomBackdrop from 'views/HousingEdge/PayRentRevamp/components/BackDropBottomSheet'
import { CredPayV2Actions } from 'views/HousingEdge/CredPay/actions'
import CreditPlusCTA from 'views/HousingEdge/CreditPlus/components/CreditPlusCTA'
import CtaStrip from 'views/HousingEdge/PayRentRevamp/components/CtaStrip'
import SaferAccessUtilityWrapper from 'views/HousingEdge/PayRentRevamp/components/SaferAccessUtilityWrapper'
import HooksCard from 'views/HousingEdge/PayRentRevamp/components/HooksCard'
import styles from 'views/HousingEdge/PayRentRevamp/styles/CredPayReturningUserDark.styles'
import { showModal } from 'components/ModalBox'
import isEmpty from 'common/utils/isEmpty'
import InsuranceCta from 'views/HousingEdge/RentProtect/InsuranceCta'
import { TraceKeys, stopTrace } from 'utils/trace/traces'
import RetPaymentBottomSheetWrapper from '../components/DarkMode/PaymentBottomSheet/RetPaymentBottomSheetWrapper'
 
const CredPayReturningUserDark = (props) => {
  const {
    onPayPress, onRentCategoryPressed, navigation, pocSubscription,
    selectedOption, clearSelectedOptionParam
  } = props
 
  let credPayOptionListItemsCount = useSelector(CredPaySelectors.getCredPayOptions)
  credPayOptionListItemsCount = credPayOptionListItemsCount?.length || 0
 
  const bottomSheetRef = useRef(null)
 
  const [upArrow, setUpArrow] = useState(true)
 
  // variables
  const snapPoints = useMemo(() => [50, credPayOptionListItemsCount > 6 ? '80%' : '55%'], [])
  const dispatch = useDispatch()
  // callbacks
  const handleSheetChanges = useCallback((index) => {
    if (index === 0) {
      setUpArrow(true)
    } else if (index === 1) {
      setUpArrow(false)
    }
  }, [])
 
  const onSheetAnimate = useCallback((fromIndex, toIndex) => {
    if (fromIndex === 1 && toIndex === 0) {
      trackEvent(EC.POC_HOME_PAGE, EA.POC_CLOSED)
    } else if (fromIndex === 0 && toIndex === 1) {
      trackEvent(EC.POC_HOME_PAGE, EA.POC_OPENED)
    }
  }, [])
 
  const handleSnapPress = useCallback((index) => {
    bottomSheetRef.current?.snapToIndex(index)
  }, [])
 
  const userName = useSelector(state => UserSelectors.getUserName(state))
  const onCategoryPressed = useCallback((item) => {
    dispatch(CredPayV2Actions.updateNewTagForCredPayOptions(item.id))
    handleSnapPress(0)
    onRentCategoryPressed(item)
  }, [])
 
  useEffect(() => {
    if (!isEmpty(selectedOption)) {
      clearSelectedOptionParam()
      onCategoryPressed(selectedOption)
    }
  }, [])
 
  const onReturningHeaderPressed = useCallback(() => {
    upArrow ? handleSnapPress(1) : handleSnapPress(0)
  }, [upArrow])
 
  const renderUserName = useMemo(() => {
    const text = `Hi ${capitalizeFirstLetter(userName)}!`
    return (
      <SimpleText
        text={text}
        style={{ fontSize: 24, lineHeight: 28, color: colors.white }}
      />
    )
  }, [userName])
 
  useEffect(() => {
    const timer = setTimeout(() => {
      trackEvent(EC.POC_HOME_PAGE, EA.OPENED)
    }, 150)
    return () => {
      clearTimeout(timer)
    }
  }, [])
 
  const renderBottomSheetHandle = useCallback(() => (
    <Pressable
      style={styles.arrowContainer}
      onPress={onReturningHeaderPressed}
    >
      <ImageComponent
        source={upArrow ? images.greyArrowUpSmall : images.arrowGreyDown}
        style={styles.upArrow}
      />
    </Pressable>
  ), [upArrow])
 
  const renderCustomBackDrop = useCallback(props => (
    <CustomBackdrop
      {...props}
      isOpened={!upArrow}
      backdropContainerStyle={styles.backdropContainerStyles}
    />
  ), [upArrow])
 
  const onPaymentDetailsButtonPressed = (item, idx) => () => {
    showModal({
      renderedComponent: () => (
        <RetPaymentBottomSheetWrapper
          selectedCardIdx={idx}
          item={item}
          onPayPress={onPayPress}
        />
      ),
      hideNativeTabBar: true
    })
  }
 
  const renderBottomSheet = useMemo(() => (
    <BottomSheet
      ref={bottomSheetRef}
      index={0}
      snapPoints={snapPoints}
      onChange={handleSheetChanges}
      onAnimate={onSheetAnimate}
      backgroundStyle={styles.bottomSheetBgStyle}
      backdropComponent={renderCustomBackDrop}
      handleComponent={renderBottomSheetHandle}
    >
      <CredPayOptionsList
        isReturningUser
        onPress={onCategoryPressed}
        onReturningHeaderPressed={onReturningHeaderPressed}
        insideBottomSheet
      />
    </BottomSheet>
  ), [upArrow])
 
  const stopSeeAllTrace = () => {
    stopTrace(TraceKeys.SEE_ALL)
  }
 
  return (
    <SafeAreaView style={styles.mainContainer}>
      <ScrollView contentContainerStyle={styles.contentContainerStyles}>
        <CredPayDarkImageBackground pocSubscription={pocSubscription}>
          <View style={styles.devMenuToggler}>
            <DevMenuToggler />
          </View>
          <View style={styles.container} onLayout={stopSeeAllTrace}>
            <PocHeader pocSubscription={pocSubscription} />
            <View style={styles.userNameContainer}>
              {renderUserName}
            </View>
          </View>
          <CredPayUserTransactionCardList
            onPayPress={onPayPress}
            onPaymentDetailsButtonPressed={onPaymentDetailsButtonPressed}
          />
          <CtaStrip />
          <OffersStrip />
          <ReferCTA fromScreen={EC.POC_HOME_PAGE} />
          <CreditPlusCTA hideNativeTabBar />
          <InsuranceCta />
          <HooksCard />
        </CredPayDarkImageBackground>
      </ScrollView>
      {renderBottomSheet}
      {/* TODO: Utility wrapper can be moved to root level */}
      <SaferAccessUtilityWrapper navigation={navigation} />
    </SafeAreaView>
  )
}
 
CredPayReturningUserDark.propTypes = {
  onPayPress: PropTypes.func.isRequired,
  onRentCategoryPressed: PropTypes.func.isRequired,
  navigation: PropTypes.object.isRequired,
  pocSubscription: PropTypes.bool.isRequired,
  selectedOption: PropTypes.object.isRequired,
  clearSelectedOptionParam: PropTypes.func.isRequired
}
 
export default CredPayReturningUserDark
