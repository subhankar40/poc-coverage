import React, { useState, useEffect } from 'react'
import {
  View, TouchableOpacity, UIManager, LayoutAnimation
} from 'react-native'
import {
  bool, func, number, object, string
} from 'prop-types'
import HighlightText from '@sanar/react-native-highlight-text'
import ImageComponent from 'common/components/ImageComponent'
import images from 'utils/images'
import isEmpty from 'common/utils/isEmpty'
import resourceID from 'common/utils/resourceID'
import { isAndroid } from 'utils/TypeChecker'
import { getEdgeLandingFaqResourceId } from 'views/RentPay/RentPayHelpers'
import { conditionalStyles, styles } from './styles/ExpandableCardWithHighlight'
 
if (
  isAndroid
    && UIManager.setLayoutAnimationEnabledExperimental
) {
  UIManager.setLayoutAnimationEnabledExperimental(true)
}
 
const QuestionTitle = ({
  ques, faqTabName, index,
  isExpanded, darkMode,
  handleClick, searchWords, highlightStyle
}) => {
  const resourceId = getEdgeLandingFaqResourceId(faqTabName, index, isExpanded)
  const titleStyles = conditionalStyles({ darkMode, isExpanded }).title
  return (
    <TouchableOpacity
      activeOpacity={0.8}
      style={styles.titleButton}
      {...{ [resourceID]: resourceId }}
      onPress={handleClick}
    >
      <HighlightText
        searchWords={[searchWords]}
        highlightStyle={[styles.highlightStyle, highlightStyle]}
        textToHighlight={ques}
        style={titleStyles}
      />
      <ImageComponent
        style={styles.plusImage}
        source={isExpanded ? images.minusIconFaq : images.plusIconFaq}
      />
    </TouchableOpacity>
  )
}
 
QuestionTitle.propTypes = {
  ques: string.isRequired,
  index: number.isRequired,
  faqTabName: string.isRequired,
  searchWords: string.isRequired,
  darkMode: bool.isRequired,
  highlightStyle: object.isRequired,
  isExpanded: bool.isRequired,
  handleClick: func.isRequired
}
 
const ExpandableCardWithHighlight = ({
  faqTabName, index, item, trackEvent, highlightStyle,
  showDivider, darkMode, searchWords
}) => {
  const [isExpanded, setIsExpanded] = useState(false)
 
  const condStyles = conditionalStyles({ darkMode }).divider
 
  // expand or collapsed based on searchWords update
  useEffect(() => {
    if (searchWords) {
      setIsExpanded(true)
    } else {
      setIsExpanded(false)
    }
  }, [searchWords])
 
  const getQus = () => !isEmpty(item.qus) ? item.qus : item.question
 
  const handleClick = () => {
    LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut)
    setIsExpanded(isExpanded => !isExpanded)
    trackEvent(isExpanded, { question: getQus() })
  }
 
  const renderSubTitle = (ans) => {
    const condStyles = conditionalStyles({ darkMode, isExpanded })
    return (
      <View style={condStyles.ansContainer}>
        <HighlightText
          searchWords={[searchWords]}
          highlightStyle={[styles.highlightStyle, highlightStyle]}
          textToHighlight={ans}
          style={condStyles.ansText}
        />
      </View>
    )
  }
 
  const getAns = () => !isEmpty(item.ans) ? item.ans : item.answer
 
  return (
    <View>
      <QuestionTitle
        ques={getQus()}
        faqTabName={faqTabName}
        index={index}
        isExpanded={isExpanded}
        darkMode={darkMode}
        handleClick={handleClick}
        searchWords={searchWords}
        highlightStyle={highlightStyle}
      />
      {renderSubTitle(getAns())}
      {showDivider && <View style={condStyles} />}
    </View>
  )
}
 
ExpandableCardWithHighlight.propTypes = {
  item: object,
  showDivider: bool,
  trackEvent: func,
  index: number,
  faqTabName: string,
  searchWords: string,
  darkMode: bool,
  highlightStyle: object
}
 
ExpandableCardWithHighlight.defaultProps = {
  item: {
    qus: '',
    ans: ''
  },
  index: 0,
  showDivider: true,
  trackEvent: () => { },
  faqTabName: '',
  searchWords: '',
  darkMode: false,
  highlightStyle: {}
}
 
export default React.memo(ExpandableCardWithHighlight)
