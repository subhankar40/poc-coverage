import React, { PureComponent, createRef } from 'react'
import {
  View, ScrollView, StyleSheet, findNodeHandle
} from 'react-native'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import { bindActionCreators } from 'redux'
 
import GenericTextInput from 'components/GenericTextInputV2'
import { SimpleText } from 'common/components/SimpleText'
import KeyboardSpacer from 'common/components/KeyboardSpacer'
import DropdownTextInput from 'common/components/DropdownTextInput'
import { otherServicesStrings } from 'common/constants/strings'
import { EA, EC } from 'constants/GAEvents'
import { SolarRooftopConstants } from 'constants/solarRooftopConstants'
import ServiceHorizontalList, { itemComponentTypes } from 'views/HousingEdge/components/ServiceHorizontalList'
import { SolarRooftopActions } from 'views/HousingEdge/SolarRooftop/actions'
import colors from 'utils/colors'
import fontUtils from 'common/utils/fontUtils'
import isEmpty from 'common/utils/isEmpty'
import { trackEvent } from 'utils/trackEvents'
 
const styles = StyleSheet.create({
  inputStyle: {
    color: colors.black0,
    fontSize: 18,
    fontFamily: fontUtils.appFontMedium
  },
  vendorsErrorText: {
    color: colors.red49,
    textAlign: 'center',
    marginTop: 20,
    marginHorizontal: 12
  }
})
 
const GACat = EC.SOLAR_ROOFTOP_FLOW
export class SRPropertyDetails extends PureComponent {
  constructor(props) {
    super(props)
    this.scrollRef = createRef()
    this.cityInputRef = createRef()
    this.otherCityNameRef = createRef()
    this.propertyTypeRef = createRef()
    this.commentsRef = createRef()
    this.vendorRef = createRef()
  }
 
  // Trigger validation logic everytime vendors change
  componentDidUpdate = (prevProps) => {
    if (prevProps.vendors.length !== this.props.vendors.length) {
      this.triggerisFormDataInvalid()
    }
  }
 
  scrollToInput = () => {
    this.activeInputRef.current?.measureLayout(
      findNodeHandle(this.scrollRef.current),
      (_, y) => {
        this.scrollRef.current?.scrollTo({ y: y - 30 })
      }
    )
  }
 
  onFocusFields = (ref) => {
    const { isKeyboardVisible } = this.props
    if (!isEmpty(ref) && this.scrollRef.current && ref.current?.isFocused) {
      if (isKeyboardVisible) {
        this.handleScroll = false
        this.activeInputRef = ref
        this.scrollToInput()
      } else {
        this.handleScroll = true
        this.activeInputRef = ref
      }
    }
  }
 
  // Validation -
 
  isCitySelectedInvalid = () => {
    const { current } = this.cityInputRef
    if (!current || !current?.value) {
      return true
    }
    return false
  }
 
  isOtherCityInvalid = () => {
    const { current } = this.otherCityNameRef
    if (this.props.cityName !== SolarRooftopConstants.Other) {
      return false
    }
    if (!current || !current?.isValid) {
      return true
    }
    return false
  }
 
  isPropertySelectedInvalid = () => {
    const { current } = this.propertyTypeRef
    if (!current || this.props.propertyTypeSelected === null) {
      return true
    }
    return false
  }
 
  isVendorSelected = () => {
    const { vendors } = this.props
    const selected = vendors?.find(e => e.isSelected)
    if (selected) {
      return true
    }
    return false
  }
 
  isVendorInvalid = () => {
    const { current } = this.vendorRef
    if (!current || !this.isVendorSelected()) {
      return true
    }
    return false
  }
 
  isFormDataInValid = () => {
    if (this.isCitySelectedInvalid() || this.isOtherCityInvalid()
      || this.isPropertySelectedInvalid() || this.isVendorInvalid()) {
      return true
    }
    return false
  }
 
  showErrorMessage = () => {
    if (this.isCitySelectedInvalid()) {
      this.cityInputRef?.current?.showError()
    }
 
    if (this.isOtherCityInvalid()) {
      this.otherCityNameRef?.current?.showError()
    }
 
    if (this.isPropertySelectedInvalid()) {
      this.propertyTypeRef?.current?.handleError(true)
    }
 
    if (this.isVendorInvalid()) {
      this.vendorRef?.current?.handleError(true)
    }
  }
 
  // Save block logic -
 
  saveStep1 = (payload = {}) => {
    const { saveSolarRooftopStepOneData } = this.props
    saveSolarRooftopStepOneData(payload)
  }
 
  areVendorsSelected = () => {
    const { vendors } = this.props
 
    const vendorIds = []
    vendors.forEach((e) => {
      if (e.isSelected) {
        vendorIds.push({ id: e?.id, name: e?.name })
      }
    })
    return vendorIds
  }
 
  onSavePress = () => {
    if (this.isFormDataInValid()) {
      return
    }
    const {
      trackingLabels, cityName, otherCityName,
      propertyTypeSelected
    } = this.props
    const payload = {
      vendorIds: this.areVendorsSelected()
    }
 
    const trackingPayload = {
      city: cityName,
      otherCity: otherCityName,
      propertyType: propertyTypeSelected,
      vendor: this.areVendorsSelected(),
      ...trackingLabels()
    }
 
    trackEvent(GACat, EA.LANDING_FORM_CLICKED, trackingPayload)
    this.saveStep1(payload)
  }
 
  triggerisFormDataInvalid = () => {
    const { isDataValid, updateIsCardDataValid } = this.props
    const isValid = !this.isFormDataInValid()
    if (isDataValid !== isValid) {
      updateIsCardDataValid({ key: 'solarRooftopStep1', value: isValid })
    }
  }
 
  emptyCityDataError = () => {
    if (this.cityInputRef?.current) {
      this.cityInputRef?.current?.showError('Something went wrong. Please try again later.')
    }
  }
 
  triggerVendorsCall = (data = {}) => {
    const { houseType = null, otherCitySelected = false } = data
    const { fetchVendors, cityUuid } = this.props
    const payload = {
      cityUuid,
      houseType,
      otherCitySelected
    }
    fetchVendors(payload)
  }
 
  /* When city changes trigger update city and cityUuid in redux
  and post that trigger vendors call async */
  onCityChange = (item) => {
    const { cityName, cityUuid } = item
    const { saveSolarRooftopStepOneData, onCityChanged } = this.props
    const payload = {
      cityName,
      cityUuid
    }
    onCityChanged && onCityChanged()
    saveSolarRooftopStepOneData(payload)
      .then(() => {
        if (this.props.cityName === SolarRooftopConstants.Other) {
          this.triggerVendorsCall({ otherCitySelected: true })
          return
        }
        this.triggerVendorsCall()
      })
  }
 
  renderCityDropDown = () => {
    const { cities, cityName } = this.props
    return (
      <DropdownTextInput
        ref={this.cityInputRef}
        value={cityName}
        placeholder={otherServicesStrings.selectCity}
        inputStyle={styles.inputStyle}
        listData={cities}
        emptyDataErrorCallback={this.emptyCityDataError}
        onValueChange={this.onCityChange}
        errorText={otherServicesStrings.oopsForgotCity}
      />
    )
  }
 
  /* Update city name directly into redux */
  otherCityNameChange = (val) => {
    const { saveSolarRooftopStepOneData } = this.props
    const payload = {
      otherCityName: val
    }
    saveSolarRooftopStepOneData(payload)
      .then(() => this.triggerisFormDataInvalid())
  }
 
  otherCityNameFocus = () => {
    this.onFocusFields(this.otherCityNameRef)
  }
 
  /* Other city textinput to be shown only when city dropdown is set to others.
   */
  renderOtherCityName = () => {
    const { cityName } = this.props
    if (cityName !== SolarRooftopConstants.Other) {
      return null
    }
    const { otherCityName } = this.props
    return (
      <View style={{ marginTop: 3 }}>
        <GenericTextInput
          ref={this.otherCityNameRef}
          type="name"
          value={otherCityName}
          autoCapitalize="none"
          autoComplete={false}
          onFocus={this.otherCityNameFocus}
          onChangeText={this.otherCityNameChange}
          blurOnSubmit={false}
          placeholder={otherServicesStrings.enterOtherCity}
          fontFamily={fontUtils.appFont}
          errorText={otherServicesStrings.enterCityName}
          inputStyle={styles.inputStyle}
          handleValuePropChange={false}
        />
      </View>
    )
  }
 
  /* Trigger Vendors API call when property type is changed
  and also update the toggle value of property type */
  onPropertyTypeChanged = (item) => {
    const { updateGridListValue, cityName } = this.props
    const payload = {
      parentKey: 'solarRooftopStep1',
      key: 'propertyTypeSelected',
      value: item.text
    }
    updateGridListValue(payload)
    const data = {
      houseType: item.text,
      otherCitySelected: cityName === SolarRooftopConstants.Other
    }
    this.triggerVendorsCall(data)
  }
 
  /* Post vendor is toggled , setToggle value and
  post that trigger validation check upadting button color */
  onVendorChanged = (item) => {
    const { toggleIsVendorSelected } = this.props
    toggleIsVendorSelected(item)
      .then(() => {
        this.triggerisFormDataInvalid()
      })
  }
 
  renderPropertyTypes = () => {
    const { propertyTypeSelected } = this.props
    return (
      <ServiceHorizontalList
        ref={this.propertyTypeRef}
        heading={otherServicesStrings.choosePropertyType}
        isSelected={propertyTypeSelected}
        data={SolarRooftopConstants.propertyTypes}
        showImage
        itemChangedCb={this.onPropertyTypeChanged}
        headingTextStyle={{ color: colors.grey21 }}
        errorText={otherServicesStrings.pleaseSelectAProperty}
      />
    )
  }
 
  /* If no vendors and API is in progress show null
   If no vendors at all ,diplay error message
   If vendors available show list */
  renderVendors = () => {
    const { vendors, vendorId, vendorsLoading } = this.props
    if (vendorsLoading) {
      return null
    }
    if (isEmpty(vendors)) {
      return (
        <SimpleText
          text={otherServicesStrings.selectADifferentCombination}
          style={styles.vendorsErrorText}
        />
      )
    }
    return (
      <ServiceHorizontalList
        ref={this.vendorRef}
        heading={otherServicesStrings.selectVendor}
        isSelected={vendorId}
        data={vendors}
        showImage
        imageType={itemComponentTypes.iconWithCheckBox}
        itemChangedCb={this.onVendorChanged}
        headingTextStyle={{ color: colors.grey21 }}
        errorText={otherServicesStrings.pleaseSelectAVendor}
 
      />
    )
  }
 
  onCommentsChanged = (val) => {
    this.saveStep1({ comments: val })
  }
 
  onCommentsFocused = () => {
    this.onFocusFields(this.commentsRef)
  }
 
  renderComments = () => {
    const { comments, vendors } = this.props
    if (isEmpty(vendors)) {
      return (
        null
      )
    }
    return (
      <View>
        <GenericTextInput
          type="name"
          value={comments}
          autoCapitalize="none"
          autoComplete={false}
          onFocus={this.onCommentsFocused}
          onChangeText={this.onCommentsChanged}
          blurOnSubmit={false}
          placeholder={otherServicesStrings.addIfAnyOtherSpecifications}
          fontFamily={fontUtils.appFont}
          inputStyle={styles.inputStyle}
          handleValuePropChange={false}
        />
      </View>
    )
  }
 
  renderContent = () => {
    const { cityName } = this.props
    return (
      <View>
        {this.renderCityDropDown()}
        {this.renderOtherCityName()}
        {(!isEmpty(cityName)) && (
          <View>
            {this.renderPropertyTypes()}
            {this.renderVendors()}
            {this.renderComments()}
          </View>
        )}
      </View>
    )
  }
 
  render() {
    const ht = this.props.isKeyboardVisible ? 400 : 150
    return (
      <ScrollView
        keyboardShouldPersistTaps="handled"
        ref={this.scrollRef}
        style={{ padding: 24 }}
      >
        {this.renderContent()}
        <View style={{ height: ht }} />
        <KeyboardSpacer forceAndroid />
      </ScrollView>
    )
  }
}
 
SRPropertyDetails.propTypes = {
  cities: PropTypes.array.isRequired,
  isDataValid: PropTypes.bool.isRequired,
  isKeyboardVisible: PropTypes.bool.isRequired,
  cityUuid: PropTypes.oneOfType([PropTypes.string, null]).isRequired,
  cityName: PropTypes.oneOfType([PropTypes.string, null]).isRequired,
  vendorId: PropTypes.oneOfType([PropTypes.number, null]).isRequired,
  otherCityName: PropTypes.oneOfType([PropTypes.string, null]).isRequired,
  propertyTypeSelected: PropTypes.oneOfType([PropTypes.string, null]).isRequired,
  updateGridListValue: PropTypes.func.isRequired,
  fetchVendors: PropTypes.func.isRequired,
  vendors: PropTypes.array.isRequired,
  comments: PropTypes.string.isRequired,
  updateIsCardDataValid: PropTypes.func.isRequired,
  saveSolarRooftopStepOneData: PropTypes.func.isRequired,
  toggleIsVendorSelected: PropTypes.func.isRequired,
  vendorsLoading: PropTypes.bool.isRequired,
  onCityChanged: PropTypes.func.isRequired,
  trackingLabels: PropTypes.func.isRequired
}
 
const mapStateToProps = ({ solarRooftop }) => ({
  cities: solarRooftop.cities,
  vendors: solarRooftop.vendors,
  vendorsLoading: solarRooftop.loaders.vendors,
  ...solarRooftop.solarRooftopStep1
})
 
const mapDispatchToProps = (dispatch) => {
  const {
    updateGridListValue, fetchVendors, toggleIsVendorSelected,
    saveSolarRooftopStepOneData, updateIsCardDataValid
 
  } = SolarRooftopActions
 
  return bindActionCreators({
    ...dispatch,
    updateGridListValue,
    fetchVendors,
    toggleIsVendorSelected,
    saveSolarRooftopStepOneData,
    updateIsCardDataValid
  }, dispatch)
}
 
export default connect(mapStateToProps, mapDispatchToProps,
  null, { forwardRef: true })(SRPropertyDetails)
