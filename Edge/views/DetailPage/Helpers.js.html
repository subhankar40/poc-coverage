export function getDifferenceInDays(dateAdded) {
  const currentTime = Date.now() // in milli seconds
  const propertyAddedTime = new Date(dateAdded).getTime() // in milli seconds
  return Math.floor((currentTime - propertyAddedTime) / (1000 * 60 * 60 * 24))
}
 
export function parseDateAdded(dateAdded) {
  const diffInDays = getDifferenceInDays(dateAdded)
  if (diffInDays <= 0) { return 'just now' }
  if (diffInDays === 1) { return '1 day ago' }
  if (diffInDays <= 30) { return `${diffInDays} days ago` }
  return 'more than a month ago'
}
 
const getLatLong = (property) => {
  const { latitude, longitude, location_coordinates } = property
  if (!isEmpty(latitude) && !isEmpty(longitude)) {
    return { latitude, longitude }
  } if (!isEmpty(location_coordinates) && typeof location_coordinates === 'string') {
    const coordinates = location_coordinates.split(',')
    if (coordinates.length > 1) {
      const lat = !isEmpty(coordinates[0]) ? parseFloat(coordinates[0]) : null
      const long = !isEmpty(coordinates[1]) ? parseFloat(coordinates[1]) : null
      return { latitude: lat, longitude: long }
    }
  }
  return {}
}

const getArea = (property) => {
  let area = ''
  const {
    inventory_configs, plot_area, unit_of_area, type
  } = property || {}
  if (inventory_configs instanceof Array && !isEmpty(inventory_configs?.[0])) {
    const { property_type, area: inventoryArea, selected_area } = inventory_configs[0]
    const propertyType = property_type?.toLowerCase()
    const areaUnit = unit_of_area || SQ_FT
    const formattedAreaUnit = `${areaUnit} built up area`
    if (propertyType === PROPERTY_TYPE.PLOT) {
      if (!isEmpty(plot_area)) area = `${plot_area} ${formattedAreaUnit}`
    } else if (inventoryArea) {
      area = `${inventoryArea} ${formattedAreaUnit}`
    }
    if (isEmpty(area) && !isRent(type) && !isEmpty(selected_area)) {
      area = `${selected_area} ${formattedAreaUnit}`
    }
  }
  if (isEmpty(area)) {
    const {
      carpet_area: carpetArea, super_built_up_area: superArea
    } = property || {}
    if (!isEmpty(carpetArea) || !isEmpty(superArea)) {
      area = `${carpetArea || superArea} sq.ft.`
    }
  }
  return area
}

      name,
      profile_picture_url,
      profile_uuid,
      seller_label: service === 'pg' ? property_manager_type_display : seller_label,

      contact_person_id,

    subTitle: type === 'project' ? getDeveloperName(developer) : getArea(property),

function arrIndex(tags, array) {
  if (!tags || !tags.length) return false
  const min = tags.map(tag => array[tag]).reduce((currMin, currVal) => Math.min(currMin, currVal))
  if (min) return min
  return false
}
 
function withinConfigCompare(a, b) {
  const afloor = arrIndex(a.tags, CONFIG_TAG_ORDER_FLOORS)
  const bfloor = arrIndex(b.tags, CONFIG_TAG_ORDER_FLOORS)
  if (afloor && bfloor && afloor !== bfloor) { return afloor > bfloor }
  const aview = arrIndex(a.tags, CONFIG_TAG_ORDER_VIEW)
  const bview = arrIndex(b.tags, CONFIG_TAG_ORDER_VIEW)
  return aview > bview
}

export function imagesExtractorNP(response, resolution) {
  let items = []
  let index = 0
  const { cover_photo_url, gallery_images, inventory_configs } = response
 
  // COVER IMAGE
  if (cover_photo_url) {
    items.push({
      lowResSrc: cover_photo_url.replace(/version/, '_p'),
      src: cover_photo_url.replace(/version/, resolution),
      category: PROJECT_IMAGES_TAG,
      caption: null,
      index,

    })

  }
 
  // GALLERY IMAGES
  if (gallery_images) {
    items = items.concat(
      gallery_images.reduce((result, image) => {

          const galleryImage = {
            lowResSrc: absolute_url ? absolute_url.replace(/version/, '_p') : image.replace(/version/, '_p'),
            src: absolute_url ? absolute_url.replace(/version/, resolution) : image.replace(/version/, resolution),
            category: PROJECT_IMAGES_TAG,
            caption: null,
            index,
            id: index
          }
          index += 1
          result.push(galleryImage)
        }
        return result
      }, [])
    )
  }
 
  // INVENTORY CONFIG
  if (inventory_configs && inventory_configs.length) {
    inventory_configs.forEach((config) => {
      const floorPlans = config.floor_plan_urls
      if (!floorPlans || !floorPlans.length) { return }
 
      // First Add 'Ground Floor' in config tag
      let addGround = false
      floorPlans.forEach((_, indx) => {
        if (floorPlans[indx].tags.length > 1) { addGround = true }
      })
 
      if (addGround) {
        floorPlans.forEach((floorPlan) => {
          if (floorPlan.tags.length <= 1) { floorPlan.tags.push(GROUND_FLOOR_CONFIG_TAG) }
        })
      }
 
      // Sort Config's Floor Plans
      floorPlans.sort(withinConfigCompare)
      const areaUnit = getAreaUnit(response)
      const configTag = getFloorPlanArea(config, areaUnit)
      items = items.concat(
        floorPlans.map((floorPlan) => {
          const galleryImage = {
            lowResSrc: floorPlan.floor_plan_url.replace(/version/, '_p'),
            src: floorPlan.floor_plan_url.replace(/version/, resolution),
            category: configTag,
            caption: `${floorPlan.tags.join(' - ')} floor plan`,
            index,
            id: index
          }
          index += 1
          return galleryImage
        })
      )
    })
  }
  return items
}

                isPhotoUnderReview: checkIsPhotoUnderReview({
                  photoUnderReview: photo_under_review, caption: image.caption
                })
              })
              index += 1
            })
          }
        })
      }
    })
  }
  return galleryItems
}
 
export function imagesExtractorForOthers(response, resolution, addInventoryConfigItems) {
  let items = []
  let index = 0
  let coverPhotoUrlIndex = null
  const {
    thumb_image_url, cover_photo_url, images_hash, inventory_configs, photo_under_review
  } = response
  if (thumb_image_url) {
    items.push({
      lowResSrc: thumb_image_url.replace(/version/, '_p'),
      src: thumb_image_url.replace(/version/, resolution),
      category: PROJECT_IMAGES_TAG,
      caption: null,
      index,
      id: index
    })
    index += 1
  }
  if (cover_photo_url && (isEmpty(thumb_image_url) || thumb_image_url !== cover_photo_url)) {
    items.push({
      lowResSrc: cover_photo_url.replace(/version/, '_p'),
      src: cover_photo_url.replace(/version/, resolution),
      category: PROJECT_IMAGES_TAG,
      caption: null,
      index,
      id: index
    })
    coverPhotoUrlIndex = index
    index += 1
  }
  const imagesHashKeys = ['flat_images', 'project_images', 'building_images']
  imagesHashKeys.forEach((eachKey) => {
    if (!isEmpty(images_hash[eachKey])) {
      images_hash[eachKey].forEach((categoryData) => {
        if (!isEmpty(categoryData.images)) {
          categoryData.images.forEach((image) => {
            const isThumbImageUrlDifferent = image.absolute_url !== thumb_image_url
            const isCoverPhotoUrlDifferent = image.absolute_url !== cover_photo_url
            // for resale properties thumb_image_url and first image of flat_images are same
            if ((isEmpty(thumb_image_url) || isThumbImageUrlDifferent)
              && (isEmpty(cover_photo_url) || isCoverPhotoUrlDifferent)) {
              items.push({
                lowResSrc: image.absolute_url?.replace(/version/, '_p'),
                src: image.absolute_url?.replace(/version/, resolution),
                category: response.type === 'project' ? PROJECT_IMAGES_TAG : categoryData.master_tag,
                caption: response.type === 'project' ? null : image.caption,
                index,
                id: index,
                isPhotoUnderReview: checkIsPhotoUnderReview({
                  photoUnderReview: photo_under_review, caption: image.caption
                })
              })
              index += 1
            } else if (photo_under_review) {
              const isPhotoUnderReview = checkIsPhotoUnderReview({
                photoUnderReview: photo_under_review, caption: image.caption
              })
              if (isPhotoUnderReview) {
                if (!isThumbImageUrlDifferent) {
                  // thumb_image_url will always be on 0th Index
                  items[0].isPhotoUnderReview = isPhotoUnderReview
                } else if (!isCoverPhotoUrlDifferent) {
                  items[coverPhotoUrlIndex].isPhotoUnderReview = isPhotoUnderReview
                }
              }
            }
          })
        }
      })
    }
  })
  if (addInventoryConfigItems && inventory_configs && inventory_configs.length > 0) {
    const inventoryConfigIds = {}
    inventory_configs.forEach((config) => {
      if (!inventoryConfigIds[config.id]) {
        const floorPlans = config.floor_plan_urls
        if (!floorPlans || !floorPlans.length) { return }
        inventoryConfigIds[config.id] = true
 
        // First Add 'Ground Floor' in config tag
        let addGround = false
        floorPlans.forEach((_, indx) => {
          if (floorPlans[indx].tags.length > 1) { addGround = true }
        })
 
        if (addGround) {
          floorPlans.forEach((floorPlan) => {
            if (floorPlan.tags.length <= 1) { floorPlan.tags.push(GROUND_FLOOR_CONFIG_TAG) }
          })
        }
 
        // Sort Config's Floor Plans
        floorPlans.sort(withinConfigCompare)
        const areaUnit = getAreaUnit(response)
        const configTag = getFloorPlanArea(config, areaUnit)
        items = items.concat(
          floorPlans.map((floorPlan) => {
            const galleryImage = {
              lowResSrc: floorPlan.floor_plan_url?.replace(/version/, '_p'),
              src: floorPlan.floor_plan_url?.replace(/version/, resolution),
              category: configTag,
              caption: `${floorPlan.tags.join(' - ')} floor plan`,
              index,
              id: index
            }
            index += 1
            return galleryImage
          })
        )
      }
    })
  }
  // This handles thumb_image_url and cover_photo_url cases for Photo_Under_Review
  if (photo_under_review && items?.length === 1) {
    items[0].isPhotoUnderReview = true
  }
  return items
}

export function parseListResponse(response, service, useNewGalleryMehtod) {
  if (isEmpty(response.hits)) return { is_last_page: true, hits: [] }
  response.hits = response.hits.map((data) => {
    const gallery_images = useNewGalleryMehtod
      ? mapGalleryImagesNew(data, service) : mapGalleryImages(data)
    const { sellers_info, contact_persons_info } = parseSellerInfoLabelForBuilderProjects(data)
    const { _id } = data
    const virtualTourUrl = getDigiTourAvailable(data)
    return {
      ...data,
      property_category_id: data?.property_category_id || data?.inventory_configs?.[0]?.property_category_id,
      property_price: getPropPrice(data, false, false),
      type: getPropertyType(data),
      coverImageUrl: getCoverImageUrl(data, gallery_images),
      gallery_images,
      sellers_info,
      contact_persons_info,
      roomTypes: data.property_category_id === 3 ? getRoomTypes(data.room_types_display) : [],
      formatted_per_sqft_rate: getFormattedPerSqftPrice(data),
      property_tags: !isEmpty(data?.property_tags) ? getPropertyTags(data.property_tags) : [],
      isProjectListing: isProjectListing(data.type),
      projectListingId: _id,
      hasVirtualTour: !isEmpty(virtualTourUrl),
      virtualTourUrl,
      id: isProjectListing(data.type) ? data.project_id : data.id
    }
  })
  return response
}

/*
  arguments: array of Ids, separator and extraParams (Object)
  return type: string
*/
export function constructIdsForBulkDedicated(arr, separator = '', extraParams = {}) {
  const { showNpMultiCard = false, allProjetListingIds = {} } = extraParams
  let result = ''
  if (!isEmpty(arr) && arr instanceof Array) {
    const arrLength = arr.length
    /*
      arr format is [id1, id2]
    */
    arr.forEach((eachItem, index) => {
      const suffix = index !== arrLength - 1 ? ',' : ''
      if (allProjetListingIds?.[eachItem]) {
        // For PendingAction properties projectListingIds format is {'id': 'configId'}
        // For all other cases projectListingIds format is {'configId': true}
        if (showNpMultiCard) {
          result += `${eachItem}${suffix}`
        }
      } else result += `${eachItem}${separator}${suffix}`
    })
  }
  return result
}
 
function parsedFavIdsData(allIds, separator = '') {
  let result = ''
  if (!isEmpty(allIds) && allIds instanceof Array) {
    const arrLength = allIds.length
    allIds.forEach((eachItem, index) => {
      const suffix = index !== arrLength - 1 ? ',' : ''
      result += `${eachItem?.id}${separator}${suffix}`
    })
  }
  return result
}
 
export function constructFavIds(arr, separator = '', showNpMultiCard = false, projectListingIds = []) {
  let result = ''
  result = parsedFavIdsData(arr, separator)
  if (showNpMultiCard) {
    const prefix = !isEmpty(result) ? ',' : ''
    result += `${prefix}${parsedFavIdsData(projectListingIds)}`
  }
  return result
}
