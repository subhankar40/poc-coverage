export const getRAPaymentBody = (store, payload) => {
  const { userTerms } = payload
  const {
    raLandingCard: {
      stateName, cityName, userType, stateList
    },
    raLandlordDetails: { isValid: landlordIsValid, ...landlordDetails },
    raPropertyDetails: { isValid: propertyIsValid, ...propertyDetails },
    raContractDetails: {
      agreementStartDate, monthlyRent, rentPaymentDay, securityAmount, noticePeriod, annualIncrement
    }, raTenantDetails: { tenantsData }, raItemListDetails, rentAgreement: { couponCode }
  } = store.value
  const { data, optedIn } = getUser(store.value)
  const isTenant = userType === 'TENANT'
  const { contactNumber, email, name } = isTenant ? tenantsData[0] : landlordDetails
  const optWhatsapp = !isEmpty(optedIn) && optedIn[contactNumber]
  const otherDetails = getOtherDetails(raItemListDetails, userTerms)
  const body = {
    paymentDto: {
      paymentDetail: {
        sourceId: 1
      }
    },
    userDto: {
      profileUuid: data?.profile_uuid,
      email,
      fullName: name,
      optWhatsapp
    },
    multipleTenantDetails: tenantsData,
    landlordDetails,
    agreementDetails: {
      annualIncrement: parseIntFromData(annualIncrement),
      sameAsPropertyAddress: false,
      monthlyRent: parseIntFromData(monthlyRent),
      securityAmount: parseIntFromData(securityAmount),
      noticePeriod,
      agreementStartDate: agreementStartDate.toString(),
      rentPaymentDay,
      propertyDetails: {
        stateUuid: getStateUuid(stateName, stateList),
        state: stateName,
        cityUuid: cityName,
        ...propertyDetails
      },
      otherDetails
    },
    createdBy: userType,
    orderType: 'RENT_AGREEMENT'
  }
  if (!isEmpty(couponCode)) {
    body.couponCode = couponCode
  }
  return body
}
 
export const getNewRAPaymentBody = (store, payload) => {

export const getPendingRaTransDetail = (orderInfo) => {
  let amountStatus = false
  let userSignStatus = false
  let noOfPendingSign = 0
  let landlordSignStatus = false
  let orderDetailsStatus = true
  let allSignPending = true
  const isTenant = orderInfo?.order?.createdBy === 'TENANT'
  if (!isEmpty(orderInfo?.orderStatus)) {
    orderInfo.orderStatus.forEach((item) => {
      const status = item.statusId === 2
      if (item.stateId === 2) {
        amountStatus = status
      } else if (item.stateId === 14) {
        landlordSignStatus = status
        if (status) {
          allSignPending = false
        }
        if (!isTenant) {
          userSignStatus = status
        } else if (!status) {
          noOfPendingSign += 1
        }
      } else if (item.stateId === 22) {
        orderDetailsStatus = status
      }
    })
  }
  if (!isEmpty(orderInfo?.order?.multipleTenantDetails)) {
    orderInfo.order.multipleTenantDetails.forEach((item) => {
      if (item?.esignStatus) {
        allSignPending = false
      }
      if (isTenant && item?.contactNumber === orderInfo.order.paymentDto.payerDetail.phone) {
        userSignStatus = item?.esignStatus
      } else if (!item?.esignStatus) {
        noOfPendingSign += 1
      }
    })
  }
  return {
    amountStatus,
    userSignStatus,
    noOfPendingSign,
    landlordSignStatus,
    isTenant,
    orderDetailsStatus,
    allSignPending
  }
}

export const getRATranStatus = (order) => {
  const { orderStatus, order: { status: mainStatus } } = order || {}
  let status = 'success'
  orderStatus?.forEach((item) => {
    if (item.statusId === 3 || (item.stateId === 23 && item.statusId === 2)) {
      status = 'failure'
    }
    if (item.statusId === 1 && status === 'success') {
      status = 'pending'
    }
  })
  if (mainStatus === TMstatus.REFUND_INITIATED || mainStatus === TMstatus.REFUND_SUCCESSFUL) {
    return mainStatus
  }
  return status
}

const parseIfJSON = (value) => {
  if (typeof value !== 'string') {
    return value
  }
  try {
    return JSON.parse(value)
  } catch (_) {
    return value
  }
}
 
export const parseCouponData = (response) => {
  const data = []
  response instanceof Array && response.forEach((item) => {
    if (!isEmpty(item) && item.offerType === stringConstants.NORMAL) {
      const { coupons, placeholders } = item
      coupons instanceof Array && coupons.forEach((coupon) => {
        if (coupon?.couponStatus === stringConstants.ACTIVE && !isEmpty(placeholders)
          && !isEmpty(placeholders[0]?.value)
          && !isEmpty(coupon?.meta_data?.value) && !isEmpty(coupon?.meta_data?.valueMetric)) {
          const { id, code, meta_data: { value, valueMetric } } = coupon
          const descriptionWithText = placeholders.filter(item => item.key === 'text')
          let description = parseIfJSON(descriptionWithText[0]?.value)
          if (Array.isArray(description)) {
            description = description[0]
          }
          const discount = `${valueMetric === stringConstants.ACTIVE ? '₹' : ''}${value}${valueMetric === stringConstants.PERCENT ? '%' : ''}`
          data.push({
            id, code, discount, description
          })
        }
      })
    }
  })
  return data
}
