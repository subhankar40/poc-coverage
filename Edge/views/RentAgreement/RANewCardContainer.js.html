import React from 'react'
import {
  Platform, StyleSheet, View, Animated, TouchableOpacity, BackHandler,
  LayoutAnimation, Keyboard, UIManager, InteractionManager
} from 'react-native'
import { connect } from 'react-redux'
import PropTypes from 'prop-types'
import LinearGradient from 'react-native-linear-gradient'
 
import UserLoginUtility from 'utils/UserLoginUtility'
import colors from 'utils/colors'
import { getHeight } from 'common/utils/deviceDimensions'
import images from 'utils/images'
import isEmpty from 'common/utils/isEmpty'
import { getCountryInformation } from 'utils/helpers'
import { trackEvent } from 'utils/trackEvents'
import { resetActionForEdgeHome, resetActionForRASuccess } from 'utils/navigationActions'
import ImageComponent from 'common/components/ImageComponent'
import { getUser } from 'selectors/user'
import { getIsInsuranceIncluded, getRaInsuranceBundlingData } from 'selectors/rentAgreement'
import MainCardComponent from '../RentPay/MainCardComponent'
import { fromMHFlow, raPrePaymentCardList } from './RAhelper'
import RAHeaderComponent from './RAHeaderComponent'
import RALandingCard from './RALandingCard'
import RAButton from './RAButton'
import { ErrorTypes, getErrorModalRef, openUrl } from '../RentPay/RentPayHelpers'
import { clearRAFlowData, createRAExternalOrder, createRAOrder } from './RAActions'
import analyticsConstants from '../RentPay/analyticsConstants'
import RAPrePaymentDetails from './RAPrePaymentDetails'
 
const styles = StyleSheet.create({
  scrollShadow: {
    height: 20,
    position: 'absolute',
    width: '100%',
    alignSelf: 'center'
  },
  backButtonContainer: {
    width: 60,
    justifyContent: 'center',
    alignItems: 'center',
    marginRight: -30,
    zIndex: 1
  },
  cardContainer: {
    position: 'absolute',
    width: '100%',
    backgroundColor: colors.white,
    marginTop: 40,
    borderRadius: 40
  }
})
 
class RaNewCardContainer extends React.PureComponent {
  constructor(props) {
    super(props)
    const initialIndex = 0
    this.state = {
      index: initialIndex,
      isKeyboardVisible: false,
      loadContent: false
    }
    this.timer = null
    this.keyboardDidShowListener = null
    this.keyboardDidHideListener = null
    this.countryInfo = getCountryInformation()
    this.countryUrl = 'in'
    this.animatedIndex = new Animated.Value(initialIndex)
    this.lastButtonPress = 0
    this.animating = false
    this.height = getHeight()
    this.cardRefs = [React.createRef(), React.createRef()]
    if (Platform.OS === 'android') {
      UIManager.setLayoutAnimationEnabledExperimental
        && UIManager.setLayoutAnimationEnabledExperimental(true)
    }
    this.userLoginUtility = null
  }
 
  componentDidMount() {
    const {
      navigation, cityName, stateName, clearRAFlowData, stopClearFlowData, dispatch
    } = this.props
    const { index } = this.state
    this.userLoginUtility = new UserLoginUtility({
      navigation, dispatch
    })
    this.keyboardDidShowListener = Keyboard.addListener('keyboardDidShow', this.keyboardDidShow)
    this.keyboardDidHideListener = Keyboard.addListener('keyboardDidHide', this.keyboardDidHide)
    navigation.addListener('focus', this.onDidFocus)
    navigation.addListener('blur', this.onDidBlur)
    if (index === 0) {
      trackEvent(analyticsConstants.RENT_AGREEMENT_LANDING_PAGE, 'hiw_seen')
      trackEvent(analyticsConstants.RENT_AGREEMENT_LANDING_PAGE, 'faq_seen')
    }
    trackEvent(analyticsConstants.RENT_AGREEMENT_FLOW, `${raPrePaymentCardList[index].eventAction}_det_seen`)
    if (!stopClearFlowData && (!isEmpty(cityName) || !isEmpty(stateName))) {
      clearRAFlowData()
    }
    InteractionManager.runAfterInteractions(() => {
      this.setState({ loadContent: true })
    })
  }
 
  componentWillReceiveProps(nextProps) {
    const {
      orderStatus: prevOrderStatus, paymentStatus: prevPaymentStatus, navigation
    } = this.props
    const {
      orderStatus: nextOrderStatus, paymentStatus: nextPaymentStatus, orderId
    } = nextProps
    if (prevPaymentStatus !== nextPaymentStatus) {
      if (nextPaymentStatus === 'success') {
        resetActionForRASuccess()
      } else if (nextPaymentStatus === 'pending') {
        getErrorModalRef()?.openModal({
          type: ErrorTypes.TRANSACTION_PENDING_RA,
          onSecondaryButtonPress: this.goToHome,
          transactionId: orderId
        })
        trackEvent(analyticsConstants.RENT_AGREEMENT_PAYMENT_PENDING, 'opened')
      }
    }
    if (prevOrderStatus !== nextOrderStatus) {
      if (nextOrderStatus === 'success') {
        this.props.hideLoader()
      } else if (nextOrderStatus === 'failure' && navigation.isFocused()) {
        this.props.hideLoader()
        // TODO: Need to check here, as this error modal comes twice
        // when RA is opened from EDGE-fragment
        //  one from here
        // other from MyTransaction.js
        getErrorModalRef()?.openModal({
          type: ErrorTypes.TRANSACTION_FAILED_RA,
          onRetry: this.retryPayment,
          transactionId: orderId
        })
        trackEvent(analyticsConstants.RENT_AGREEMENT_PAYMENT_FAILED, 'opened')
      } else if (nextOrderStatus === 'pending') {
        this.props.showLoader()
      }
    }
  }
 
  componentDidUpdate(prevProps, prevState) {
    const { index } = this.state
    if (index !== prevState.index) {
      trackEvent(analyticsConstants.RENT_AGREEMENT_FLOW, `${raPrePaymentCardList[index].eventAction}_det_seen`)
    }
  }
 
  componentWillUnmount() {
    const { navigation } = this.props
    this.keyboardDidHideListener && this.keyboardDidHideListener.remove()
    this.keyboardDidShowListener && this.keyboardDidShowListener.remove()
    navigation.removeListener('focus', this.onDidFocus)
    navigation.removeListener('blur', this.onDidBlur)
    this.backPressListener?.remove()
    this.timer && clearTimeout(this.timer)
  }
 
  keyboardDidShow = () => {
    this.setState({ isKeyboardVisible: true })
  }
 
  keyboardDidHide = () => {
    this.setState({ isKeyboardVisible: false })
  }
 
  onDidFocus = () => {
    this.backPressListener = BackHandler.addEventListener('hardwareBackPress', this.onBack)
  }
 
  onDidBlur = () => {
    this.backPressListener?.remove()
  }
 
  goToHome = () => {
    getErrorModalRef() && getErrorModalRef().onClose()
    resetActionForEdgeHome(true)
    trackEvent(analyticsConstants.RENT_AGREEMENT_PAYMENT_PENDING, 'go_to_home')
  }
 
  retryPayment = () => {
    getErrorModalRef() && getErrorModalRef().onClose()
    this.createRAOrder()
    trackEvent(analyticsConstants.RENT_AGREEMENT_PAYMENT_FAILED, 'retry')
  }
 
  shouldHandlePress = () => {
    const now = Date.now()
    const debounceTime = 500
    if (this.lastButtonPress && (now - this.lastButtonPress) < debounceTime) {
      return false
    }
    this.lastButtonPress = now
    return !this.animating
  }
 
  createAnimation = toValue => (
    Animated.spring(this.animatedIndex, {
      toValue,
      useNativeDriver: true,
      bounciness: 6
    })
  )
 
  cardAnimation = (index) => {
    this.animating = true
    this.createAnimation(index).start(() => {
      this.animating = false
      const newRef = this.cardRefs[index].current
      if (newRef?.isFormDataInValid() && newRef?.focus) {
        newRef.focus()
      } else {
        Keyboard.dismiss()
      }
    })
  }
 
  onBack = () => {
    if (this.shouldHandlePress()) {
      const { index } = this.state
      if (index !== 0) {
        this.cardAnimation(index - 1)
        LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut)
        this.setState({
          index: index - 1
        })
      } else {
        this.props.openExitModal()
      }
    }
    return true
  }
 
  externalOrderCallback = (isSuccess, url = null) => {
    const { hideLoader } = this.props
    hideLoader()
    if (isSuccess) {
      openUrl(url)
    } else {
      const subtitle = 'Something went wrong. Please try again later.'
      getErrorModalRef() && getErrorModalRef()
        .openModal({ type: ErrorTypes.ACCOUNT_ERROR, subtitle })
    }
  }
 
  createRAExternalOrder = () => {
    const {
      createRAExternalOrder, stateName, cityName, userType, showLoader
    } = this.props
    const userDetails = this.cardRefs[1].current?.getUserDataForLead() || {}
    const payload = {
      ...userDetails,
      state: stateName,
      city: cityName,
      entityType: userType,
      callback: this.externalOrderCallback
    }
    showLoader()
    createRAExternalOrder(payload)
  }
 
  createRAOrder = () => {
    const userDetails = this.cardRefs[1].current?.getUserData() || {}
    const payload = {
      ...userDetails,
      fromPrePayment: true,
      insurancePlanId: this.props.insurancePlanId
    }
    this.props.createRAOrder(payload)
  }
 
  onNext = () => {
    const { user, navigation: { goBack }, isMHFlow } = this.props
    if (this.shouldHandlePress()) {
      const { index } = this.state
      const { netStatus } = this.props
      const currentRef = this.cardRefs[index].current
      if (currentRef?.isFormDataInValid()) {
        currentRef?.showErrorMessage()
      } else if (netStatus === 'offline') {
        getErrorModalRef() && getErrorModalRef().openModal({
          type: ErrorTypes.NETWORK_ERROR, onSecondaryButtonPress: goBack
        })
      } else {
        this.userLoginUtility.checkUserAndProceed(user).then(() => {
          currentRef?.onSavePress()
          if (index === raPrePaymentCardList.length - 1) {
            isMHFlow ? this.createRAExternalOrder() : this.createRAOrder()
          } else {
            this.cardRefs[index + 1].current?.updateStatus()
            this.cardAnimation(index + 1)
            LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut)
            this.setState({
              index: index + 1
            })
          }
          const label = currentRef?.getEventLabel ? currentRef?.getEventLabel() : {}
          trackEvent(analyticsConstants.RENT_AGREEMENT_FLOW, `${raPrePaymentCardList[index].eventAction}_cta_clicked`, label)
        })
      }
    }
  }
 
  renderLandingCard = () => (
    <RALandingCard
      ref={this.cardRefs[0]}
      loadContent={this.state.loadContent}
    />
  )
 
  renderPrePaymentDetailsCard = () => (
    <RAPrePaymentDetails
      ref={this.cardRefs[1]}
      isKeyboardVisible={this.state.isKeyboardVisible}
      navigation={this.props.navigation}
      onpenInsuranceModal={this.props.onpenInsuranceModal}
    />
  )
 
  renderButton = () => {
    const { index } = this.state
    return (
      <RAButton
        index={index}
        onPress={this.onNext}
        fromPrePayment
      />
    )
  }
 
  renderCardContent = (id) => {
    switch (id) {
      case 'BASIC_DETAILS':
        return this.renderLandingCard()
      case 'BASIC_DETAILS_1':
        return this.renderPrePaymentDetailsCard()
      default:
        return null
    }
  }
 
  renderBackButton = () => (
    <TouchableOpacity
      activeOpacity={1}
      onPress={this.onBack}
      style={styles.backButtonContainer}
    >
      <ImageComponent
        style={[{ transform: [{ rotate: '90deg' }] }, Platform.OS === 'ios' ? { height: 22, width: 22 } : {}]}
        source={images.dropDownGreyDown}
      />
    </TouchableOpacity>
  )
 
  renderHeader = index => (
    <View style={{ paddingTop: index === 0 ? 0 : 4 }}>
      <View style={{ height: 80, width: '100%', flexDirection: 'row' }}>
        {index !== 0 && this.renderBackButton()}
        <RAHeaderComponent
          index={0}
        />
      </View>
    </View>
  )
 
  renderHeaderShadow = () => (
    <LinearGradient
      style={styles.scrollShadow}
      colors={[colors.black4, colors.transparent1]}
    />
  )
 
  getContainerHeight = (index) => {
    const { safeAreaHeight } = this.props
    const maxContainerHeight = (this.height - safeAreaHeight) - 88
    return index !== 0 ? maxContainerHeight : maxContainerHeight - 328
  }
 
  renderCards = (item, index) => (
    <View key={item.id}>
      {this.renderHeader(index)}
      <View style={{ width: '100%', height: this.getContainerHeight(index) }}>
        {this.renderCardContent(item.id, index)}
        {index !== 0 && this.renderHeaderShadow()}
      </View>
    </View>
  )
 
  renderCard = () => {
    const transform = [{
      translateY: this.animatedIndex.interpolate({
        inputRange: [0, 1],
        outputRange: [328, 0],
        extrapolate: 'clamp'
      })
    }]
    return (
      <Animated.View
        style={[styles.cardContainer, { transform, height: this.height + 50 }]}
      >
        <MainCardComponent
          views={raPrePaymentCardList.map(this.renderCards)}
          animatedIndex={this.animatedIndex}
        />
      </Animated.View>
    )
  }
 
  render() {
    const { loadContent } = this.state
    return (
      <View style={{ flex: 1 }}>
        {this.props.renderBackground()}
        {this.renderCard()}
        {loadContent && this.renderButton()}
      </View>
    )
  }
}
 
RaNewCardContainer.propTypes = {
  netStatus: PropTypes.string.isRequired,
  user: PropTypes.object.isRequired,
  safeAreaHeight: PropTypes.number.isRequired,
  navigation: PropTypes.object.isRequired,
  showLoader: PropTypes.func.isRequired,
  hideLoader: PropTypes.func.isRequired,
  openExitModal: PropTypes.func.isRequired,
  isMHFlow: PropTypes.bool.isRequired,
  createRAExternalOrder: PropTypes.func.isRequired,
  renderBackground: PropTypes.func.isRequired,
  userType: PropTypes.string.isRequired,
  stateName: PropTypes.string.isRequired,
  cityName: PropTypes.string.isRequired,
  orderStatus: PropTypes.string.isRequired,
  paymentStatus: PropTypes.string.isRequired,
  orderId: PropTypes.number.isRequired,
  createRAOrder: PropTypes.func.isRequired,
  stopClearFlowData: PropTypes.bool.isRequired,
  dispatch: PropTypes.func.isRequired,
  clearRAFlowData: PropTypes.func.isRequired,
  insurancePlanId: PropTypes.number.isRequired,
  onpenInsuranceModal: PropTypes.func.isRequired
}
 
const mapStateToProps = (state) => {
  const {
    netInfo: { status: netStatus },
    raLandingCard,
    raOrderInfo
  } = state
  const user = getUser(state)
  const { userType, stateName, cityName } = raLandingCard
  const { orderId, orderStatus, paymentStatus } = raOrderInfo
  return {
    user,
    netStatus,
    isMHFlow: fromMHFlow(stateName),
    userType,
    stateName,
    cityName,
    orderId,
    orderStatus,
    paymentStatus,
    insurancePlanId: getIsInsuranceIncluded(state)
      ? getRaInsuranceBundlingData(state)?.insurancePlanId : null
  }
}
 
const mapDispatchToProps = dispatch => ({
  dispatch,
  createRAExternalOrder: (...args) => dispatch(createRAExternalOrder(...args)),
  createRAOrder: (...args) => dispatch(createRAOrder(...args)),
  clearRAFlowData: (...args) => dispatch(clearRAFlowData(...args))
})
 
export default connect(
  mapStateToProps,
  mapDispatchToProps,
  null,
  { forwardRef: true }
)(RaNewCardContainer)
