import React from 'react'
import { View } from 'react-native'
import PropTypes from 'prop-types'
import ImageComponent from 'common/components/ImageComponent'
import GenericText from 'common/components/react-native-generic-text'
import images from 'utils/images'
import colors from 'utils/colors'
import isEmpty from 'common/utils/isEmpty'
import fontUtils from 'common/utils/fontUtils'
 
// just for Reference const STATUS = { 1: 'PENDING', 2: 'SUCCESS', 3: 'FAILED' }
 
const STATUS_ICON = {
  1: images.transPendingIcon, 2: images.transSuccessIcon, 3: images.transFailedIcon
}
 
export default class RATransactionStatus extends React.PureComponent {
  renderText = ({
    text, size, lineHeight, color, font, ...restProps
  }) => (
    <GenericText
      size={size || 12}
      lineHeight={lineHeight || 14}
      color={color || colors.grey20}
      font={font || fontUtils.appFont}
      {...restProps}
    >
      {text}
    </GenericText>
  )
 
  renderStateIcon = id => (
    <ImageComponent source={STATUS_ICON[id]} style={{ height: 36, width: 36, resizeMode: 'contain' }} />
  )
 
  getLocalDateTime = (date) => {
    const dateHrs = date.getHours()
    let hours = dateHrs > 12 ? dateHrs % 12 : dateHrs
    if (hours < 10) hours = `0${hours}`
    let minutes = date.getMinutes()
    if (minutes < 10) minutes = `0${minutes}`
    const timeOfDay = date.getHours() < 12 ? 'AM' : 'PM'
    return `${hours}:${minutes} ${timeOfDay}`
  }
 
  getDateAndTime = (value) => {
    if (!isEmpty(value)) {
      const dateObj = new Date(value)
      return `${dateObj.toDateString()}, ${this.getLocalDateTime(dateObj)}`
    }
    return null
  }
 
  getStatusArray = () => {
    if (isEmpty(this.props.order?.orderStatus) || isEmpty(this.props.order?.order)) {
      return []
    }
    const { order: { orderStatus, order } } = this.props
    const statusArr = [...orderStatus.filter(item => item.statusId === 2 && item.stateId !== 1),
      ...orderStatus.filter(item => item.statusId !== 2 && item.stateId !== 1)]
    const statusObj = {}
    orderStatus.forEach((item) => {
      statusObj[item.stateId] = item
    })
    const isTenant = order?.createdBy === 'TENANT'
    if (!isEmpty(statusObj[23]) && !isEmpty(statusObj[2])) {
      return [{
        text: statusObj[2].statusId === 3 ? 'Payment Failed' : 'Amount Received',
        subtext: isEmpty(statusObj[2].date) ? '' : this.getDateAndTime(statusObj[2].date),
        statusId: statusObj[2].statusId
      }, {
        text: 'Agreement Cancelled',
        subtext: isEmpty(statusObj[23].date) ? '' : this.getDateAndTime(statusObj[23].date),
        statusId: 2
      }]
    }
    if (!isEmpty(statusObj[22]) && statusObj[22].statusId === 1) {
      return [{
        text: statusObj[2].statusId === 3 ? 'Payment Failed' : 'Amount Received',
        subtext: isEmpty(statusObj[2].date) ? '' : this.getDateAndTime(statusObj[2].date),
        statusId: statusObj[2].statusId
      }, {
        text: 'Agreement details pending',
        subtext: 'Enter all details',
        statusId: 1
      }, {
        text: 'Digital sign pending',
        subtext: 'Download signed agreement after this',
        statusId: 1
      }]
    }
    const result = []
    statusArr.forEach((value, index) => {
      const { stateId, date, statusId } = value
      let text = ''
      let subtext = ''
      const isLastIndex = index === statusArr.length - 1
      if (stateId === 2) {
        if (statusId === 3) {
          text = 'Payment Failed'
        } else if (statusObj[22]?.statusId === 2) {
          text = 'Agreement created successfully'
        } else {
          text = 'Amount Received'
        }
        subtext = isEmpty(date) ? '' : this.getDateAndTime(date)
      } else if (stateId === 16) {
        const isMultiTenant = this.props.order?.order?.multipleTenantDetails?.length > 1
        text = `${!isMultiTenant && isTenant ? 'Your' : "Tenants'"} digital sign ${statusId === 2 ? 'done' : 'pending'}`
        if (!isEmpty(date)) {
          subtext = this.getDateAndTime(date)
        } else if (isLastIndex) {
          subtext = 'Download fully signed agreement after this'
        } else if (isMultiTenant) {
          subtext = 'Download agreement with all tenant’s sign'
        } else {
          subtext = `Download agreement with ${isTenant ? 'your' : 'tenant'} sign after this`
        }
      } else if (stateId === 14) {
        text = `${isTenant ? 'Landlord' : 'Your'} digital sign ${statusId === 2 ? 'done' : 'pending'}`
        if (!isEmpty(date)) {
          subtext = this.getDateAndTime(date)
        } else if (isLastIndex) {
          subtext = 'Download fully signed agreement after this'
        } else {
          subtext = `Download agreement with ${isTenant ? 'landlord' : 'your'} sign after this`
        }
      }
      if (!isEmpty(text, subtext)) {
        result.push({ text, subtext, statusId })
      }
    })
    return result
  }
 
  renderLeftInfoView = orderStatusArray => orderStatusArray.map((value, index) => {
    if (value) {
      const { text, subtext } = value
      return (
        <View>
          <View style={{ height: 36, paddingRight: 24, justifyContent: 'center' }}>
            {this.renderText({
              text, size: 16, lineHeight: 19, numberOfLines: 1
            })}
            {this.renderText({
              text: subtext, color: colors.darkGrey36, numberOfLines: 1
            })}
          </View>
          {index < orderStatusArray.length - 1 && <View style={{ height: 19 }} />}
        </View>
      )
    }
    return null
  })
 
  renderRightStateIcons = orderStatusArray => orderStatusArray.map((value, index) => {
    if (value) {
      const statusId = value.statusId
      const nextStatusId = index + 1 < orderStatusArray.length
        ? orderStatusArray[index + 1].statusId : null
      return (
        <View style={{ alignItems: 'center' }}>
          {this.renderStateIcon(statusId)}
          {!!nextStatusId && (nextStatusId === 2
            ? <View style={{ width: 2.4, height: 19, backgroundColor: colors.seafoamBlue }} />
            : this.renderDottedLine())}
        </View>
      )
    }
    return null
  })
 
  renderDottedLine = () => (
    <View>
      <View style={{ width: 2.4, height: 5, backgroundColor: colors.seafoamBlue }} />
      <View style={{ width: 2.4, height: 2 }} />
      <View style={{ width: 2.4, height: 5, backgroundColor: colors.seafoamBlue }} />
      <View style={{ width: 2.4, height: 2 }} />
      <View style={{ width: 2.4, height: 5, backgroundColor: colors.seafoamBlue }} />
    </View>
  )
 
  renderTitle = () => (
    <View style={{ paddingBottom: 21 }}>
      {this.renderText({ text: 'Transaction Status' })}
    </View>
  )
 
  renderContent = () => {
    const orderStatusArray = this.getStatusArray()
    return (
      <View style={{ flexDirection: 'row', paddingBottom: 24 }}>
        <View style={{ flex: 1 }}>
          {this.renderLeftInfoView(orderStatusArray)}
        </View>
        <View>
          {this.renderRightStateIcons(orderStatusArray)}
        </View>
      </View>
    )
  }
 
  render() {
    const { containerStyle, showTitle } = this.props
    return (
      <View style={[{ paddingHorizontal: 32 }, containerStyle]}>
        {showTitle && this.renderTitle()}
        {this.renderContent()}
      </View>
    )
  }
}
 
RATransactionStatus.propTypes = {
  order: PropTypes.object.isRequired,
  containerStyle: PropTypes.object,
  showTitle: PropTypes.bool
}
 
RATransactionStatus.defaultProps = {
  containerStyle: {},
  showTitle: true
}
