import { isEmpty } from 'ramda'
import { createSelector, createStructuredSelector } from 'reselect'
import { EMPTY_STATE } from 'common/constants/AppConstants'
import { getServiceName } from 'utils/flatmatesHelpers'
 
const getCitySelect = state => state.citySelect
 
const getCoordinateDetails = state => state.cityCoordinates
 
const getServiceObject = state => state?.common?.service
 
const getFilterPropertyScreenTotal = state => state.filterPropertyScreen.total
 
const getFilterPropertyScreenFetching = state => state.filterPropertyScreen.fetching
 
const getTrendingProjectsListFromStore = state => state.trendingProjects.trendingProjectsList
 
const getFilterParamsBase = createSelector(
  state => state.filterParams?.[1].base || EMPTY_STATE.ARRAY, (base) => {
    if (isEmpty(base) && Array.isArray(base)) { return EMPTY_STATE.ARRAY }
    return base
  }
)
 
const getFilterParamsCityUuid = state => state.filterParams?.[1]?.city_uuid
    || EMPTY_STATE.STRING
 
export const getCitiesOnServiceBasis = createSelector(
  getCitySelect, getServiceObject,
  (citySelect, service) => {
    const { name, isFlatmates } = service || {}
    return citySelect[getServiceName(name, isFlatmates)]
  }
)
 
export const getFilterPropertyScreenParams = createSelector(
  getFilterPropertyScreenTotal, getFilterPropertyScreenFetching,
  (total, fetching) => ({ total, fetching })
)
 
export const getSelectedFilterParams = createSelector(
  getFilterParamsBase, getFilterParamsCityUuid,
  (base, cityUuid) => ({ base, city_uuid: cityUuid })
)
 
export const getTrendingProjectsList = createSelector(getTrendingProjectsListFromStore,
  list => list)
 
export const getAddress = createSelector(
  getCoordinateDetails,
  data => data?.address
)
 
export const getCoordinatesLatitude = createSelector(
  getCoordinateDetails,
  data => data?.latitude
)
 
export const getCoordinatesLongitude = createSelector(
  getCoordinateDetails,
  data => data?.longitude
)
 
export const getCoordinates = createStructuredSelector({
  latitude: getCoordinatesLatitude,
  longitude: getCoordinatesLongitude
})
