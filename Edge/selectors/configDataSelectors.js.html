export const getRaTncUrl = (configData) => {

export const getInsurancePlans = createSelector(
  getHousingProtectConfigV2,
  (configData) => {
    const { plans = [] } = configData
    return plans?.filter?.((item) => {
      const { androidVersion, iosVersion, enableOnIlp } = item
      const currentAppVersion = getNumFromVersionStr(DeviceInfo.getVersion())
      const version = getNumFromVersionStr(Platform.select({
        ios: iosVersion || currentAppVersion,
        android: androidVersion || currentAppVersion
      }))
      return !!enableOnIlp && currentAppVersion >= version
    }) || []
  }
)
