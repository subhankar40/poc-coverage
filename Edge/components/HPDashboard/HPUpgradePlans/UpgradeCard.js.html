import React, { useState, useMemo } from 'react'
import PropTypes from 'prop-types'
import HPCardTag from 'components/HpRevampv1/HPCardTag'
import { View } from 'react-native'
import GradientBlurCover from './GradientBlurCover'
import Content from './Content'
import { getBreakedString } from './utils'
import { CARD_INNER_CONTENT_PADDING, BOX_WRAPPER_SIZE } from './styles'
import { checkIsPopularValue, isMoneyBack } from '../utils'
 
const MONEYBACK_GUARANTEE3 = '100% MoneyBack Guarantee.'
const MOST_POPULAR = 'Most Popular'
 
function UpgradeCard(props) {
  const {
    style, item, onSeeAll, onGstPress, onBuy, onInfoIconClick
  } = props
  const [tagStyle, setTagStyle] = useState()
 
  const isPopular = checkIsPopularValue(item)
  const isMoneyBackCheck = isMoneyBack(item)
 
  const onTagLayoutChange = (evt) => {
    const { layout } = evt.nativeEvent
    // This style will take hp popular and moneyBack tag to center of emoji icon
    if (layout?.height) {
      const top = CARD_INNER_CONTENT_PADDING + Math.ceil(Math.abs(((layout?.height || 0) - 45) / 2))
      setTagStyle({
        top
      })
    }
  }
 
  const hpTagStyles = useMemo(() => [{
    right: -(BOX_WRAPPER_SIZE),
    alignSelf: 'center',
    position: 'absolute',
    width: '55%'
  }, tagStyle], [tagStyle])
 
  return (
    <View style={style}>
      <GradientBlurCover>
        <Content
          item={item}
          onSeeAll={onSeeAll}
          onGstPress={onGstPress}
          onBuy={onBuy}
          onInfoIconClick={onInfoIconClick}
        />
      </GradientBlurCover>
      {(isPopular || isMoneyBackCheck) ? (
        <HPCardTag
          onLayout={onTagLayoutChange}
          isPopular={isPopular}
          title={isMoneyBackCheck
            ? getBreakedString(item?.product_detail?.[0]?.golden_tag || '', 2)
                  || MONEYBACK_GUARANTEE3
            : MOST_POPULAR}
          isSideView
          style={hpTagStyles}
        />
      ) : null}
    </View>
  )
}
 
UpgradeCard.propTypes = {
  style: PropTypes.object,
  item: PropTypes.object,
  onSeeAll: PropTypes.func,
  onGstPress: PropTypes.func,
  onBuy: PropTypes.func,
  onInfoIconClick: PropTypes.func
}
UpgradeCard.defaultProps = {
  style: null,
  item: {},
  onSeeAll: () => {},
  onGstPress: () => {},
  onBuy: () => {},
  onInfoIconClick: () => {}
}
 
export default UpgradeCard
