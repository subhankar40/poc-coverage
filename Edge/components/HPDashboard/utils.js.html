export const getNewSelectedCity = (filterParams, citySelect) => {
  const serviceName = get(filterParams, STRINGS.SERVICE_NAME, SERVICE_TYPE.BUY_LC)
  const cities = citySelect[serviceName]
  let currentCity
  if (!isEmpty(cities) && cities.top_cities) {
    const allCities = cities.top_cities.concat(cities.other_cities)
    currentCity = find(propEq(STRINGS.UUID, filterParams.city_uuid))(allCities)
  }
 
  return currentCity
}
 
export const isCityDifferent = (selectedCity, filterParams, citySelect) => {
  const newSelectedCity = getNewSelectedCity(filterParams, citySelect)
  const selectedCityuuid = selectedCity?.city_uuid || ''
  const newSelectedCityuuid = newSelectedCity?.city_uuid || ''
  return newSelectedCity && selectedCityuuid !== newSelectedCityuuid
}

function CreateLayoutConfigHandler() {
  let layoutConfigs = new Map()
  const pRef = {
    ref: null
  }
 
  return ({
    setLayoutForIndex(key, value) {
      if (key) {
        layoutConfigs.set(key, value)
      } else {
        // throw new Error('Index should be number!')
      }
    },
    getLayoutForIndex(key = '') {
      return (layoutConfigs.get(key))
    },
    getLength() {
      return (layoutConfigs?.size)
    },
    clearLayoutConfigs() {
      layoutConfigs.clear()
      pRef.ref = null
    },
    initLayoutConfigs(sections = [], value = { height: 0 }, parentRef = null) {
      pRef.ref = parentRef
      const keyValues = sections.map(sectionName => [sectionName, value])
      layoutConfigs = new Map(keyValues)
    },
    getParentRef() {
      return pRef.ref
    },
    getLayoutHandler: childRef => (() => (new Promise((resolve) => {
      childRef.current?.measureLayout(
        findNodeHandle(pRef.ref?.current),
        (x, y, pageX, pageY) => {
          resolve?.({
            x, y, pageX, pageY
          })
        }
      )
    })))
  })
}

export function getFormattedPlanName(plan = '') {
  const isSpecial = plan.includes('+')
  if (isSpecial) {
    const splitData = plan.split('+')
    const str = []
    splitData.forEach((s) => {
      if (s.length > 1) {
        str.push(s)
      }
    })
    return (str)
  }
  return ([plan])
}
