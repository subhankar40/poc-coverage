import React, { PureComponent } from 'react'
import {
  View, TouchableOpacity, StyleSheet, Text
} from 'react-native'
import PropTypes from 'prop-types'
import colors from 'utils/colors'
import { moderateScale, scale } from 'common/utils/scalingFunctions'
import images from 'utils/images'
import fontUtils from 'common/utils/fontUtils'
import isEmpty from 'utils/isEmpty'
import { windowWidth } from 'utils/helpers'
import RequestImageView from 'components/Lists/RequestImageView'
import ImageComponent from 'common/components/ImageComponent'
// import { withNavigationRoute } from 'utils/hocs/navigationHOC'
import { getIsXDays } from 'common/utils/date'
import { getStreetInfo } from 'utils/GetAddressUtil'
import NewBridgeModule from 'common/utils/NewBridgeModule'
import FastImageBackground from 'common/components/FastImageBackground'
import { isIos } from 'common/utils/TypeChecker'
import DistanceBanner from './DistanceBanner'
import {
  getFurnishType, getPriceToRender,
  getPropertyImageUri, getProjectSubtitle, getPropertyArea,
  isProject
} from './utils'
import ProgressiveImage from './ProgressiveImage'
import SellerInfo from './SellerInfo'
import CheckoutMoreBtn from '../CheckoutMoreBtn'
import PropertyLockCta from './PropertyLockCta'
 
const cardWidth = windowWidth - scale(48)
 
const styles = StyleSheet.create({
  singleCardContainer: {
    width: cardWidth,
    marginRight: scale(16)
  },
  propertyDetailsContainer: {
    // width: scale(280),
    // marginRight: scale(16)
  },
  propertyCardContainer: {
    height: 913 / 758 * 340
  },
  lockedImageContainer: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0
  },
  propertyImage: {
    width: '100%',
    height: scale(144)
    // width: scale(271)
  },
  offersBannerContainer: {
    backgroundColor: colors.paleGrey8,
    borderRadius: scale(10),
    paddingVertical: scale(6),
    flexDirection: 'row',
    alignItems: 'center',
    justifyContent: 'space-between',
    paddingHorizontal: scale(8),
    marginBottom: scale(16)
  },
  offersIcon: {
    height: scale(10), width: scale(10), marginRight: scale(6)
  },
  forwardArrowIcon: {
    height: scale(8), width: scale(8), tintColor: colors.grey5, transform: [{ rotate: '270deg' }]
  },
  singleCardImage: { width: cardWidth, height: scale(144) },
  bottomSectionContainer: {
    flexDirection: 'row',
    justifyContent: 'space-between',
    borderTopWidth: 1,
    alignItems: 'center',
    borderColor: colors.grey11,
    overflow: 'hidden'
  },
  lockTitleStyle: {
    ...fontUtils.getFontStyle('600'),
    fontSize: 14,
    color: colors.white
  },
  lockSubTitleStyle: {
    ...fontUtils.getFontStyle('300'),
    fontSize: 12,
    color: colors.white
  },
  lockBottomStyle: {
    width: '100%',
    height: '100%',
    justifyContent: 'center',
    alignItems: 'center',
    overflow: 'hidden',
    borderColor: 'rgba(255,255,255,0.2)',
    borderWidth: 1,
    borderBottomLeftRadius: 16,
    borderBottomRightRadius: 16
  },
  lockBottomContentStyle: {
    flexDirection: 'row',
    alignItems: 'center'
  },
  lockIconStyle: {
    width: 20,
    height: 20
  },
  lockTitleWrapperStyle: {
    marginLeft: 15
  },
  lockCtaLayerStyle: {
    position: 'absolute',
    width: '100%',
    height: 85,
    bottom: 0
  }

class NearbyPropertiesCard extends PureComponent {
  renderDistanceBanner=() => <DistanceBanner item={this.props.item} />
 
  renderPropertyImage = () => {
    const {
      item, isLockCard
    } = this.props
    const imageUri = getPropertyImageUri(item)
    const url = imageUri ? { uri: imageUri } : images.listingPlaceholder
    const imageStyle = [
      styles.propertyImage,
      isLockCard ? { height: isIos ? scale(180) : scale(172) } : {}
    ]
    return (
      <View
        style={[{ overflow: 'hidden' }, isLockCard ? styles.lockedImageContainer : {}]}
      >
        <ProgressiveImage
          imageSource={url}
          thumbnailSource={images.listingPlaceholder}
          containerStyle={imageStyle}
          imageStyle={{ borderRadius: 10 }}
          imageLayout="cover"
        />
        {!isEmpty(item.min_distance_locality) && this.renderDistanceBanner()}
        {isEmpty(imageUri) && (
          <>
            <View style={{ position: 'absolute', bottom: scale(12), alignSelf: 'center' }}>
              <RequestImageView
                handleRequestImageCTA={this.onNavigateToDetailPage}
                propertyId={item.id}
                type={item.type}
                showNewUI
              />
            </View>
          </>
        )}
      </View>
    )
  }
 
    renderText = ({
      text, size, lineHeight, style, color, ...restProps
    }) => (
      <Text
        style={{
          fontSize: moderateScale(size || 12),
          lineHeight: moderateScale(lineHeight || 14),
          color: color || colors.black2,
          ...(style || {})
        }}
        {...restProps}
      >
        {text}
      </Text>
    )
 
    renderSecondaryText = () => {
      const { item: { developer_information: devInfo } } = this.props
      if (devInfo instanceof Array && devInfo[0]?.name) {
        return (
          <View style={{ flexDirection: 'row', marginTop: scale(2) }}>
            {this.renderText({
              text: `by ${devInfo[0]?.name}`,
              style: {
                fontFamily: fontUtils.appFont,
                color: colors.warmGrey
              }
            })}
          </View>
        )
      }
      return null
    }
 
    renderPrice=() => {
      const { item } = this.props
      const price = getPriceToRender(item)
      return this.renderText({
        text: price,
        size: 16,
        lineHeight: moderateScale(18),
        style: {
          letterSpacing: 0.23, fontFamily: fontUtils.appFontMedium, fontWeight: '500', color: colors.white
        }
      })
    }
 
    renderAddress=() => {
      const { item } = this.props
      const propAddress = getStreetInfo(item, item.type)
      return this.renderText({
        text: propAddress,
        numberOfLines: 1,
        style: {
          fontFamily: fontUtils.appFont,
          letterSpacing: 0,
          marginTop: scale(2),
          color: '#999999'
        }
      })
    }
 
    renderAreaText = () => {
      const {
        item, item: { type, title }
      } = this.props
      const area = type === 'project' ? getProjectSubtitle(title) : getPropertyArea(item)
      if (!area) { return null }
      return (
        this.renderText({
          text: area,
          style: {
            fontFamily: fontUtils.appFont,
            letterSpacing: 0,
            color: colors.white
          }
        })
      )
    }
 
    renderPrimaryText=() => {
      const { item: { type, name, title } } = this.props
      return this.renderText({
        text: type === 'project' ? name : title,
        size: 14,
        lineHeight: moderateScale(16),
        style: { marginTop: scale(10), ...fontUtils.getFontStyle('500'), color: colors.white }
      })
    }
 
    onNavigateToDetailPage = () => {
      const {
        item: {
          id, type
        },
        isLockCard
      } = this.props
      if (!isLockCard) {
        NewBridgeModule.sendNavigationEvent({
          action: 'navigateToPdp',
          params: {
            propertyId: String(id),
            type
          }
        })
      }
    }
 
    renderBottomSection = () => {
      const {
        item: {
          sellers_info,
          isComm,
          updated_at: displayDate, type,
          is_verified: isVerified, verified_at: verifiedAt
        },
        isLockCard,
        onLockCard, rmPackageName
      } = this.props
 
      const vdate = verifiedAt || displayDate
      const showVerify = getIsXDays(verifiedAt, 30)
      const finalDate = isVerified && showVerify ? vdate : displayDate
 
      if (isLockCard) {
        return (
          <TouchableOpacity
            onPress={onLockCard}
            style={styles.lockCtaLayerStyle}
            activeOpacity={1}
          >
            <PropertyLockCta hpPackageName={rmPackageName} />
          </TouchableOpacity>
        )
      }
 
      return (
        <View style={styles.bottomSectionContainer}>
          <SellerInfo
            showDate={!isEmpty(displayDate) && !isProject(type)}
            info={!isEmpty(sellers_info) ? sellers_info : []}
            isCommercial={isComm}
            displayDate={finalDate}
            verifiedAt={verifiedAt}
          />
          <CheckoutMoreBtn
            onPress={this.onNavigateToDetailPage}
          />
        </View>
 
      )
    }
 
    renderAreaAndPrice=() => {
      const {
        type, price_on_request, formatted_per_sqft_rate, show_per_sqft_rate
      } = this.props.item
      const priceText = type !== 'rent' && !price_on_request && !isEmpty(formatted_per_sqft_rate)
        ? `${formatted_per_sqft_rate} ` : ''
      const showSqftRate = type === 'project' ? show_per_sqft_rate : true
      const showPriceText = showSqftRate && priceText !== ''
      return (
        <View style={{ flexDirection: 'row', marginTop: scale(9), alignItems: 'center' }}>
          {this.renderAreaText()}
          {showPriceText && this.renderText({ text: '    ', style: { color: colors.white } })}
          {showPriceText && this.renderText({
            text: priceText,
            style: { flex: 1, fontFamily: fontUtils.appFont, color: colors.white },
            numberOfLines: 1
          })}
        </View>
      )
    }
 
    getSubTitle=() => {
      const { item } = this.props
      const { inventory_configs } = item
      if (inventory_configs && inventory_configs instanceof Array
        && !isEmpty(inventory_configs[0])) {
        const area = getPropertyArea(item)
        const furnishType = getFurnishType(inventory_configs)
        return (
          <View style={{ flexDirection: 'row', marginTop: scale(9), alignItems: 'center' }}>
            {!isEmpty(furnishType) && this.renderText({
              text: furnishType,
              lineHeight: moderateScale(16),
              style: { fontFamily: fontUtils.appFont, color: colors.white }
            })}
            {!isEmpty(furnishType) && this.renderText({ text: '    ', style: { color: colors.grey21 } })}
            {this.renderText({
              text: area,
              lineHeight: moderateScale(16),
              style: { fontFamily: fontUtils.appFont, color: colors.white }
            })}
          </View>
        )
      }
      return <View />
    }
 
    renderOffersBanner=(offersText = '') => (
      <View style={styles.offersBannerContainer}>
        <View style={{ flexDirection: 'row', alignItems: 'center' }}>
          <ImageComponent
            source={images.offerIcon}
            style={styles.offersIcon}
          />
          {this.renderText({
            text: offersText,
            color: colors.grey74
          })}
        </View>
        <ImageComponent
          source={images.filterPillDown}
          style={styles.forwardArrowIcon}
        />
      </View>
    )
 
    render() {
      const {
        item: { type, has_offer: hasOffer, offer_text: offerText },
        style, isLockCard
      } = this.props
      const containerStyle = [styles.propertyDetailsContainer, style]
      return (
        <TouchableOpacity
          activeOpacity={1}
          onPress={this.onNavigateToDetailPage}
          style={containerStyle}
        >
          {isLockCard ? (
            <>
              <FastImageBackground
                source={images.propertyCard}
                style={styles.propertyCardContainer}
                imageStyle={{
                  height: undefined,
                  width: undefined
                }}
              />
              {this.renderPropertyImage()}
              {this.renderBottomSection()}
            </>
          ) : (
            <>
              {this.renderPropertyImage()}
              <View style={{ marginVertical: scale(10), padding: scale(8) }}>
                {hasOffer && this.renderOffersBanner(offerText)}
                {this.renderPrice()}
                {this.renderPrimaryText()}
                {this.renderAddress()}
                {type !== 'rent' && this.renderAreaAndPrice()}
                {type === 'rent' && this.getSubTitle()}
              </View>
              {this.renderBottomSection()}
            </>
          )
          }
        </TouchableOpacity>
      )
    }
}
 
NearbyPropertiesCard.propTypes = {
  item: PropTypes.object.isRequired,
  navigation: PropTypes.shape({ navigate: PropTypes.func.isRequired }).isRequired,
  style: PropTypes.object,
  isLockCard: PropTypes.bool,
  onLockCard: PropTypes.func,
  rmPackageName: PropTypes.string
}
 
NearbyPropertiesCard.defaultProps = {
  style: null,
  isLockCard: false,
  onLockCard: () => {},
  rmPackageName: ''
}
 
// export default withNavigationRoute(NearbyPropertiesCard)
export default (NearbyPropertiesCard)
