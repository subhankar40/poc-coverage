import React, { PureComponent } from 'react'
import PropTypes from 'prop-types'
import {
  View, StyleSheet, TouchableNativeFeedback
} from 'react-native'
import { useNavigation } from '@react-navigation/native'
import { connect } from 'react-redux'
import { SOURCES } from 'common/constants/GAEvents'
import { showNewToast } from 'common/actions/toast'
import NewBridgeModule from 'common/utils/NewBridgeModule'
import images from 'utils/images'
import constants from 'common/utils/constants'
import isEmpty from 'common/utils/isEmpty'
import { trackEvent } from 'common/utils/trackEvents'
import { Touchable } from 'common/components/GenericButton'
import { scale } from 'common/utils/scalingFunctions'
import colors from 'utils/colors'
import resourceID from 'common/utils/resourceID'
import ImageComponent from 'common/components/ImageComponent'
import { SelectedCitySelectors } from 'selectors/selectedCity'
import VoiceSearchHomePage from './VoiceSearchHomePage'
 
const Touch = TouchableNativeFeedback
 
const styles = StyleSheet.create({
  micIcon: {
    width: 16,
    height: 21
  },
  tooltip: {
    position: 'absolute',
    top: -36,
    right: -18,
    height: 32,
    width: 56
  },
  purpleMicIcon: {
    height: scale(25),
    width: scale(25),
    tintColor: colors.purple_5d
  }
})
 
class VoiceSearch extends PureComponent {
  constructor(props) {
    super(props)
    this.state = {
      isInternetConnected: true,
      showToolTip: true
    }
    this.micIconStyle = props.placement === 'locality_search_bar' ? { height: 27, width: 20 } : styles.micIcon
    this.textString = ''
  }
 
  componentDidMount() {
    if (!this.getStatus()) {
      return
    }
    const { voiceSearchPlacement } = this.props
    this.timer = setTimeout(() => {
      this.setState({
        showToolTip: false
      })
    }, voiceSearchPlacement.toolTipTime)
    this.props.dispatch({ type: 'INIT_TEXT_SEARCH' })
  }
 
  componentWillReceiveProps(nextProps) {
    const { searchStatus } = this.props.textSearch
    const newSearchStatus = nextProps.textSearch.searchStatus
    const { extractedFilters } = nextProps.textSearch
    if (newSearchStatus !== searchStatus && newSearchStatus === 'success') {
      trackEvent('voice_search', 'success', { placement: nextProps.placement, search_string: this.textString })
      this.dispatchSearchActions(extractedFilters)
    } else if (newSearchStatus !== searchStatus && newSearchStatus === 'failure') {
      trackEvent('voice_search', 'failure', { placement: nextProps.placement, search_string: this.textString })
      this.props.dispatch(showNewToast('No Matching Results Found\nTry Typing in SearchBar'))
      this.props.dispatch({
        type: 'HIDE_LOADER'
      })
      this.props.dispatch({ type: 'INIT_TEXT_SEARCH' })
    }
  }
 
  componentWillUnmount() {
    // TODO: remove this listener after correcting netInfo implementation
    this.timer && clearTimeout(this.timer)
  }
 
  getStatus = () => {
    const {
      service, voiceSearchPlacement, source
    } = this.props
    const name = service?.name
    const shouldShow = !isEmpty(voiceSearchPlacement) ? voiceSearchPlacement[source] : false
    if (name !== 'pg' && shouldShow) {
      return true
    }
    return true
  }
 
  dispatchSearchActions = (extractedFilters) => {
    const {
      source, dispatch, placement, navigation: { reset }
    } = this.props
    const hasBase = extractedFilters.base && extractedFilters.base.length > 0
    const type = hasBase
      ? extractedFilters.base[0].type
      : null
    const projectId = hasBase
      ? extractedFilters.base[0].id
      : null
    this.props.dispatch({ type: 'CLEAR_LISTING' })
    this.props.dispatch({ type: 'INIT_TEXT_SEARCH' })
    this.props.dispatch({
      type: 'HIDE_LOADER'
    })
    const params = type === 'project' ? {
      id: projectId, type
    } : {
      initFetchListing: true
    }
    const analyticsData = {}
    analyticsData.search_type = 'voice'
    analyticsData.source = !isEmpty(source) ? source : placement
    analyticsData.search_string = this.textString
    analyticsData.placement = placement
    params.analyticsData = analyticsData
    params.source = 'voice_search'
    const route = type === 'project' ? 'dedicated' : 'search'
    reset({ index: 1, routes: [{ name: 'home' }, { name: route, params }] })
    dispatch({
      type: 'CHANGE_FILTER',
      payload: { filters: extractedFilters, source: 'voiceSearch' }
    })
  }
 
  getText = (text) => {
    console.log(text)
    if (isEmpty(text)) {
      trackEvent('voice_search', 'no_voice', { placement: this.props.placement })
    }
    this.textString = text
    this.onSpeechRecognized(text)
  }
 
  onMicPress = () => {
    const { isInternetConnected } = this.state
    if (isInternetConnected) {
      NewBridgeModule.startSpeechReco(this.getText)
      // trackEvent('search', 'voice_search', { placement: this.props.placement })
    } else {
      this.props.dispatch(showNewToast(constants.SPEECH_OFFLINE_TOAST))
    }
  }
 
  shouldCallMoondragon = () => {
    const { source } = this.props
    return !(source === 'city_select' || source === 'add_locality_header'
      || source === 'requirementGethering'
      || source === 'filters' || source === SOURCES.PDP_TRANSACTIONS)
  }
 
  onSpeechRecognized = (text) => {
    if (isEmpty(text)) {
      return
    }
    if (!this.shouldCallMoondragon()) {
      this.props.fillTextInput(text)
      return
    }
    const { selectedCity } = this.props
    this.props.dispatch({
      type: 'TEXT_SEARCH',
      payload: {
        polygonUuid: selectedCity.uuid,
        query: text
      }
    })
  }
 
  renderToolTip = () => (
    <ImageComponent style={styles.tooltip} source={images.voiceTooltip} resizeMode="contain" />
  )
 
  render() {
    const { source, placement, isDark } = this.props
    if (source === 'home_search') {
      return (
        <VoiceSearchHomePage onPressMic={this.onMicPress} />
      )
    }
    if (source === 'home_header_search' || placement === constants.LOCALITY_SEARCH_BAR_NEW) {
      return (
        <Touchable onPress={this.onMicPress}>
          <ImageComponent
            source={images.voiceSearchIconHomePage}
            style={styles.purpleMicIcon}
          />
        </Touchable>
      )
    }
    if (!this.getStatus()) {
      return <View />
    }
    return (
      <View>
        <Touch {...{ [resourceID]: 'voice_search' }} onPress={this.onMicPress} activeOpacity={0.6}>
          <View style={{ marginRight: 12, justifyContent: 'center' }}>
            {this.state.showToolTip && source !== 'top_search_bar' && this.renderToolTip()}
            <ImageComponent
              tintColor={isDark ? colors.reef : undefined}
              source={images.newMicIcon}
              style={this.micIconStyle}
              resizeMode="contain"
            />
          </View>
        </Touch>
      </View>
    )
  }
}
 
VoiceSearch.propTypes = {
  dispatch: PropTypes.func.isRequired,
  selectedCity: PropTypes.object.isRequired,
  service: PropTypes.object.isRequired,
  textSearch: PropTypes.object.isRequired,
  source: PropTypes.string,
  placement: PropTypes.string.isRequired,
  locality: PropTypes.string.isRequired,
  fillTextInput: PropTypes.func,
  voiceSearchPlacement: PropTypes.object.isRequired,
  navigation: PropTypes.object.isRequired,
  isDark: PropTypes.bool
}
 
VoiceSearch.defaultProps = {
  source: null,
  fillTextInput: null,
  isDark: false
}
 
const VoiceSearchComponent = (props) => {
  const navigation = useNavigation()
  return <VoiceSearch {...props} navigation={navigation} />
}
