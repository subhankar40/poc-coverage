import React, { PureComponent } from 'react'
import PropTypes from 'prop-types'
import { connect } from 'react-redux'
import { StyleSheet, View } from 'react-native'
import { OtpRetrieval } from '@housing/react-native-sms-retriever'
import GenericText from 'common/components/react-native-generic-text'
import Button, { Touchable } from 'common/components/GenericButton'
import isEmpty from 'common/utils/isEmpty'
import colors from 'utils/colors'
 
const styles = StyleSheet.create({
  container: {
    padding: 16
  },
  linkLike: {
    color: colors.purple,
    lineHeight: 25
  },
  editButton: {
    width: 50,
    height: 30
  },
  textInput: {
    marginTop: 38
  },
  resendButton: {
    height: 30,
    width: 60
  }
})
 
class ProfileOTPView extends PureComponent {
  constructor(props) {
    super(props)
    this.state = {
      otp: null
    }
  }
 
  componentDidMount() {
    this.sendVerificationOTP()
  }
 
  onChangeotp = otp => this.setState({ otp })
 
  onSubmit = () => {
    if (this.state.otp && this.otpRef && this.otpRef.isValid) {
      const { phone, countryCode, navigation } = this.props
      this.props.dispatch({
        type: 'PROFILE_VERIFY_OTP',
        payload: {
          phone,
          country_url_name: countryCode,
          otp: this.state.otp,
          method: 'phone',
          login_auth_token: this.props.user.login_auth_token,
          callback: (isVerified) => {
            if (isVerified) {
              navigation?.goBack?.()
            }
          }
        }
      })
    }
  }
 
  editNumber = () => this.props.navigation.goBack(null)
 
  sendVerificationOTP = () => {
    const { phone, countryCode, user } = this.props
    const { phone_number } = user //eslint-disable-line
    if (
      // eslint-disable-next-line
      (!phone_number || phone.toString() !== phone_number.toString()) &&
      user
      && user.login_auth_token
    ) {
      this.props.dispatch({
        type: 'REGISTER_USER_PHONE',
        payload: {
          phone,
          country_url_name: countryCode,
          login_auth_token: user.login_auth_token
        }
      })
      return
    }
    this.props.dispatch({
      type: 'SEND_VERIFICATION_OTP',
      payload: {
        phone,
        country_url_name: countryCode,
        method: 'phone'
      }
    })
  }
 
  onResend = () => {
    this.otpRef && this.otpRef.resendClicked && this.otpRef.resendClicked()
    this.sendVerificationOTP()
  }
 
  render() {
    const { otp } = this.state
    return (
      <View style={styles.container}>
        <GenericText>Please verify your number.</GenericText>
        <View style={{ flexDirection: 'row' }}>
          <GenericText>
            {`OTP sent to ${this.props.phone}`}
          </GenericText>
          <Touchable onPress={this.editNumber} style={styles.editButton}>
            <GenericText style={styles.linkLike} align="center">
              {' '}
              Edit
            </GenericText>
          </Touchable>
        </View>
        <View style={styles.textInput}>
          <OtpRetrieval
            type="otp"
            value={otp}
            refCallback={(node) => {
              this.otpRef = node
            }}
            autoCorrect={false}
            autoCapitalize="none"
            autoFocus
            onChangeText={this.onChangeotp}
            position={this.props.abTesting.otpPosition}
            placeholder="Verify OTP"
            errorText="Enter a valid otp"
          />
        </View>
        <View style={{ flexDirection: 'row' }}>
          <GenericText>Haven't received OTP yet?</GenericText>
          <Touchable onPress={this.onResend} style={styles.resendButton}>
            <GenericText style={styles.linkLike} align="center">
              {' '}
              Resend
            </GenericText>
          </Touchable>
        </View>
        <Button
          type="primary"
          onPress={this.onSubmit}
          disabled={isEmpty(otp)}
          marginLeft={0}
          marginRight={0}
        >
          Verify
        </Button>
      </View>
    )
  }
}
 
ProfileOTPView.propTypes = {
  navigation: PropTypes.shape({
    goBack: PropTypes.func.isRequired
  }).isRequired,
  phone: PropTypes.string.isRequired,
  countryCode: PropTypes.string.isRequired,
  dispatch: PropTypes.func.isRequired,
  user: PropTypes.object.isRequired,
  abTesting: PropTypes.shape({
    otpPosition: PropTypes.number.isRequired
  }).isRequired
}
 
export default connect(({ abTesting }) => ({ abTesting }))(ProfileOTPView)
