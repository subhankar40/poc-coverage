import React, { useCallback } from 'react'
import { View, Text, TouchableOpacity } from 'react-native'
import LinearGradient from 'react-native-linear-gradient'
import PropTypes from 'prop-types'
import HPCardTag from './HPCardTag'
import HPCheckCircle from './HPCheckCircle'
import HpEmojiIcon from './HpEmojiIcon'
import HPPriceCard from './HPPriceCard'
import HPSeeMoreItem from './HPSeeMoreItem'
import HPTickItem from './HPTickItem'
import HPGradientLayer from './HPGradientLayer'
import { hpCardVariant2Style as styles } from './styles'
import { strings, GRADIENT_COLORS, GRADIENT_CONST } from './utils'
 
const getInActiveLayer = style => ({
  colors: GRADIENT_COLORS.COLOR_4,
  start: GRADIENT_CONST.XY__01_01,
  end: GRADIENT_CONST.XY_0308,
  style: [styles.main, styles.mainUnSelectedViewStyle, style]
})
const getActiveLayer = style => ({
  colors: GRADIENT_COLORS.COLOR_5,
  start: GRADIENT_CONST.XY_0801,
  end: GRADIENT_CONST.XY__021,
  style: [styles.main, styles.mainSelectedViewStyle, style]
})
 
const getAppropriateStyle = (isSelected, style) => (isSelected ? getActiveLayer(style)
  : getInActiveLayer(style))
 
function HPCardVariant2(props) {
  const {
    item, isSelected, isPopular, isMoneyBack, style,
    onGstPress, onSeeMore, onCardClick, couponMetaInfo, showFallbackPrice,
    discountMetoInfo
  } = props
 
  const onGstClick = useCallback(() => {
    onGstPress?.(item)
  }, [item, couponMetaInfo, showFallbackPrice, onGstPress])
  const onSeeMoreClick = useCallback(() => {
    onSeeMore?.(item)
  }, [item, couponMetaInfo, showFallbackPrice, onSeeMore])
  const onClick = useCallback(() => {
    onCardClick?.(item)
  }, [item, couponMetaInfo, showFallbackPrice, onCardClick])
  return (
    <View>
      <LinearGradient
        {...getAppropriateStyle(isSelected, style)}
      >
        <HPGradientLayer
          isActive={isSelected}
          variant="b"
        />
        <TouchableOpacity
          onPress={onClick}
        >
          <View style={styles.firstRowStyle}>
            <HPCheckCircle
              isActive={isSelected}
              style={styles.circleCheckStyle}
            />
            <View style={styles.emijiTitleStyle}>
              <HpEmojiIcon
                size={28}
                emojiType={item?.priority}
                style={styles.emojiStyle}
              />
              <Text style={styles.titleStyle}>{item?.product_detail?.[0]?.name || ''}</Text>
            </View>
            <HPTickItem
              title={item?.product_detail?.[0]?.benefits?.[0] || ''}
              titleStyle={styles.tickTitleStyle}
            />
            <HPSeeMoreItem
              onPress={onSeeMoreClick}
              title={strings.SEE_ALL_FEATURES}
              style={styles.seeMoreStyle}
            />
          </View>
          <View style={styles.secondRowStyle}>
            <HPPriceCard
              item={item}
              variant="b"
              skinStyle={{ borderRadius: 8 }}
              onGstPress={onGstClick}
              couponMetaInfo={couponMetaInfo}
              showFallbackPrice={showFallbackPrice}
              discountMetoInfo={discountMetoInfo}
            />
          </View>
        </TouchableOpacity>
      </LinearGradient>
      {(isPopular || isMoneyBack) ? (
        <HPCardTag
          isPopular={isPopular}
          title={isMoneyBack
            ? item?.product_detail?.[0]?.golden_tag || strings.MONEYBACK_GUARANTEE3
            : strings.MOST_POPULAR}
          titleStyle={styles.tagTxtStyle}
          style={styles.tagWrapperStyle}
          contentStyle={styles.tagContentStyle}
          item={item}
        />
      ) : null}
    </View>
  )
}
 
HPCardVariant2.propTypes = {
  item: PropTypes.object,
  isSelected: PropTypes.bool,
  isPopular: PropTypes.bool,
  isMoneyBack: PropTypes.bool,
  style: PropTypes.object,
  onGstPress: PropTypes.func,
  onSeeMore: PropTypes.func,
  onCardClick: PropTypes.func,
  couponMetaInfo: PropTypes.object,
  showFallbackPrice: PropTypes.bool,
  discountMetoInfo: PropTypes.object
}
HPCardVariant2.defaultProps = {
  item: {},
  isSelected: false,
  isPopular: false,
  isMoneyBack: false,
  style: null,
  onGstPress: () => {},
  onSeeMore: () => {},
  onCardClick: () => {},
  couponMetaInfo: null,
  showFallbackPrice: false,
  discountMetoInfo: null
}
 
export default HPCardVariant2
