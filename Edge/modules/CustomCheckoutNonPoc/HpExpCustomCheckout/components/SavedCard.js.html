import React, {
  useMemo, memo, useEffect, useCallback, useState
} from 'react'
import { View, Text, Pressable } from 'react-native'
import GradientWrapper from 'common/components/commonComponents/GradientWrapper'
import { useStyles } from 'common/utils/hooks/useStyles'
import ImageComponent from 'common/components/ImageComponent'
import isEmpty from 'common/utils/isEmpty'
import CheckBox from '../../components/CheckBox'
import { GRAD_COLORS, GRAD_COORDS } from '../../colors'
import FloatingLabelTextInput from '../../components/FloatingLabelTextInput'
import { CVV_INPUTFIELD_STYLE_CONFIG } from '../../utils'
import { lastFourDightStyle, createSavedCardStyles } from './styles'
import strings from '../../strings'
 
const NO_DOTS_PLACEHOLDER = 4
 
const LastDigits = (props) => {
  const { lastDigits = '' } = props
  const dots = useMemo(() => Array.from({ length: NO_DOTS_PLACEHOLDER }, (_, index) => index), [])
  return (
    <View style={lastFourDightStyle.main}>
      <View style={lastFourDightStyle.dotRowStyle}>
        {dots.map(() => (<View style={lastFourDightStyle.dotStyle} />))}
      </View>
      <Text style={lastFourDightStyle.titleStyle}>{lastDigits}</Text>
    </View>
  )
}
 
function SavedCard(props) {
  const {
    onPress, removeBottomLine, item, style, contentStyle, isActive,
    cvv, onChangeText, isKeyboardOpen, onCenterSavedCards
  } = props
  const [layoutConfig, setConfig] = useState({})
  const styles = useStyles(createSavedCardStyles, { mainStyle: style, contentStyle })
 
  useEffect(() => {
    if (isActive && isKeyboardOpen && !isEmpty(layoutConfig)) {
      onCenterSavedCards?.(layoutConfig?.y + 10)
    }
  }, [layoutConfig, isKeyboardOpen, isActive])
 
  const onViewLayout = useCallback((e) => {
    setConfig(e?.nativeEvent?.layout)
  }, [layoutConfig])
  return (
    <View
      onLayout={onViewLayout}
      style={styles.main}
    >
      <Pressable
        style={styles.contentWrapperStyle}
        onPress={onPress}
      >
        {isActive ? (
          <GradientWrapper
            style={styles.bgLayerStyle}
            colors={GRAD_COLORS.color4}
            {...GRAD_COORDS.COORD4}
          />
        ) : null}
        <View style={styles.content}>
          <View style={styles.networkWrapperStyle}>
            <ImageComponent
              source={{ uri: item?.card?.network_icon }}
              style={styles.networkIconStyle}
            />
          </View>
          <LastDigits lastDigits={item?.card?.last4 || ''} />
          <CheckBox isActive={isActive} onPress={onPress} />
        </View>
        {isActive ? (
          <View style={styles.cvvViewWrapperStyle}>
            <Text style={styles.cvvTitleStyle}>{strings.ent_cvv}</Text>
            <FloatingLabelTextInput
              placeholder={strings.cvv}
              theme={CVV_INPUTFIELD_STYLE_CONFIG}
              autoFocus
              value={cvv}
              onChangeText={onChangeText}
              maxLength={4}
              secureTextEntry
            />
          </View>
        ) : null}
      </Pressable>
      {!removeBottomLine ? <View style={styles.bottomLineStyle} /> : null}
    </View>
  )
}
 
export default memo(SavedCard)
