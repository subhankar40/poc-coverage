import React, { memo, useEffect, useState } from 'react'
import {
  Keyboard, View, Pressable, ActivityIndicator
} from 'react-native'
import FastImage from 'react-native-fast-image'
import LinearGradient from 'react-native-linear-gradient'
import { useNavigation } from '@react-navigation/native'
import { useDispatch, useSelector } from 'react-redux'
import { useSafeAreaInsets } from 'react-native-safe-area-context'
 
import ImageComponent from 'common/components/ImageComponent'
import { useStyles } from 'common/utils/hooks/useStyles'
import { otherServicesStrings } from 'constants/strings'
import { SCREENS } from 'constants/screens'
import { GRADIENT_CONST } from 'components/HpRevampv1/utils'
import { showModal, hideModal } from 'components/ModalBox'
import { SimpleText, fontType, textTypes } from 'common/components/SimpleText'
import { UserSelectors } from 'selectors/user'
import colors from 'utils/colors'
import images from 'utils/images'
import isEmpty from 'utils/isEmpty'
import { styles } from 'modules/POC_Revamp_v2/styles/PocProfileStyles'
import {
  cameraImageStyles,
  flexRowV1Styles, flexRowV2Styles
} from 'modules/POC_Revamp_v2/styles/CommonStyles'
import { PocProfileActions } from 'modules/POC_Revamp_v2/actions/profileActions'
import { PocProfileSelectors } from 'modules/POC_Revamp_v2/selectors/profileSelector'
import EditProfile from 'modules/POC_Revamp_v2/components/EditProfile'
import LogoutModal from 'modules/POC_Revamp_v2/components/LogoutModal'
import { profileGradientColors } from 'modules/POC_Revamp_v2/utils/utils'
import { capitalizeFirstLetter } from 'utils/helpers'
import { trackEvent } from 'utils/trackEvents'
import { EA, EC } from 'constants/GAEvents'
import PersonalPicSelectionModal from './PersonalPicSelectionModal'
 
const PocProfile = () => {
  const [showLogoutModal, setShowLogoutModal] = useState(false)
  const dispatch = useDispatch()
  const navigation = useNavigation()
  const inset = useSafeAreaInsets()
  const bgImage = useStyles(styles.bg, {inset} )
  const name = useSelector(PocProfileSelectors.getTenantName) || ''
  const avatarImage = useSelector(PocProfileSelectors.getSelectedAvatarImage)
  const imageData = useSelector(PocProfileSelectors.getImageData)
  const { userPhone: tenantPhoneNumber, countryCode } = useSelector(
    UserSelectors.getUserPhoneAndEmail
  )
  const [showPersonalPicModal, setShowPersonalPicModal] = useState(false)
  const showDpLoader = useSelector(PocProfileSelectors.getUpdateProfileLoader)
 
  const closePersonalPicSelection = () => {
    setShowPersonalPicModal(false)
  }
 
  const onShowLogoutModal = () => {
    setShowLogoutModal(showLogoutModal => !showLogoutModal)
  }
 
  const onCloseLogoutModal = () => {
    setShowLogoutModal(false)
  }
 
  const navigateToMyTrans = () => {
    trackEvent(EC.POC_PROFILE, EA.MY_TRANSACTIONS_CLICKED)
    navigation.navigate(SCREENS.MTP_SCREEN)
  }
 
  const navigateToHS = () => {
    trackEvent(EC.POC_PROFILE, EA.HELP_AND_SUPPORT_CLICKED)
    navigation.navigate(SCREENS.HELP_AND_SUPPORT_SCREEN)
  }
 
  const closeEditModal = () => {
    if (Keyboard.isVisible) {
      Keyboard.dismiss()
    }
    dispatch(PocProfileActions.setNewAvatar(null))
    hideModal()
  }
 
  const openEditModal = () => {
    trackEvent(EC.POC_PROFILE, EA.EDIT_CLICKED)
    showModal({
      renderedComponent: () => (
        <EditProfile
          closeModal={closeEditModal}
        />
      ),
      showDarkTheme: true,
      panGestureHeaderStyles: styles.zIndex0,
      modalHeaderComponent: () => <View />,
      headerComponent: () => <View />,
      onModalClose: () => {
        dispatch(PocProfileActions.setImageData(null))
        dispatch(PocProfileActions.setNewAvatar(null))
      }
    })
  }
 
  const openPersonalPicSelection = () => {
    if (showDpLoader) { return }
    setShowPersonalPicModal(true)
  }
 
  const renderDp = () => {
    if (!showDpLoader) {
      if (!isEmpty(imageData)) {
        return (
          <ImageComponent
            source={{ uri: imageData?.uri }}
            style={styles.profilePicStyles}
            resizeMode="contain"
          />
        )
      }
      return (
        <ImageComponent
          source={{ uri: avatarImage }}
          style={styles.profilePicStyles}
          resizeMode="contain"
        />
      )
    }
    return (
      <View style={styles.loaderContainer}>
        <ActivityIndicator color={colors.pink_ff_5a} size={26} />
      </View>
    )
  }
 
  useEffect(() => {
    dispatch(PocProfileActions.fetchAllAvatarsReq())
  }, [])
 
  return (
    <View style={styles.fullScreen}>
      <FastImage
        source={images.profileHeaderBg}
        style={bgImage}
      >
        <View style={flexRowV2Styles}>
          <View style={flexRowV1Styles}>
            <Pressable onPress={openPersonalPicSelection}>
              {renderDp()}
              <View style={styles.cameraIcon}>
                <ImageComponent
                  source={images.camera}
                  style={cameraImageStyles}
                  resizeMode="contain"
                />
              </View>
            </Pressable>
            <View style={styles.marginLeft16}>
              {!isEmpty(name) && (
                <SimpleText
                  text={`${capitalizeFirstLetter(name)}`}
                  type={textTypes.regular}
                  fontType={fontType.medium}
                  style={styles.nameStyles}
                />
              )}
              {!isEmpty(tenantPhoneNumber) && (
                <SimpleText
                  text={`${countryCode}-${tenantPhoneNumber}`}
                  type={textTypes.small}
                  fontType={fontType.normal}
                  style={styles.phoneNumStyles}
                />
              )}
            </View>
          </View>
          <Pressable onPress={openEditModal}>
            <ImageComponent
              source={images.editButton}
              style={styles.editImgStyles}
              resizeMode="contain"
            />
          </Pressable>
        </View>
      </FastImage>
      <LinearGradient
        colors={profileGradientColors}
        style={styles.linearGradientStyles}
        start={GRADIENT_CONST.XY_00}
        end={GRADIENT_CONST.XY_01}
      />
      <View style={styles.contentStyles}>
        <Pressable style={flexRowV2Styles} onPress={navigateToMyTrans}>
          <View style={flexRowV1Styles}>
            <ImageComponent
              source={images.payRentTransactionIcon}
              style={styles.iconImgStyles}
              resizeMode="contain"
            />
 
            <SimpleText
              text={otherServicesStrings.myTransactions}
              type={textTypes.small}
              fontType={fontType.normal}
              style={styles.titleStyles}
            />
          </View>
          <ImageComponent
            source={images.rightArrowIcon}
            style={styles.forwardArrowImg}
            tintColor={colors.grey_c1}
            resizeMode="contain"
          />
        </Pressable>
        <View style={styles.dividerStyles} />
        <Pressable style={flexRowV2Styles} onPress={navigateToHS}>
          <View style={flexRowV1Styles}>
            <ImageComponent
              source={images.payRentHelpIcon}
              style={styles.iconImgStyles}
              resizeMode="contain"
            />
 
            <SimpleText
              text={otherServicesStrings.helpAndSupport}
              type={textTypes.small}
              fontType={fontType.normal}
              style={styles.titleStyles}
            />
          </View>
          <ImageComponent
            source={images.rightArrowIcon}
            style={styles.forwardArrowImg}
            tintColor={colors.grey_c1}
            resizeMode="contain"
          />
        </Pressable>
        <View style={styles.dividerStyles} />
        <Pressable onPress={onShowLogoutModal}>
          <View style={flexRowV2Styles}>
            <View style={flexRowV1Styles}>
              <ImageComponent
                source={images.logoutIcon}
                style={styles.iconImgStyles}
                tintColor={colors.grey_a7}
                resizeMode="contain"
              />
              <SimpleText
                text={otherServicesStrings.logOut}
                type={textTypes.small}
                fontType={fontType.normal}
                style={styles.titleStyles}
              />
            </View>
          </View>
        </Pressable>
      </View>
      <LogoutModal showLogoutModal={showLogoutModal} closeModals={onCloseLogoutModal} />
      <PersonalPicSelectionModal
        showPersonalPicModal={showPersonalPicModal}
        onClose={closePersonalPicSelection}
        directlyUploadImage
      />
    </View>
  )
}
 
export default memo(PocProfile)
