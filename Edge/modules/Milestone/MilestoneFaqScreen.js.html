const MilestoneFaqScreen = () => {
  const { top: topInset } = useSafeAreaInsets()
  const styles = useStyles(createStyle, { topInset })
  const navigation = useNavigation()
  const dispatch = useDispatch()
 
  const milestoneFaqs = useSelector(MilestoneSelectors.milestoneFaqs)
  const diamondCount = useSelector(MilestoneSelectors.getDiamondMaxCount)
 
  useEffect(() => {
    if (isEmpty(milestoneFaqs)) {
      dispatch(MilestoneActions.fetchMilestoneFaqs())
    }
  }, [])
 
  const onBackPress = () => {
    if (navigation.canGoBack()) {
      navigation.goBack()
    } else {
      CredPayUtils.navigateToPayOnCredit()
    }
  }
 
  const onClose = () => {
    navigation.navigate(SCREENS.MILESTONE_SCREEN)
  }
 
  const renderHeader = useMemo(() => (
    <>
      <View style={styles.statusBar} />
      <View style={styles.headerWrapper}>
        <HeaderBg />
        <Pressable style={styles.btnWrapper} onPress={onBackPress}>
          <ImageComponent
            source={images.backIcon3d}
            style={styles.icons}
            resizeMode="contain"
          />
        </Pressable>
        <ImageComponent
          source={images.finopolyIcon}
          style={styles.finopoly}
          resizeMode="cover"
        />
        <HeaderStreak />
        <Pressable style={[styles.btnWrapper, styles.leftSpace]} onPress={onClose}>
          <ImageComponent
            source={images.crossCircle}
            style={styles.icons}
            resizeMode="contain"
          />
        </Pressable>
      </View>
    </>
  ), [])
 
  const onKnowMoreClick = () => {
    navigation.navigate(SCREENS.MILESTONE_STORY_SCREEN)
  }
 
  const renderHowToPlay = useMemo(() => (
    <View style={styles.topSpace}>
      <SimpleText
        text={milestoneStrings.howToPlay}
        style={styles.htp}
        fontType={fontType.medium}
      />
      <Pressable
        style={styles.knowMoreCta}
        onPress={onKnowMoreClick}
      >
        <LottieViewWrapper
          source={images.milestoneStory3}
          progress={0.15}
          resizeMode="cover"
          style={styles.coverIcon}
        />
        <View style={styles.titleTxt}>
          <View style={styles.row}>
            <SimpleText
              text="Earn "
              type={textTypes.large}
              fontType={fontType.medium}
              style={styles.lightWhite}
            />
            <SimpleText
              text={milestoneStrings.diamond(diamondCount)}
              type={textTypes.large}
              fontType={fontType.medium}
              style={styles.lightBlue}
            />
          </View>
          <SimpleText
            text={milestoneStrings.collectDiamondTo(diamondCount)}
            type={textTypes.small}
            style={styles.lightWhite}
          />
        </View>
        <View style={styles.poster}>
          <ImageComponent
            source={images.play}
            style={styles.posterIcon}
          />
        </View>
      </Pressable>
    </View>
  ), [])
 
  // eslint-disable-next-line react/prop-types
  const renderFaqItem = ({ item, index }) => (
    <ExpandableCard
      key={item.qus ? item.qus : item.question}
      item={item}
      index={index}
      showDivider={index !== milestoneFaqs.length - 1}
      darkMode
      questionOpacity={0.8}
    />
  )
 
  const renderFaqs = useMemo(() => (
    <View style={styles.faqWrapper}>
      <SimpleText
        text={milestoneStrings.faq}
        style={styles.faqTitle}
        fontType={fontType.medium}
      />
      <FlatList
        data={milestoneFaqs || []}
        showsVerticalScrollIndicator={false}
        renderItem={renderFaqItem}
        keyExtractor={item => item.id}
        style={styles.verticalSpace}
      />
    </View>
  ), [milestoneFaqs])
 
  const renderBottomSection = useMemo(() => (
    <View
      style={styles.cardsWrapper}
    >
      <BottomBg style={styles.bottomBg} />
      {renderHowToPlay}
      {renderFaqs}
    </View>
  ), [renderFaqs])
 
  return (
    <View style={styles.screenWrapper}>
      {renderHeader}
      {renderBottomSection}
    </View>
  )
}
 
export default MilestoneFaqScreen
