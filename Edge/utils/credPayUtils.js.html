export const parseRewardsHistoryData = (inputData, currentData = null) => {
  const outputData = Object.assign([], currentData)
 
  inputData.forEach((item) => {
    if (
      !isEmpty(item?.reward?.createdAt)
      && [
        rewardTypes.COIN,
        rewardTypes.SUBSCRIPTION,
        rewardTypes.SCRATCH_CARD,
        rewardTypes.REFERRAL,
        rewardTypes.CASHBACK,
        rewardTypes.WALLET,
        rewardTypes.DISCOUNT,
        rewardTypes.MILESTONE
      ].includes(item?.reward?.rewardType)
    ) {
      const key = new Date(item.reward.createdAt).getFullYear()
      const indx = findIndexIfPresent(outputData, key)
      if (indx !== -1) {
        const dataAtIndex = outputData[indx].data
        dataAtIndex.push(item)
      } else {
        const obj = {
          title: key,
          data: [item]
        }
        outputData.push(obj)
      }
    }
  })
  return outputData
}

const getServiceToShow = (expPastTransactionData, rentTypeId, variant, rpEligible, subscribed) => {
  if (variant === 1) {
    return cardServices.default
  }
  const isRentalService = rentTypeId === 1 || rentTypeId === 2
  const {
    rentalTransaction = [], nonRentalTransaction = []
  } = expPastTransactionData || {}
 
  // Rental variables
  const rentalArLength = rentalTransaction?.length
  const rpShownCount = rentalTransaction.filter(
    itm => itm.service === cardServices.rentProtect
  ).length
  const rpPurchasedCount = rentalTransaction.filter(
    itm => itm.service === cardServices.rentProtect && itm.purchased
  ).length
  const cpShownCount = rentalTransaction.filter(
    itm => itm.service === cardServices.creditPlus
  ).length
  const cpPurchasedCount = rentalTransaction.filter(
    itm => itm.service === cardServices.creditPlus && itm.purchased
  ).length
 
  // NOTE: `.reverse` method changes the original referenced Object/Array
  // ie: it is changing data in redux so spreading data in new Array
  const lastVisibleService = [...rentalTransaction]
    .reverse()
    .find(
      itm => itm.service === cardServices.creditPlus
        || itm.service === cardServices.rentProtect
    )
  const lastTransaction = rentalTransaction[rentalArLength - 1]
 
  // Non Rental variables
  const nrCpCount = nonRentalTransaction.filter(
    itm => itm.service === cardServices.creditPlus
  ).length
  const nrCpPurchasedCount = nonRentalTransaction.filter(
    itm => itm.service === cardServices.creditPlus && itm.purchased
  ).length
 
  if (variant === 2) {
    if (isRentalService) {
      // 1(a)(i) If nothing shown in last 3 payments show CP (if not eligible for CP show RP)
      if (
        rpShownCount === 0
        && cpShownCount === 0
      ) {
        if (!subscribed) {
          return cardServices.creditPlus
        }
        if (rpEligible) {
          return cardServices.rentProtect
        }
      }
      // 1(a)(ii) If both CP and RP shown at least once AND none purchased, don't show anything
      if (
        rpShownCount > 0
        && rpPurchasedCount === 0
        && cpShownCount > 0
        && cpPurchasedCount === 0
      ) {
        return cardServices.none
      }
      // 1(a)(iii) CP shown on last payment, show RP
      if (lastTransaction?.service === cardServices.creditPlus && rpEligible) {
        return cardServices.rentProtect
      }
      // 1(a)(iv) RP shown on last payment, show CP
      if (lastTransaction?.service === cardServices.rentProtect && !subscribed) {
        return cardServices.creditPlus
      }
      // 1(a)(v) If nothing shown on last payment flip whatever was shown last
      if (
        lastTransaction?.service === cardServices.none
        && lastVisibleService?.service === cardServices.rentProtect
        && !subscribed
      ) {
        return cardServices.creditPlus
      }
      // 1(a)(v) If nothing shown on last payment flip whatever was shown last
      if (
        lastTransaction?.service === cardServices.none
        && lastVisibleService?.service === cardServices.creditPlus
        && rpEligible
      ) {
        return cardServices.rentProtect
      }
      return cardServices.none
    }
    if (!isRentalService) {
      // 2(a)(i) If CP shown <2 times in last 3 payments show CP
      if (nrCpCount < 2 && !subscribed) {
        return cardServices.creditPlus
      }
      // 2(a)(ii) If CP shown >=2 times AND it is not purchased any time, dont show anything
      if (nrCpCount >= 2 && nrCpPurchasedCount === 0) {
        return cardServices.none
      }
      // 2(a)(iii) If CP shown >=2 times AND purchased >=1 time
      // AND user not subscribed currently, show CP
      if (nrCpCount >= 2 && nrCpPurchasedCount >= 1 && !subscribed) {
        return cardServices.creditPlus
      }
    }
    return cardServices.none
  }
  // For variant 3 we're showing RP without RP eligible check
  if (variant === 3) {
    if (isRentalService) {
      // 1(a)(i) If nothing shown in last 3 payments show CP (if not eligible for CP show RP)
      if (
        rpShownCount === 0
        && cpShownCount === 0
      ) {
        if (!subscribed) {
          return cardServices.creditPlus
        }
        if (rpEligible) {
          return cardServices.rentProtect
        }
        return cardServices.none
      }
      // 1(a)(ii) If both CP and RP shown at least once AND none purchased, don't show anything
      if (
        (rpShownCount > 0 && rpPurchasedCount === 0)
        && (cpShownCount > 0 && cpPurchasedCount === 0)
      ) {
        return cardServices.none
      }
      // 1(a)(iii) If CP shown on last payment, show RP
      if (lastTransaction?.service === cardServices.creditPlus && rpEligible) {
        return cardServices.rentProtect
      }
      // 1(a)(iv)-1 If RP shown on last payment AND
      // User purchased RP, show RP
      if (
        lastTransaction?.service === cardServices.rentProtect
        && lastTransaction?.purchased
        && rpEligible
      ) {
        return cardServices.rentProtect
      }
      // 1(a)(iv)-2 If RP shown on last payment AND
      // User didnt purchase RP, show CP
      if (
        lastTransaction?.service === cardServices.rentProtect
        && !lastTransaction?.purchased
        && !subscribed
      ) {
        return cardServices.creditPlus
      }
      // 1(a)(v) If nothing shown on last payment flip whatever was shown last
      if (
        lastTransaction?.service === cardServices.none
        && lastVisibleService?.service === cardServices.creditPlus
        && rpEligible
      ) {
        return cardServices.rentProtect
      }
      // 1(a)(v) If nothing shown on last payment flip whatever was shown last
      if (
        lastTransaction?.service === cardServices.none
        && lastVisibleService?.service === cardServices.rentProtect
        && !subscribed
      ) {
        return cardServices.creditPlus
      }
    }
    if (!isRentalService) {
      if (nrCpCount < 2 && !subscribed) {
        return cardServices.creditPlus
      }
      if (nrCpCount >= 2 && nrCpPurchasedCount === 0) {
        return cardServices.none
      }
      if (nrCpCount >= 2 && nrCpPurchasedCount >= 1 && !subscribed) {
        return cardServices.creditPlus
      }
    }
    return cardServices.none
  }
  return cardServices.none
}
