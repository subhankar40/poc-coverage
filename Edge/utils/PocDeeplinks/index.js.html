  isRewardsPageDeeplink, isDeeplinkForPath, MY_REWARDS, INSTANT_LOAN,
  RENT_NOW_PAY_LATER, getPayLaterParamsDeeplink, HOME_LOANS, NIRO_PL, PERSONAL_LOAN,
  MY_TRANSACTIONS, isNiroPlDeeplink, MILESTONE, CREDIT_LINE, LOANS, EDIT_FLOW,
  getMyTransactionParamsDeeplink

export function navigateToRNPL(routeParams) {
  const params = {
    ...routeParams,
    fromScreen: 'direct_link'
  }
  resetActionForInstantLoanLanding([{ routeName: SCREENS.BNPL_SCREEN, routeParams: params }])
  return HIDE_LOADER_ACTION
}
 
export function navigateToLoansLanding() {
  resetActionForInstantLoanLanding()
}
 
export function navgiateToPLNiro() {
  const params = {
    fromScreen: 'direct_link'
  }
  resetActionForInstantLoanLanding([{ routeName: SCREENS.NIRO_PL_WEB_VIEW, routeParams: params }])
  return HIDE_LOADER_ACTION
}

export function navigateToCreditLine(url) {
  const params = {
    fromScreen: 'direct_link',
    url: isEmpty(url) ? 'https://housing.com/edge/line-of-credit' : url
  }
  resetActionForInstantLoanLanding([
    { routeName: SCREENS.CREDIT_LINE_WEBVIEW, routeParams: params }
  ])
  return HIDE_LOADER_ACTION
}
 
export const handlePOCv2Deeplinks = ({ url }) => (dispatch, getState) => {
  const isLoggedIn = UserSelectors.getIsLoggedIn(getState())
  const { PAY_RENT } = attributeNames
  let isPOCUrl = true
  if (POCSelectors.getIsPOCRevampV2(getState())) {
    // POC Revamp v2 handling
    const POCv2LoansCommonRoute = {
      routeName: SCREENS.PAY_ON_CREDIT_V2,
      routeParams: { tabName: PocTabRoutes.loansTab }
    }
 
    const isLoggedInAndRentPaid = isLoggedIn && POCSelectors.getIsRentPaidPreviously(getState())
    if (isDeeplinkForPath(url, MY_REWARDS)) {
      // Rewards flow
      if (isLoggedInAndRentPaid) {
        resetActionForPayOnCredit(null, { tabName: PocTabRoutes.rewardsTab }) // Tested
      } else {
        resetActionForPayOnCredit()
      }
    }
    else if (url.includes(PAY_RENT)) {
      // Pay On Credit
      CredPayUtils.navigateToPayOnCredit({ from: PocUtils.fromNavScreen.direct }) // Tested
    }
    else if (isDeeplinkForPath(url, INSTANT_LOAN) || isDeeplinkForPath(url, LOANS)) {
      // LOANS LANDING
      if (isLoggedInAndRentPaid) {
        CredPayUtils.navigateToPayOnCredit({ tabName: PocTabRoutes.loansTab })
      } else {
        navigateToLoansLanding()
      }
    }
    else if (isDeeplinkForPath(url, EDIT_FLOW)) {
      // Edit Flow
      const params = {
        tabName: PocTabRoutes.paymentsTab,
        openCardDetails: isLoggedInAndRentPaid ? true : false
      }
      resetActionForPayOnCredit(null, params)
    }
    else if (isDeeplinkForPath(url, RENT_NOW_PAY_LATER)) {
      // RNPL FLOW
      const paramsPayLater = getPayLaterParamsDeeplink(url)
      if (isLoggedInAndRentPaid) {
        const routeNames = [
          POCv2LoansCommonRoute,
          { routeName: SCREENS.BNPL_SCREEN, routeParams: paramsPayLater }
        ]
        resetActionForPayOnCredit(routeNames) // Tested
      } else {
        navigateToRNPL(paramsPayLater) // Tested
      }
    }
    else if (isDeeplinkForPath(url, HOME_LOANS)) {
      // HOME LOANS FLOW
      if (isLoggedInAndRentPaid) {
        const routeNames = [
          POCv2LoansCommonRoute,
          { routeName: SCREENS.HOME_LOAN_LANDING_SCREEN, routeParams: {} }
        ]
        resetActionForPayOnCredit(routeNames) // Tested
      } else {
        navigateToHomeLoans() // Tested
      }
    }
    else if (isDeeplinkForPath(url, NIRO_PL)) {
      // QUICK CASH FLOW
      if (isLoggedInAndRentPaid) {
        const routeNames = [
          POCv2LoansCommonRoute,
          { routeName: SCREENS.NIRO_PL_WEB_VIEW, routeParams: {} }
        ]
        resetActionForPayOnCredit(routeNames) // Tested
      } else {
        navgiateToPLNiro() // Tested
      }
    }
    else if (isDeeplinkForPath(url, PERSONAL_LOAN)) {
      // PERSONAL LOANS FLOW
      if (isLoggedInAndRentPaid) {
        const routeNames = [
          POCv2LoansCommonRoute,
          { routeName: SCREENS.PERSONAL_LOAN_LANDING, routeParams: {} }
        ]
        resetActionForPayOnCredit(routeNames) // Tested
      } else {
        navigateToPersonalLoan() // Tested
      }
    }
    else if (isDeeplinkForPath(url, MY_TRANSACTIONS)) {
      // POC Transactions
      if (isLoggedInAndRentPaid) {
        const paramsTransaction = getMyTransactionParamsDeeplink(url)
        const { filter, orderId } = paramsTransaction
        const {
          index, new: isNewMTP
        } = transactionTabMapping?.[filter] || transactionTabMapping.rent_agreement
        const routeParams = { index, orderId }
        const myTransactionRoute = isNewMTP ? SCREENS.MTP_SCREEN : SCREENS.MY_TRANSACTIONS
 
        const routeNames = [
          {
            routeName: SCREENS.PAY_ON_CREDIT_V2,
            routeParams: { tabName: PocTabRoutes.profileTab }
          },
          { routeName: myTransactionRoute, routeParams }
        ]
        resetActionForPayOnCredit(routeNames) // Tested
      } else {
        resetActionForMyTransaction()
      }
      // LINE OF CREDIT
    } else if (isDeeplinkForPath(url, CREDIT_LINE)) {
      navigateToCreditLine(url)
      // MILESTONE
    } else if (isDeeplinkForPath(url, MILESTONE)) {
      resetActionForPayOnCredit()
    } else {
      isPOCUrl = false
    }
  } else {
    // OLD deeplinks flows from before
    // eslint-disable-next-line no-lonely-if
    if (isRewardsPageDeeplink(url)) {
      dispatch(handleRewardPage())
    } else if (url.includes(PAY_RENT)) {
      CredPayUtils.navigateToPayOnCredit()
    } else if (isDeeplinkForPath(url, RENT_NOW_PAY_LATER)) {
      const paramsPayLater = getPayLaterParamsDeeplink(url)
      navigateToRNPL(paramsPayLater)
    } else if (isDeeplinkForPath(url, CREDIT_LINE)) {
      navigateToCreditLine(url)
      // MILESTONE
    } else if (url.includes('home-loans')) {
      navigateToHomeLoans()
    } else if (url.includes('personal-loan')) {
      navigateToPersonalLoan()
    } else if (isNiroPlDeeplink(url)) {
      navgiateToPLNiro()
    } else if (url.includes('my-transactions')) {
      resetActionForMyTransaction()
    } else if (isDeeplinkForPath(url, MILESTONE)) {
      resetActionOnPayRentHome()
    } else {
      isPOCUrl = false
    }
  }
  return Promise.resolve({ isPOCUrl })
}
