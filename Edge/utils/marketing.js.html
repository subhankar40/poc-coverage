function sendAppRemarketingEvent(gaEvent, service, extraParams) {
  const { category, action, label } = gaEvent
  let pageType = null
  let itemId = null
  let price = null
  let firebaseEvent = null
  let fbEvent = null
 
  if (category === 'details_page' && action === 'open') {
    pageType = 'offerdetail'
    firebaseEvent = FIREBASE_EVENT_NAMES.VIEW_ITEM
    fbEvent = FB_EVENT_NAMES.VIEW_CONTENT
  } else if (category === 'conversion' && action === 'open_crf') {
    pageType = getCrfPageType(label.source)
    firebaseEvent = FIREBASE_EVENT_NAMES.ADD_TO_CART
    fbEvent = FB_EVENT_NAMES.ADD_TO_CART
  } else if (category === 'conversion' && action === 'submitted_crf') {
    pageType = getCrfPageType(label.source)
    firebaseEvent = FIREBASE_EVENT_NAMES.PURCHASE
    fbEvent = FB_EVENT_NAMES.PURCHASE
  } else if (category === 'app_remarketing' && action === 'serp_view') {
    const { source } = label
    firebaseEvent = FIREBASE_EVENT_NAMES.VIEW_ITEM_LIST
    if (source === 'filters') {
      pageType = 'filters'
      fbEvent = FB_EVENT_NAMES.SEARCH
    } else {
      pageType = 'searchresults'
      fbEvent = FB_EVENT_NAMES.VIEW_CONTENT
    }
  }
 
  if (!isEmpty(firebaseEvent) && !isEmpty(fbEvent)) {
    itemId = formatPropertyId(label.service_type, label.id)
    price = label.price // eslint-disable-line
    if (!isEmpty(pageType) && !isEmpty(itemId) && !isEmpty(price) && !isEmpty(service)) {
      const eventParams = { pageType, itemId, price }
      sendEventToFirebase(firebaseEvent, eventParams, service, extraParams)
      sendEventToFb(fbEvent, eventParams, extraParams)
    }
  }
}
