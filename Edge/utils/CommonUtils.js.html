export const getDiscountedValue = (
  couponMetaInfo = [], actualPrice = 0, discountMetaInfo, mlDiscountValue = 0
) => {
  const metaInfo = (couponMetaInfo?.[0]?.coupons?.[0]?.meta_data || discountMetaInfo || {})
  const discountedPriceFactor = mlDiscountValue || Number(metaInfo?.value || 0)
  const valueMetric = mlDiscountValue ? 'PERCENT' : (metaInfo?.valueMetric || metaInfo?.value_metric)
  switch (valueMetric) {
    case 'PERCENT':
      return Math.ceil(Number(actualPrice) - (Number(actualPrice) * discountedPriceFactor / 100))
    default:
      return actualPrice
  }
}

export const getFloorNumber = (floorNumber, { suffix = '' } = {}) => {
  const floorMap = {
    '-2': 'B2',
    '-1': 'B1',
    0: 'Ground'
  }
  if (floorNumber < 1) {
    return `${floorMap[floorNumber]}`
  }
  return `${floorNumber}${getSuffix(floorNumber)}${suffix}`
}
 
export const parseFloorNumbers = (floorNumbers = [], totalFloors) => {
  const prefix = floorNumbers.map(floor => getFloorNumber(floor)).join(', ')
  const suffix = !isEmpty(totalFloors)
    ? ` of ${totalFloors} Floors`
    : ` ${getValueStringByCount(
      SINGULAR_PLURAL_WORDS.FLOOR,
      floorNumbers.length
    )}`
  return prefix + suffix
}

export const encryptRequestedBody = (body = {}) => {
  /**
   * Will encrypt the body using AES algo with AES secret key
   * Will encrypt AES secret key with RSA public key using RSA algo
   */
  const requestedBody = {
    data: '',
    key: ''
  }
  const serializedData = JSON.stringify(body)
 
  // This is used for RSA
  const rsa = new JSEncrypt()
  rsa.setPublicKey(Config.RSA_PUBLIC_KEY)
  // This is used for AES
 
  const aesSecretKey = CryptoJS.enc.Utf8.parse(Config.AES_SECRET_KEY)
  const iv = CryptoJS.enc.Utf8.parse(Config?.AES_SECRET_KEY?.substring(0, 16))
 
  const encryptedBody = CryptoJS.AES.encrypt(serializedData,
    aesSecretKey,
    {
      iv,
      mode: CryptoJS.mode.CBC,
      padding: CryptoJS.pad.Pkcs7
    })
 
  const encryptedKey = rsa.encrypt(Config.AES_SECRET_KEY)
 
  requestedBody.data = (encryptedBody.toString())
  requestedBody.key = (encryptedKey.toString())
  return requestedBody
}
 
export function createRefHandler() {
  const refs = new Set([])
  class RefHandler {
    getRefs= () => [...refs.values()]
 
    setRefs = (ref) => {
      if (ref) {
        if (!refs.has(ref)) {
          refs.add(ref)
        }
      }
    }
 
    clearRefs = () => {
      refs.clear()
    }
  }
  return RefHandler
}
