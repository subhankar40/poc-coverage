const showNewTruecallerCard = (snapshot) => {
  try {
    const truecallerPayload = JSON.parse(asObject(snapshot.new_truecaller_card))
    return truecallerPayload
  } catch (e) {
    logError(e)
    return truecallerValues
  }
}
 
const makeFunctionCall = (func, snapshot, callback) => {
  func && typeof func === 'function'
    && callback && typeof callback === 'function' && func(snapshot, callback)
}
 
const cityDataVersion = (snapshot, callback) => {
  const cityDataVersion = snapshot.city_data_version?.value
  callback(cityDataVersion)
}
 
const getRCValuesFromNative = (callbacks) => {
  NewBridgeModule.fetchRemoteConfigData('Edge').then((values) => {
    const snapshot = JSON.parse(values || '{}')
    const {
      saveConfigValues,
      fetchCityDataVersion
    } = callbacks
 
    setRentRelevanceV2(asBoolean(snapshot?.rent_relevance_v2) || false)
    setSerpCTAExpType(asString(snapshot.new_serp_card_cta_seller) || DEFAULT)
    setKhojBuyFilterApiFlag(asBoolean(snapshot?.use_khoj_buy_filter_api) || false)
    setKhojPGFilterApiFlag(asBoolean(snapshot?.use_khoj_pg_filter_api) || false)
    setOptimizedGenericTextInputVisibility(
      asBoolean(snapshot?.show_optimized_generic_text_input) || false
    )
    setAppExitTime(asNumber(snapshot.app_exit_time))
