  getSavedSellerContact, isPropertyStateContacted
} from './helpers'
import isEmpty from './isEmpty'
 
export const hasSavedSellerNumber = ({ flatState, id, fltcntId = null }) => {
  const isPropertyContacted = isPropertyStateContacted(flatState, id, fltcntId)
  const sellerContact = getSavedSellerContact(flatState, id, fltcntId)
  return { hasSavedContact: isPropertyContacted && !isEmpty(sellerContact), sellerContact }
}
 
export const getPlacement = (params) => {
  let placement = 'Details_similar_properties'
  if (!isEmpty(params)) {
    if (params?.placement) placement = params?.placement
    else if (params?.analyticsData?.originalSource?.includes('deal')) {
      placement = params?.analyticsData?.originalSource
    }
  }
  return placement
}
 
export const getFlatState = ({
  id, flatState, isProjectListing = false, projectListingId = ''
}) => {
  const keyId = isProjectListing && !isEmpty(projectListingId) ? projectListingId : id
  return !isEmpty(flatState?.[keyId]) ? flatState[keyId]?.state : ''
}
 
export const getFromLabel = (mainSource = '', fromViewPhone = false, fromWhatsapp = false, fromHousingChat = false) => {
  let from = mainSource
  if (fromViewPhone) {
    from = `${from}_view_phone`
  } else if (fromWhatsapp) {
    from = `${from}_whatsapp`
  } else if (fromHousingChat) {
    from = `${from}_housing_chat`
  } else {
    from = `${from}_contact`
  }
  return from
}
 
export const getAnalyticsData = (route) => {
  const analyticsData = !isEmpty(route?.params?.analyticsData) ? route?.params.analyticsData : {}
  if (isEmpty(analyticsData.search_type)) {
    analyticsData.search_type = 'manual'
  }
  return analyticsData
}
 
export const getPropertyDetails = (params) => {
  const {
    type, id, detail, routeParams, isProjectListing, projectListingId
  } = params || {}
  if (isEmpty(type) || isEmpty(id)) return {}
  const key = getPropertyKey(type, id, isProjectListing, projectListingId)
  const property = detail[key] && detail[key].gallery_images ? detail[key] : {
    ...(routeParams?.item || {}), forceFetch: true, id, type
  }
  return property || {}
}

export const isMultipleDeveloper = (source = '', di = []) => {
  if (di?.length > 0 && (multipleDevelopersNameAdProd.includes(source))) {
    return di.map(item => item.name).join(', ')
  }
  if (di?.length > 0 && constants.FLTCNT_PROJECTS.includes(source)) return di?.[0]?.name
  return ''
}
