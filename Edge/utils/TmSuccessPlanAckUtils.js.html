/**
 * This replaceAll function is used to replace multiple word with multiple word like
 * const str = "name: <fname> <lname> "
 * const mapStr = {
 * <fname>: Manish,
 * <lname>: Sharma
 * }
 * const rst = replaceAll(str, mapStr)
 * rst // name: Manish Sharma
 */
export function replaceAll(str, mapObj) {
  const re = new RegExp(Object.keys(mapObj).join('|'), 'gi')
 
  return str.replace(re, matched => mapObj[matched.toLowerCase()])
}
 
export const getAvatarImage = (item) => {
  switch (item.priority) {
    case HP_PACKAGES_PRIORITIES.BASIC:
      return images.hpUserBasicImg
    case HP_PACKAGES_PRIORITIES.PREMIUM:
      return images.hpUserPremiumImg
    case HP_PACKAGES_PRIORITIES.ASSISTED:
      return images.hpUserAssistedImg
    case HP_PACKAGES_PRIORITIES.MONEYBACK:
      return images.hpUserAssistedImg
    default:
      return images.hpUserBasicImg
  }
}
