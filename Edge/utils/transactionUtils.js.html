import { TMstatusMap } from 'views/TenenatMonetisation/TMhelper'
import { getPendingRaTransDetail, getRATranStatus, isCancelledOrder } from 'views/RentAgreement/RAhelper'
import isEmpty from 'common/utils/isEmpty'
import colors from 'utils/colors'
import { STRINGS } from 'constants/HpNewHookConstant'
 
const TENANT_MONETIZATION = 'TENANT_MONETIZATION'
const RENT_AGREEMENT = 'RENT_AGREEMENT'
 
const titleText = (status, id, item, fromHpChat = false) => {
  const orderType = item?.order?.orderType
  switch (status) {
    case 'success': {
      if (orderType === 'TENANT_MONETIZATION') {
        return fromHpChat ? STRINGS.PAID_SUCCESSFUL : STRINGS.WOW_PAID_SUCCESSFUL
      }
      if (orderType === 'RENT_AGREEMENT') {
        return 'Signed agreement generated'
      }
      return id === 11 ? 'Refund processed' : 'Payment processed to beneficiary'
    }
    case 'pending':
      if (orderType === 'RENT_AGREEMENT') {
        const {
          amountStatus, userSignStatus, noOfPendingSign, landlordSignStatus, isTenant,
          orderDetailsStatus
        } = getPendingRaTransDetail(item)
        if (!amountStatus) {
          return 'Payment pending'
        }
        if (!orderDetailsStatus) {
          return 'Agreement details pending'
        }
        if (!userSignStatus) {
          return 'Your digital sign Pending'
        }
        if (noOfPendingSign === 1 && !landlordSignStatus && isTenant) {
          return 'Landlord digital sign pending'
        }
        return `${noOfPendingSign} other${noOfPendingSign > 1 ? "s'" : ''} digital sign pending`
      }
      return id === 11 ? 'Refund in process' : 'Payment in process'
    case 'failure':
      if (orderType === 'RENT_AGREEMENT' && isCancelledOrder(item)) {
        return 'Order cancelled'
      }
      return 'Payment failed'
    default:
      break
  }
  return ''
}
 
const transactionStatusMap = (state, id) => {
  switch (state) {
    case 1:
      return {
        statusId: 1,
        stateId: id,
        status: 'pending',
        color: colors.yellow7
      }
    case 2:
      return {
        statusId: 2,
        stateId: id,
        status: 'success',
        color: colors.green
      }
    case 3:
      return {
        statusId: 3,
        stateId: id,
        status: 'failure',
        color: colors.red4
      }
    default:
      return {
        statusId: 3,
        stateId: id,
        status: 'failure',
        color: colors.black2
      }
  }
}
 
const getStatus = (item) => {
  const { orderStatus, order } = item
  if (order.orderType === TENANT_MONETIZATION) {
    const { status } = order
    return ({ status: TMstatusMap[status] })
  }
  if (order.orderType === RENT_AGREEMENT) {
    return { status: getRATranStatus(item) }
  }
  if (!isEmpty(orderStatus)
    && orderStatus[orderStatus.length - 1]
    && orderStatus[orderStatus.length - 1].statusId) {
    const state = orderStatus[orderStatus.length - 1].statusId
    const id = orderStatus[orderStatus.length - 1].stateId
    return transactionStatusMap(state, id)
  }
  return {}
}
 
export const TransactionUtils = {
  getStatus,
  titleText
}
