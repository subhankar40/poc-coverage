export const getUTMParamsFromURL = (url) => {
  const campaignDetails = {}
  if (isNilOrEmpty(url)) {
    return campaignDetails
  }
  const [, queryString] = url.split('?')
  const queryParams = (queryString || '').split('&').reduce((result, item) => {
    if (item !== '' && !isNilOrEmpty(item)) {
      const nextResult = result || {}
      const [key, value] = item.split('=')
      nextResult[key] = value
      return nextResult
    }
    return result
  }, null)
  if (isNilOrEmpty(queryParams)) {
    return campaignDetails
  }
  /* eslint-disable camelcase */
  const {
    utm_source, utm_medium, utm_campaign, gclid, sourcemedium = null
  } = queryParams
  if (!isNilOrEmpty(utm_campaign)) {
    campaignDetails.utm_campaign = utm_campaign
  }
  if (!isNilOrEmpty(utm_source)) {
    campaignDetails.utm_source = utm_source
  }
  if (!isNilOrEmpty(utm_medium)) {
    campaignDetails.utm_medium = utm_medium
  }
  if (!isNilOrEmpty(gclid)) {
    campaignDetails.gclid = gclid
    campaignDetails.sourcemedium = sourcemedium || 'google / cpc'
  }
  return campaignDetails
}

export const getParamsFromDeeplink = (deeplink, isDeferred) => {
  const paramSplit = isDeferred ? '&' : '%26'
  const keySplit = isDeferred ? '=' : '%3D'
  const keyValue = deeplink ? deeplink.split(paramSplit) : []
  const params = {}
  keyValue.forEach((param) => {
    if (param) {
      const values = param.split(keySplit)
      if (values.length > 1) {
        const key = values[0]
        const value = values[1]
        switch (key) {
          case 'type': {
            params.type = value
            break
          }
          case 'poly': {
            params.poly = value
            break
          }
          case 'latitude': {
            params.latitude = value
            break
          }
          case 'longitude': {
            params.longitude = value
            break
          }
          case 'city_uuid': {
            params.city_uuid = value
            break
          }
          default: {
            break
          }
        }
      }
    }
  })
  return params
}
 
export const getPayRentParamsDeeplink = (url) => {
  const Url = require('url')
  const scheme = Url.parse(url, true)
  const { pathname, hostname } = scheme
  if (hostname === HOUSING_HOST && !isNilOrEmpty(pathname)
    && (pathname.startsWith(PAY_RENT_PATH) || pathname.startsWith(PAY_RENT_EDGE_PATH))) {
    const { phone, next } = scheme?.query
    if (!isNilOrEmpty(phone) && !isNilOrEmpty(next)) {
      const [, orderId] = next.split('?ref=')
      if (!isNilOrEmpty(orderId)) {
        if (orderId.includes('/')) {
          return { phone, orderId: orderId.substring(0, orderId.length - 1) }
        }
        return { phone, orderId }
      }
    } else {
      return {}
    }
  }
  return {}
}

export const getHpParams = (url = '') => {
  const REG_EXP = /-(\d+)$/
  const rst = {}
  const Url = require('url')
  const scheme = Url.parse(url, true)
  const { query = {}, pathname } = scheme
  const match = pathname.match(REG_EXP)
  if (match && match[1]) { rst.pkgId = match[1] }
  const coupon = query?.couponCode || query?.coupon || ''
  const premiumFilter = !!query?.premiumFilter || false
  const initBuyFlow = !!(query?.initBuyFlow && rst.pkgId) || false
  return ({
    ...rst, ...query, coupon, premiumFilter, initBuyFlow
  })
}
