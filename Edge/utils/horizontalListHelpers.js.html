export function getImageUrl(data) {

    && !isEmpty(data.gallery_images)
    && data.gallery_images instanceof Array

export const getTopProjects = (featuredDeveloper, excludeList) => {
  const topProjects = []
  if (!isEmpty(featuredDeveloper)) {
    featuredDeveloper.forEach((project) => {
      if (!isEmpty(project.top_projects)) {
        project.top_projects.forEach((topProject) => {
          const featuredProject = topProject
          featuredProject.gallery_images = [{
            absolute_url: topProject.thumb_image_url
          }]
          if (isEmpty(excludeList) || isEmpty(excludeList[featuredProject.id])) {
            topProjects.push(featuredProject)
          }
        })
      }
    })
  }
  return topProjects
}

export const changeFilterConfig = (
  filterConfig, data, isBrandFilter = false, isEstablishment = false
) => {
  if (isEmpty(data)) return filterConfig
  if (isBrandFilter) {
    filterConfig.brand_names = [data.brand_name]
    return filterConfig
  }
  if (isEstablishment) {
    filterConfig.base = [{
      type: 'EST',
      uuid: data.id,
      label: data.name
    }]
    return filterConfig
  }
  const { key } = data
  switch (key) {
    case BOYS: {
      const selectedIds = [0]
      filterConfig.gender_ids = selectedIds
      break
    }
    case GIRLS: {
      const selectedIds = [1]
      filterConfig.gender_ids = selectedIds
      break
    }
    case FOOD_AVAILABLE: {
      const selectedIds = [true]
      filterConfig.food_available = selectedIds
      break
    }
    case PRIVATE_ROOM: {
      const selectedIds = [1]
      filterConfig.room_type = selectedIds
      break
    }
    default:
      break
  }
  return filterConfig
}

export function getSubscriptionType({ item }) {
  const { isCommercial } = item
  const sellersInfo = !isEmpty(item) ? item.sellers_info : null
  // removed checkForSeller condition, if any issue occure please connect with Sahil Goyal
  if (!isEmpty(sellersInfo) && sellersInfo instanceof Array && !isEmpty(sellersInfo[0])) {
    const {
      listing_order_subscription_type: type,
      is_commercial_titanium: isCommercialTitanium,
      contact_person_id: contactPersonId
    } = sellersInfo[0]
    const isOwner = (contactPersonId === constants.OWNER_ID)
    if (isOwner && type === constants.PREMIUM_FC) {
      return { hasAdTag: true, type: constants.PREMIUM, badgeImage: images.primeBrokerBadge }
    }
    if (isOwner && type === constants.PREMIUM_PLUS_FC) {
      return { hasAdTag: true, type: constants.PREMIUM_PLUS, badgeImage: images.primeBrokerBadge }
    }
    if (
      isCommercial && (isCommercialTitanium || isPremiumListingType(type))) {
      return { hasAdTag: true, type: constants.PREMIUM, badgeImage: images.primeBrokerBadge }
    }
    if (!isOwner && !isCommercial && sellersInfo?.length < 2 && isPremiumListingType(type)) {
      return { hasAdTag: true, type: '', badgeImage: images.primeBrokerBadge }
    }
  }
  return { hasAdTag: false, type: '', badgeImage: null }
}
