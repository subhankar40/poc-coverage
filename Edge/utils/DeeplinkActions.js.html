export function handleHpBlockerDeeplinkUrl(url) {
  return () => {
    const Url = require('url')
    const urlData = Url.parse(url, true)
    const { query, pathname } = urlData
    const skipExperiment = checkForSkipExperiment(url)
    handleHousingPremium({
      isPgService: !!pathname?.includes?.('paying-guests'),
      fromPage: SOURCES.DEEP_LINKS,
      coupon: query?.couponCode,
      skipExperiment

}
 
export function handleHpPromiseDeeplinkUrl(url) {

  }
}

export function handleInsurancePage(url, moengageType = '', tabType = '', order = null) {
  return () => {
    const Url = require('url')
    const scheme = Url.parse(url, true)
    const { query, path } = scheme
    getOrderId(path)
    const type = isEmpty(moengageType) ? query?.type || '' : moengageType
    const skipExperiment = checkForSkipExperiment(url)
    const tab = isEmpty(tabType) ? query?.tab || '' : tabType
    const orderId = isEmpty(order) ? getOrderId(path) || '' : order
    resetActionOnEdgeHome(
      SCREENS.INSURANCE_PAGE,
      {
        type,
        skipExperiment,
        from: 'deeplink',
        tab,
        orderId
      }
    )
  }
}

function handlePayRentViaWhatsapp(params) {
  return (dispatch, getState) => {
    const {
      common: { user, onboarding: { showProfiling } }, selectedCity, rentPay: { isRentPaid }
    } = getState()
    const { isLoggedIn, data } = user
    const payload = { orderId: params.orderId }
    if (isEmpty(selectedCity.uuid)) {
      dispatch(setDefaultCity())
    }
    if (isLoggedIn) {
      if (params.phone === data.phone_number) {
        dispatch(fetchOrderDetailsByOrderId(payload))
      } else {
        resetToRouteOtp('loginOtpView', { onLoginComplete: bindActionCreators(onLoginViaPayRent(payload, fetchOrderDetailsByOrderId), dispatch), phoneNumber: params.phone }, showProfiling, isRentPaid)
      }
    } else {
      resetToRouteOtp('loginOtpView', { onLoginComplete: bindActionCreators(onLoginViaPayRent(payload, fetchOrderDetailsByOrderId), dispatch), phoneNumber: params.phone }, showProfiling, isRentPaid)
    }
  }
}
 
function handleMyTransactionPage(params) {
  const { phone, filter, orderId } = params
  return (dispatch, getState) => {
    const {
      rentPay: { isRentPaid }, common: { onboarding: { showProfiling } }, selectedCity
    } = getState()
    const reduxState = getState()
    const isLoggedIn = getIsLoggedIn(reduxState)
    const phoneNumber = getUserPhoneNos(reduxState)
    if (isEmpty(selectedCity?.uuid)) {
      dispatch(setDefaultCity())
    }
    if (isLoggedIn) {
      if (phone === phoneNumber) {
        navigateToMyTransactionPage(filter, orderId)()
      } else {
        resetToRouteOtp(
          SCREENS.LOGIN_OTP_VIEW,
          { onLoginComplete: navigateToMyTransactionPage(filter, orderId), phoneNumber: phone },
          showProfiling, isRentPaid
        )
      }
    } else {
      resetToRouteOtp(
        SCREENS.LOGIN_OTP_VIEW,
        { onLoginComplete: navigateToMyTransactionPage(filter, orderId), phoneNumber: phone },
        showProfiling,
        false
      )
    }
  }
}

function getUrlFromShortenedUrl(url) {
  const { PAY_RENT } = attributeNames
 
  return (dispatch, getState) => {
    const { common: { onboarding: { showProfiling } } } = getState()
    fetch(url).then((response) => {
      const expandedURL = !isEmpty(response) ? response.url : ''
      setUTMParamsFromURL(expandedURL)
      const paramsTransaction = getMyTransactionParamsDeeplink(expandedURL)
      if (!isEmpty(paramsTransaction?.filter) && !isEmpty(paramsTransaction?.phone)) {
        addTraceAttribute(PAY_RENT)
        dispatch(handleMyTransactionPage(paramsTransaction))
        return
      }
      const paramsPayRent = getPayRentParamsDeeplink(expandedURL)
      if (!isEmpty(paramsPayRent?.phone) && !isEmpty(paramsPayRent?.orderId)) {
        addTraceAttribute(PAY_RENT)
        dispatch(handlePayRentViaWhatsapp(paramsPayRent))
        return
      }
      if (isBNPLDeeplink(expandedURL)) {
        const paramsPayLater = getPayLaterParamsDeeplink(expandedURL)
        dispatch(handlePayLaterPage(paramsPayLater))
      }
      if (isHpSachetDeeplink(expandedURL)) {
        dispatch(handleHpSachetDeeplinkUrl(expandedURL))
        return
      }
      if (isHpBlockerDeeplink(expandedURL)) {
        dispatch(handleHpBlockerDeeplinkUrl(expandedURL))
        return
      }
      if (isHpPromiseDeeplink(expandedURL)) {
        dispatch(handleHpPromiseDeeplinkUrl(expandedURL))
        return
      }
      // Housing premium Landing page deeplink
      if (isHPLandingPage(expandedURL)) {
        dispatch(handleTenantLandingPage({ url }))
        return
      }
 
      dispatch(onOpenLinkingURL(isValidURL(expandedURL) ? expandedURL : url))
    }).catch(() => {
      navigateToLandingScreen(showProfiling)
    })
  }
}

// eslint-disable-next-line no-unused-vars
export function handleOpenURL(data, sendEvent = true) {
  const { url } = data
  const { PAY_RENT } = attributeNames
 
  // eslint-disable-next-line no-unused-vars
  return (dispatch, getState) => {
    const {
      selectedCity
    } = getState() || {}
    if (isDynamicLinkUrl(url)) return
    setUTMParamsFromURL(url)
    // sendEvent && SendFirstEvent.sendEvents()
    const paramsTransaction = getMyTransactionParamsDeeplink(url)
    if (!isEmpty(paramsTransaction?.filter) && !isEmpty(paramsTransaction?.phone)) {
      addTraceAttribute(PAY_RENT)
      dispatch(handleMyTransactionPage(paramsTransaction))
      return
    }
 
    const paramsPayRent = getPayRentParamsDeeplink(url)
    if (!isEmpty(paramsPayRent?.phone) && !isEmpty(paramsPayRent?.orderId)) {
      addTraceAttribute(PAY_RENT)
      dispatch(handlePayRentViaWhatsapp(paramsPayRent))
      return
    }
 
    if (isEmpty(selectedCity?.uuid)
      && (url.includes('my-rewards')
        || url.includes(PAY_RENT) || url.includes('my-transactions'))) {
      dispatch(setDefaultCity())
    }
    if (isInsuranceDeeplink(url)) {
      dispatch(handleInsurancePage(url))
    } else if (url.includes('premium-exclusive')) {
      dispatch(handlePremiumExclusivePage({ url }))
    } else if (url.includes('tenant-monetization')) {
      dispatch(handleTenantLandingPage({ url }))
    }
    else if (url.includes('rent-agreement')) {
      navigateToScreenWithEdgeLanding('rentAgreement')
    } else if (url.includes('home-interiors')) {
      navigateToScreenWithEdgeLanding(SCREENS.HI_LANDING_PAGE)
    } else if (url.includes('packers-and-movers')) {
      navigateToScreenWithEdgeLanding(SCREENS.PACKERS_MOVERS_LANDING)
    } else if (url.includes('hsng')) {
      dispatch(getUrlFromShortenedUrl(url))
    } else if (url.includes('solar-rooftop')) {
      navigateToScreenWithEdgeLanding(SCREENS.SOLAR_ROOFTOP_LANDING_SCREEN)
    } else if (url.includes('bill-payments')) {
      navigateToScreenWithEdgeLanding(SCREENS.SETU_LANDING_SCREEN)
    } else if (url.includes('fixed-deposit')) {
      navigateToScreenWithEdgeLanding(SCREENS.FD_LANDING_SCREEN)
    } else if (url.includes('edge-landing')) {
      navigateToEdgeLanding()
    } else if (url.includes('property-management')) {
      navigateToScreenWithEdgeLanding(SCREENS.PROPERTY_MANAGEMENT_LANDING_SCREEN)
    } else if (url.includes('report-fraud')) {
      navigateToReportFraud()
    } else if (isHpSachetDeeplink(url)) {
      dispatch(handleHpSachetDeeplinkUrl(url))
    } else if (isHpBlockerDeeplink(url)) {
      dispatch(handleHpBlockerDeeplinkUrl(url))
    } else if (isHpPromiseDeeplink(url)) {
      dispatch(handleHpPromiseDeeplinkUrl(url))
    } else if (isHPLandingPage(url)) {
      // Housing premium Landing page deeplink
      dispatch(handleTenantLandingPage({ url }))
    } else if (url.includes('shared-listing')) {
      dispatch(handleSharedPropetiesDeeplinkUrl(url))
    } else {
      dispatch(handlePOCv2Deeplinks({ url }))
        .then(({ isPOCUrl }) => {
          if (isPOCUrl) {
            // The URL was a POC url and there no need of further execution
            return
          }
          dispatch(onOpenLinkingURL(url))
        })
    }
  }
}
