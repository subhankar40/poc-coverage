export function getPriceForAnalytics(property) {
  const {
    inventory_configs, type, starting_price, room_offerings
  } = property
  let propertyPrice = null
  if (isPg(type)) {
    propertyPrice = parseFloat(starting_price)
  } else if (isFM(type) && !isEmpty(room_offerings?.[0]?.price_details?.rent)) {
    propertyPrice = room_offerings[0].price_details.rent
  } else if (
    !isEmpty(inventory_configs)
    && !isEmpty(inventory_configs[0])
    && !isEmpty(inventory_configs[0].price)
  ) {
    propertyPrice = inventory_configs[0].price
  }
  return propertyPrice
}
 
const getSellerDetailsForCD = (data) => {
  const { profile, is_owner: isOwner, profile_type: profileType } = data
  if (isEmpty(profile?.profile_details) || isEmpty(profileType)) return {}
  const { operating_cities: opCities, is_paid_profile: isPaidProfile } = profile.profile_details
  const city = opCities || []
  const status = isPaidProfile ? 'Paid' : 'Unpaid'
  let userType = profileType
  if (userType === Constants.USER && isOwner) {
    userType = Constants.OWNER
  }
  return { city, status, userType }
}
 
export function getConfig(user, brokerPackages) {
  const { isLoggedIn, data } = user
  const brokerCategory = brokerPackages?.brokerCategory
  if (isLoggedIn && data) {
    const { profile_uuid: loginId, login_auth_token: password, uuid } = data
    return {
      loginId, password, uuid, ...getSellerDetailsForCD(data), brokerCategory
    }
  }
  return {}
}

export function getFilterParams(filterParams) {
  if (isEmpty(filterParams)) {
    return null
  }
  const ids = Object.keys(filterParams)
  const stackId = parseInt(ids[ids.length - 1], 10) || 1
  return filterParams[stackId]
}
 
export function constructLabel(filterParams, campaignDetails,
  service, selectedCity, flag, label) {
  const SEARCH_FLAG = Analytics.EVENT_TYPE.FLAG_SEARCH_INFO
  const FILTER_KEY = Analytics.CONSTANTS.FILTERS
  const sendScreenName = getCurrentScreenName() === 'postLoginScreen' ? 'onboarding' : getCurrentScreenName()
  return {
    ...label,
    ...(!isEmpty(campaignDetails) ? campaignDetails : {}),
    screen_name: label.screen_name || sendScreenName,
    service,
    ...(!isEmpty(selectedCity) ? { selectedCity } : {}),
    ...(flag === SEARCH_FLAG
      ? { [FILTER_KEY]: formatFiltersForAnalytics(filterParams, service) }
      : {})
  }
}

export function getOwnerPropertyNotifData(category, label, fParams, selectedCity) {
  const { screen_name, service } = label
  if (service === 'rent') {
    let ownerFilterParams = {}
    if (category === 'New_Onboarding_SU' || category === 'New_Onboarding') { // onboarding dropoff
      if (!isEmpty(fParams) && !isEmpty(fParams.base)) {
        ownerFilterParams = { ...fParams }
      } else if (!isEmpty(selectedCity) && !isEmpty(selectedCity.uuid)) {
        ownerFilterParams = {
          ...fParams,
          city_uuid: selectedCity.uuid,
          base: [{
            type: 'POLY', uuid: selectedCity.uuid, label: selectedCity.name, addedByDefault: true
          }]
        }
      }
    } else if (screen_name === 'search' && !isEmpty(fParams)
      && !isEmpty(fParams.base)) { // SRP dropoff
      ownerFilterParams = { ...fParams }
    }
    if (!isEmpty(ownerFilterParams)) {
      ownerFilterParams.prop_by = [2]
      const ownerPropertiesUrl = getFilterUrl(getFilterParamsFromState(ownerFilterParams), 'rent')
      return { ownerFilterParams: JSON.stringify(ownerFilterParams), ownerPropertiesUrl }
    }
  }
  return {}
}
