export const getPopularCities = () => popularCities
 
export const updatePopularCities = (cityDetails) => {
  if (!isEmpty(cityDetails) && !isEmpty(cityDetails.uuid)) {
    const {
      name, city_id, uuid, city_uuid, url_name
    } = cityDetails
    const parsedDetails = {
      name,
      city_id: city_id || uuid,
      uuid,
      city_uuid,
      url_name
    }
    const filteredIndex = popularCities.findIndex(eachCity => eachCity.uuid === cityDetails.uuid)
    if (filteredIndex > 0) {
      popularCities.splice(filteredIndex, 1)
      popularCities.unshift(parsedDetails)
    } else if (filteredIndex === -1) {
      popularCities.unshift(parsedDetails)
    }
  }
}
 
export const updatePopularCitiesForCitySelect = (topCities, ipBasedCity) => {
  popularCities = topCities
  updatePopularCities(ipBasedCity)
  return popularCities
}
 
export const getAlphabeticOtherCitiesData = (otherCities) => {
  let alphabeticOtherCities = {}
  otherCities.forEach((city) => {
    const cityName = city?.name || 'City'
    const firstCityAlphabet = cityName.charAt(0)
    if (firstCityAlphabet in alphabeticOtherCities) {
      alphabeticOtherCities[firstCityAlphabet].push(city)
    } else {
      alphabeticOtherCities = {
        ...alphabeticOtherCities,
        [firstCityAlphabet]: [city]
      }
    }
  })
  return alphabeticOtherCities
}
 
export const getDefaultAlphabetFilter = (alphabeticOtherCities) => {
  if (!isEmpty(alphabeticOtherCities)) {
    return Object.keys(alphabeticOtherCities)[0]
  }
  return 'A'
}
 
export const getDefaultCity = () => {
  const popularCities = getPopularCities()
  const defaultCity = popularCities.find(eachCity => DEFAULT_CITY_MUMBAI_UUID === eachCity.uuid)
  return defaultCity
}
