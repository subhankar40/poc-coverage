export function getLocality(template) {
  const { uuid, type, id } = template
  if (type === null || (isEmpty(id) && isEmpty(uuid))) { return null }
  if (type === 'project') {
    const project = {
      type: 'project',
      id: (!isEmpty(id)) ? parseInt(id, 10) : parseInt(uuid, 10)
    }
    return project
  }
  const locality = {}
  locality.uuid = uuid
  if (type === 'locality-config' || type === 'locality') {
    locality.type = 'POLY'
    locality.label = !isEmpty(template) ? (template.polygon_name || template.display_name || 'Any') : 'Any'// null check
  } else if (type === 'establishment') {
    locality.type = 'EST'
    locality.label = !isEmpty(template) ? (template.display_name || 'Any') : 'Any'// null check
  } else {
    return null
  }
  return locality
}
 
export function getFilters(filters, amenities) {
  const extractedFilters = {}
  Object.keys(filters || {}).forEach((key) => {
    const value = filters[key]
    if (!isEmpty(value)) {
      switch (key) {
        case 'bhk_id': {
          extractedFilters.apartment = value.split(',').map((id) => {
            const newId = Number(id)
            return newId === 100 ? 5 : newId // 4BHK id in moondragon
          })
          break
        }
        case 'min_price': {
          extractedFilters.min_price = parseFloat(value)
          break
        }
        case 'max_price': {
          extractedFilters.max_price = parseFloat(value)
          break
        }
        case 'type': {
          extractedFilters.type = value
          break
        }
        // handle possession
        case 'min_poss': {
          extractedFilters.min_poss = parseInt(value, 10)
          break
        }
        case 'max_poss': {
          extractedFilters.max_poss = parseInt(value, 10)
          break
        }
        // Ready to move
        case 'max_age': {
          if (Number(value) === 0) {
            extractedFilters.min_poss = null
            extractedFilters.max_poss = 0
          }
          break
        }
        case 'contact_person_id': {
          extractedFilters.prop_by = value.split(',').map(Number)
          break
        }
        case 'property_type_id': {
          extractedFilters.property = value.split(',').map(Number)
          break
        }
        case 'number_of_toilets': {
          extractedFilters.bathrooms = parseInt(value, 10)
          break
        }
        case 'facing': {
          extractedFilters.facing = value.split(',')
          break
        }
        case 'lease_type_id': {
          extractedFilters.lease_type_ids = value.split(',').map(Number)
          break
        }
        case 'power_backup_type_id': {
          extractedFilters.power_backup_type_id = value.split(',').map(Number)
          break
        }
        case 'furnished_type_id': {
          extractedFilters.furnish_type_id = value.split(',').map(Number)
          break
        }
        default:
          break
      }
    }
  })
  // handle amenities
  if (!isEmpty(amenities)) {
    amenities.forEach((id) => {
      // eslint-disable-next-line camelcase
      const amenity_filter = amenitiesMap[id]
      if (!isEmpty(amenity_filter)) {
        extractedFilters[amenity_filter] = true
      }
    })
  }
  // handle budget filters
  const { budget } = filters
  if (!isEmpty(budget)) {
    Object.keys(budget).forEach((key) => {
      const value = budget[key]
      switch (key) {
        case 'min_price': {
          extractedFilters.min_price = parseFloat(value)
          break
        }
        case 'max_price': {
          extractedFilters.max_price = parseFloat(value)
          break
        }
        default:
          break
      }
    })
  }
 
  return extractedFilters
}
 
export function prepareFilters(service, filters, amenities) {
  const serviceDefaults = clone(getServiceDefaults(service))
  const defaultFilters = serviceDefaults.initialState
  const extractedFilters = getFilters(filters, amenities, service)
  const newFilters = merge(defaultFilters, extractedFilters)
  return newFilters
}
