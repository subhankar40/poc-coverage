import React, { useEffect } from 'react'
// import { NativeModules } from 'react-native'
import { NavigationContainer as NavContainer, useNavigationContainerRef } from '@react-navigation/native'
import { useFlipper, useReduxDevToolsExtension } from '@react-navigation/devtools'
// import { useDispatch, useSelector } from 'react-redux'
import { useDispatch, useSelector } from 'react-redux'
 
import { cleanUpState, sendScreenEvent, setStatusBarView } from 'navigators/Actions'
import {
  setNavigationRef, routeNameRef, getNavigationRef, removeNavigationRef
} from 'utils/AllRefs'
import { setNavigationRef as setNavigationCommonRef } from 'common/utils/AllRefs'
import { getRoutingInstrumentation } from 'utils/sentryUtils'
import NavigationQueue from 'utils/NavigationQueue'
// import SendFirstEvent from 'utils/SendFirstEvent'
import { trackEvent } from 'utils/trackEvents'
import EventQueue from 'common/utils/EventQueue'
import { SCREENS, SHOW_BOTTOM_TAB_SCREENS } from 'constants/screens'
import { TAB_BAR_HIDDEN_FROM, hideShowTab } from 'utils/hideShowTab'
import BundleInfoInstance from 'common/utils/BundleInfo'
import { handleOpenURL } from 'utils/DeeplinkActions'
import { MOENGAGE } from 'utils/nativeConstants'
import { POCSelectors } from 'modules/POC_Revamp_v2/selectors/pocSelector'
import { EA, EC } from 'constants/GAEvents'
import { onReceiveMoENotif } from './MoENotifHandler'
 
const NavigationContainer = (props) => {
  const {
    push_from: pushFrom, category, deeplinkUrl, isDeeplink, children, rootTag, edgeDripUrl,
    coupon, isPgService, pkgId, type, initialRouteName, from, fromPage, initBuyFlow,
    utm_campaign: utmCampaign, utm_source: utmSource, utm_medium: utmMedium
  } = props || {}
  const initialNavRef = useNavigationContainerRef()
  const isPocRevampV2 = useSelector(POCSelectors.getIsPOCRevampV2)
  const dispatch = useDispatch()
 
  useEffect(() => () => {
    hideShowTab(false, TAB_BAR_HIDDEN_FROM.SCREEN)
    removeNavigationRef()
  }, [])
 
  const setNavRef = (ref) => {
    if (ref) {
      setNavigationRef(ref, rootTag)
      setNavigationCommonRef(ref)
      EventQueue.events.forEach(({ category, action, label }) => {
        trackEvent(category, action, label)
      })
      EventQueue.clear()
    }
  }
 
  const onScreenChange = (screen) => {
    // send value true to hide bottom nav tab
    if (screen === SCREENS.PAY_ON_CREDIT) {
      hideShowTab(isPocRevampV2)
    } else {
      hideShowTab(!SHOW_BOTTOM_TAB_SCREENS[screen], TAB_BAR_HIDDEN_FROM.SCREEN)
    }
    const sendScreen = screen === 'postLoginScreen' ? 'onboarding' : screen
    sendScreenEvent(sendScreen)
  }
 
  const clearNavQueue = () => {
    if (
      getNavigationRef()?.current?.dispatch
      && NavigationQueue?.actions?.length > 0
    ) {
      const dispatch = getNavigationRef()?.current?.dispatch
      NavigationQueue?.actions?.map(action => dispatch(action))
      NavigationQueue?.clear()
    }
  }
 
  useFlipper(initialNavRef)
  useReduxDevToolsExtension(initialNavRef)
 
  const handleDeeplinksAndMoe = () => {
    BundleInfoInstance.setFromDeeplink(pushFrom === MOENGAGE || !!isDeeplink)
    if (pushFrom === MOENGAGE) {
      dispatch(onReceiveMoENotif({
        category: category || '',
        edgeDripUrl: edgeDripUrl || '',
        coupon: coupon || '',
        isPgService: isPgService || false,
        pkgId: pkgId || '',
        type: type || '',
        initBuyFlow: !!initBuyFlow,
        utmCampaign,
        utmMedium,
        utmSource
      }))
    }
    if (isDeeplink) {
      dispatch(handleOpenURL({
        url: deeplinkUrl
      }))
    }
  }
 
  const onReady = () => {
    trackEvent(EC.EDGE_BUNDLE, EA.LOADED, {
      initialRouteName,
      from: from || fromPage || ''
    })
    setNavRef(initialNavRef)
    getRoutingInstrumentation().registerNavigationContainer(getNavigationRef())
    clearNavQueue()
    routeNameRef.current = getNavigationRef().current?.getCurrentRoute()?.name === 'postLoginScreen' ? 'onboarding' : getNavigationRef().current?.getCurrentRoute()?.name
    handleDeeplinksAndMoe()
  }
 
  const onStateChange = () => {
    const previousRouteName = routeNameRef.current
    const currentRouteName = getNavigationRef().current?.getCurrentRoute()?.name
    if (previousRouteName !== currentRouteName && typeof currentRouteName === 'string') {
      // NativeModules.UtilNativeModule.crashlyticsLog(`Changed route ${currentRouteName}`)
      onScreenChange(currentRouteName)
      dispatch(cleanUpState(previousRouteName, currentRouteName))
    }
    setStatusBarView(previousRouteName, currentRouteName)
    routeNameRef.current = currentRouteName
  }
 
  return (
    <NavContainer
      ref={initialNavRef}
      onReady={onReady}
      onStateChange={onStateChange}
    >
      {children}
    </NavContainer>
  )
}
 
export default NavigationContainer
