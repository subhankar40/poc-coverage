import {
  catchError,
  debounceTime, distinctUntilChanged, mergeMap, of, retry, switchMap
} from 'rxjs'
import { Platform } from 'react-native'
 
import isEmpty from 'common/utils/isEmpty'
import { ofType } from 'redux-observable'
import { trackEvent } from 'utils/trackEvents'
import { isPg, isRent } from 'utils/TypeChecker'
import { filterSuggestions, getSelectedLocality } from 'utils/helpers'
import { EA, EC } from 'constants/GAEvents'
import { showNewToast } from 'actions/toastActions'
import getApi from '../config/index'
 
function getExcludeEntities(base, lastItemRemoved = false) {
  const excludeEntities = []
  if (!lastItemRemoved && base && base.length) {
    base.forEach((b) => {
      if (b.type === 'POLY') {
        excludeEntities.push({
          uuid: b.uuid,
          super_type: 'polygon',
          class: 'regions'
        })
      } else if (b.type === 'DEV') {
        excludeEntities.push({
          uuid: b.uuid,
          super_type: null,
          class: 'builders'
        })
      }
    })
  }
  return excludeEntities
}
 
export default function localitySelect(action$, store, { ajax }) {
  return action$.pipe(ofType('LOCALITY_AUTOCOMPLETE', 'CITY_AUTOCOMPLETE'),
    debounceTime(100),
    distinctUntilChanged(),
    switchMap(({
      payload: {
        text, cursor, base, service: newService, lastItemRemoved, fromCitySelectScreen = false
      }
    }) => {
      const {
        service: appLevelService,
        selectedCity,
        requirmentGathering,
        brokerOwnerSubscription
      } = store.value
      /**
       * newSelectedCity:- this is used in requirment gathering flow
       * isGatheringActive:- if user in Requirment gathering flow then it will be true
       */
      const cityObject = getSelectedLocality(requirmentGathering, brokerOwnerSubscription)
      const newSelectedService = cityObject?.service
      const newCity = cityObject?.selectedCity
      const service = !isEmpty(newSelectedService) ? newSelectedService : appLevelService
      const { uuid } = !isEmpty(newCity) ? newCity : selectedCity
      const { name } = service || {}
      const isFlatmates = !isEmpty(newService?.isFlatmates)
        ? newService.isFlatmates
        : service?.isFlatmates || false
      const apiList = getApi(name)
      const excludeEntities = JSON.stringify(getExcludeEntities(base, lastItemRemoved))
      let api = `${apiList.searchSuggest}/?source=${Platform.OS}&cursor=${cursor}&exclude_entities=${excludeEntities}&string=${text}`
      if (fromCitySelectScreen) api = `${api}&type=regions,project`
      else api = `${api}&polygon_uuid=${uuid}`
      if (isRent(name) && !isFlatmates) {
        api = `${api}&show_projects=true`
      }
      if (isFlatmates) {
        api = `${api}&service_type=flatmate`
      } else if (name !== 'pg') {
        api = `${api}&service_type=${name}`
      } else {
        api = `${api}&service_type=rent`
      }
      if (service?.isCommercial) {
        api = `${api}&region_super_type=polygon&type=regions`
      }
      const type = fromCitySelectScreen ? 'CITY_SUGGEST' : 'LOCALITY_SUGGEST'
      return ajax.getJSON(api).pipe(
        retry(3),
        mergeMap(({ response }) => {
          if (isEmpty(response.data)) {
            trackEvent(EC.SEARCH, EA.NO_RESULTS_FOUND, { searchString: text })
          }
          const data = fromCitySelectScreen || isPg(name) ? filterSuggestions(response, name)
            : response
          return of({
            type,
            payload: { data }
          })
        }),
        catchError(error => of({
          type,
          payload: {
            error,
            extra: {
              service, text, cursor, uuid
            }
          },
          error: true
        }))
      )
    }),
    catchError(() => ([
      showNewToast('Something went wrong, Please try again!')
    ])))
}
