const placeholdersFormatoffer = (placeholders) => {
  if (!isEmpty(placeholders)) {
    let offerText = ''
    let offerBigText = ''
    let howItWorks = []
    let aestheticImageUrl = ''
    let terms = []
    placeholders.forEach((element) => {
      if (element.key && element.key === 'text') offerText = element.value
      else if (element.key && element.key === 'bigText') offerBigText = element.value
      else if (element.key && element.key === 'howItWorks') howItWorks = element.value
      else if (element.key && element.key === 'terms') terms = element.value
      else if (element.key && element.key === 'aestheticImageUrl') aestheticImageUrl = element.value
    })
    return {
      offerText,
      offerBigText,
      howItWorks,
      aestheticImageUrl,
      terms
    }
  }
  return {}
}
const formatResp = jsonList => jsonList.map((coupon) => {
  const { id, couponDetails: { couponStatus, couponCode, offerDetails } } = coupon
  const {
    brandDetails, validityTo, placeholders, offerType
  } = offerDetails
  const brand = {}
  let offer = {}
  offer = placeholdersFormatoffer(placeholders)
  if (!isEmpty(offerType)) { offer.offerType = offerType }
 
  brand.name = brandDetails?.name
  if (!isEmpty(brandDetails.placeholders)) {
    brandDetails.placeholders.forEach((element) => {
      if (element.key && element.key === 'logoUrl') brand.url = element.value
    })
  }
  return {
    id,
    couponStatus,
    offer,
    brand,
    code: couponCode,
    expiryDate: getFromattedDate(validityTo, true)
  }
})
 
const formatRespforstaticRewards = (data) => {
  const BrandMap = new Map()
  data.brands.forEach((coupon) => {
    const { name, placeholders, id } = coupon || {}
    let url = ''
    if (!isEmpty(placeholders)) {
      placeholders.forEach((element) => {
        if (element.key && element.key === 'logoUrl') url = element.value
      })
    }
    BrandMap.set(id, { name, url })
  })
 
  const couponList = data.offers.map((coupon) => {
    // eslint-disable-next-line camelcase
    const { offerType, placeholders, brand_id } = coupon
    let offer = {}
    offer = placeholdersFormatoffer(placeholders)
    if (!isEmpty(offerType)) { offer.offerType = offerType }
    const brand = BrandMap.get(brand_id)
    return {
      offer,
      brand
    }
  })
  return couponList
}

const getTodaysDate = () => {
  let today = new Date()
  const dd = String(today.getDate()).padStart(2, '0')
  const mm = String(today.getMonth() + 1).padStart(2, '0')
  const yyyy = today.getFullYear()
  today = `${yyyy}-${mm}-${dd}`
  return today
}
 
function fetchRewards(action, store, { ajax }) {
  return action.pipe(ofType('FETCH_REWARDS'), mergeMap(({ payload }) => {
    const {
      type, start, onFetched
    } = payload
    // TODO: Added to get auth token while development
    const actions = []
    actions.push({ type: 'FETCHING_REWARDS' })
    const api = `${SERVICE.rentPay}/gambit/v1/coupon-assign/history?serviceType=HOUSING_RENT&entityType=ORDER&couponStatus=${type}&start=${start}&rows=10`
    return merge(
      from(actions),
      ajax.get(api, headersWithAuthToken(store)).pipe(
        mergeMap(({ response }) => {
          const { totalCount, data } = response
          if (onFetched && typeof onFetched === 'function') {
            onFetched(totalCount)
          }
          if (data != null) {
            const couponList = formatResp(data.couponAssignmentDetails)
            return [
              {
                type: 'SAVE_REWARDS',
                payload: {
                  type,
                  isSellerRewardsArrayEmpty: totalCount === 0,
                  count: totalCount,
                  data: couponList,
                  resetList: type === 'ACTIVE' && start === 0
                }
              }
            ]
          }
          return [{
            type: 'FETCHING_REWARDS_FAILED'
          }]
        }),
        catchError(err => ([{
          type: 'SHOW_NEW_TOAST',
          payload: {
            message:
              err && err.xhr && err.xhr.response
                ? err.xhr.response.message
                : 'Something went wrong. Please try again'
          }
        }, {
          type: 'FETCHING_REWARDS_FAILED'
        }]))
      )
    )
  }))
}
 
function fetchPostTransactionRewards(action, store, { ajax }) {
  return action.pipe(ofType('FETCH_POST_TRANSACTION_REWARDS'), mergeMap(({ payload }) => {
    const { entityId, delayTime, retryCount } = payload
    const api = `${SERVICE.rentPay}/apollo/v1/data/order/reward?orderId=${entityId}`
    const { user = {} } = store.value?.common
    const authToken = path(['data', 'login_auth_token'], user)
    let time = 1000
    if (delayTime) {
      time = delayTime
    }
    let count = 0
    if (retryCount) {
      count = retryCount
    }
    return ajax({
      url: api,
      method: 'GET',
      headers: {
        'housing-auth-token': `${authToken}`
      }
    }).pipe(
      delay(time),
      mergeMap(({ response }) => {
        const { data, responseCode } = response
        if (data != null) {
          const couponList = formatResp(data.couponAssignmentDetails)
          return [
            {
              type: 'UPDATE_POST_TRANSACTION_REWARDS',
              payload: {
                data: couponList
              }
            }
          ]
        }
        if (responseCode === 489 && count < 3) {
          return [
            {
              type: 'FETCH_POST_TRANSACTION_REWARDS',
              payload: {
                entityId,
                retryCount: count + 1,
                delayTime: 500
              }
            }
          ]
        }
        return []
      }),
      catchError(err => ([{
        type: 'SHOW_NEW_TOAST',
        payload: {
          message:
            err && err.error && err.error.msg
              ? err.error.msg
              : 'Something went wrong. Please try again'
        }
      }]))
    )
  }))
}

const fetchRewardsVouchers = ($action, store, { ajax }) => (
  $action.pipe(ofType(RentPayRewardsActionTypes.FETCH_REWARDS_VOUCHERS),
    switchMap((action) => {
      const { start, callback, rows = 10 } = action.payload || {}
      if (isEmpty(start)) {
        return []
      }
 
      const state = store.value
      const isPocV2 = POCSelectors.getIsPOCRevampV2(state)
      const body = isPocV2 ? getRewardsBodyV2(state) : getRewardsBody(state)
 
      const { rewards: { pageNo, totalCount, availableCoupons } } = store.value
      if (totalCount <= availableCoupons.length) {
        return []
      }
      const api = apis.getPocSuccessRewards(start, rows)
 
      const requestOptions = getRequestOptions(
        api, requestTypes.POST, headersWithAuthToken(store), body
      )
 
      return ajax(requestOptions).pipe(
        switchMap(({ response }) => {
          if (!response || isEmpty(response?.data)) {
            return []
          }
          const formattedData = CredPayUtils.parseRewardsData(response?.data)
          if (callback) {
            callback(formattedData)
          }
          const payload = {
            data: formattedData,
            pageNo: pageNo + 1,
            totalCount: response?.totalCount
          }
 
          return [RentPayRewardActions.fetchRewardsVouchersSuccess(payload)]
        }),
        catchError(err => [RentPayRewardActions.fetchRewardsVouchersFailure(err)])
      )
    }))
)

const getRandomElementArr = (inputArray, numOfElement = 1) => {
  let baseArray = [...inputArray]
  const filtered = []
  for (let i = 0; i < numOfElement; i += 1) {
    const index = Math.round(Math.random() * (baseArray.length - 1))
    filtered.push(baseArray[index])
    baseArray = baseArray.filter((_, i) => i !== index)
  }
  return filtered
}
