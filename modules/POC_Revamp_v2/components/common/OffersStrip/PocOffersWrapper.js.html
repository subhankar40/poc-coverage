import React, {
  memo, useCallback, useEffect, useRef
} from 'react'
import { useDispatch, useSelector } from 'react-redux'
import { Keyboard, StyleSheet, View } from 'react-native'
import {
  bool, func, object, string, number
} from 'prop-types'
 
import { EMPTY_STATE } from 'common/constants/AppConstants'
import { showModal } from 'components/ModalBox'
import { trackEvent } from 'common/utils/trackEvents'
import { WINDOW_HEIGHT, WINDOW_WIDTH } from 'common/utils/deviceDimensions'
import { EA, EC } from 'constants/GAEvents'
import { selfClearTimeout } from 'common/utils/CommonUtils'
import { OfferSelector } from 'modules/POC_Revamp_v2/selectors/offers'
import PocOffers from 'modules/POC_Revamp_v2/components/common/OffersStrip/PocOffers'
import OfferStripCollapsedExpandedCard from 'views/HousingEdge/PayRentRevamp/components/OffersStrip/OfferStripCollapsedExpandedCard'
import colors from 'utils/colors'
import { CredPayUtils } from 'utils/credPayUtils'
import { handleOpenURL } from 'utils/DeeplinkActions'
import { POCSelectors } from 'modules/POC_Revamp_v2/selectors/pocSelector'
import isEmpty from 'utils/isEmpty'
import resourceID from 'common/utils/resourceID'
import nuOffersTestIds from './nuOffersTestIds'
 
const styles = StyleSheet.create({
  individualOfferContainer: {
    backgroundColor: colors.black37,
    paddingHorizontal: 24,
    paddingVertical: 32
  }
})
 
const PocOffersWrapper = ({
  fromPage,
  showItemSeparator, onPaymentCategoryPress, offerStyle,
  dotsContainerStyles, customRowStyles, tileWidth, tileHeight, onCurrentIndexChange,
  scrollAnimationDuration, allowAutoScroll
}) => {
  const offerListRef = useRef(null)
  const pocOffersV2 = useSelector(OfferSelector.getPocOffers)
  const dispatch = useDispatch()
  const credPayOptions = useSelector(POCSelectors.getCredPayOptions)
 
  const isOfferFetching = useSelector(OfferSelector.getIsOfferFetching)
 
  const renderIndividualOffer = item => (
    <View style={styles.individualOfferContainer}>
      <OfferStripCollapsedExpandedCard item={item} isExpanded />
    </View>
  )
 
  const onModalClose = () => {
    offerListRef?.current?.startAutoPlay?.()
  }
 
  const onPressOffer = useCallback((item) => {
    const { actionType, actionValue } = item || {}
    const actionValueInteger = parseInt(actionValue, 10) || 0
    trackEvent(EC.POC_LANDING_PAGE, EA.OFFER_CLICKED,
      { offer_name: item?.offerPlaceHolders?.offerText || item?.brandName })
    let filteredItem = {}
    switch (actionType) {
      case CredPayUtils.offerActionTypes.TNC:
        Keyboard.dismiss()
        showModal({
          renderedComponent: () => renderIndividualOffer(item),
          onModalClose
        })
        break
      case CredPayUtils.offerActionTypes.PaymentOption:
        Keyboard.dismiss()
        filteredItem = credPayOptions?.filter(i => i?.id === actionValueInteger)
        trackEvent(EC.POC_LANDING_PAGE, EA.CTA_CLICKED, { categoy_name: filteredItem?.[0]?.type })
        onPaymentCategoryPress(filteredItem?.[0], true)
        selfClearTimeout(() => { offerListRef?.current?.startAutoPlay?.() }, 0)
        break
      case CredPayUtils.offerActionTypes.Redirection:
        Keyboard.dismiss()
        dispatch(handleOpenURL({ url: actionValue }))
        break
      default:
        Keyboard.dismiss()
        showModal({
          renderedComponent: () => renderIndividualOffer(item),
          onModalClose
        })
        break
    }
  }, [credPayOptions])
 
  useEffect(() => {
    if (!isEmpty(pocOffersV2)) {
      trackEvent(EC.POC_LANDING_PAGE, EA.OFFER_SEEN)
    }
  }, [pocOffersV2])
 
  if (isOfferFetching) {
    return null
  }
 
  return (
    <View
      {...{ [resourceID]: nuOffersTestIds.NU_OFFERS_PAYMENT }}
    >
      <PocOffers
        ref={offerListRef}
        offers={pocOffersV2}
        offerStyle={offerStyle}
        onPressOffer={onPressOffer}
        fromPage={fromPage}
        showItemSeparator={showItemSeparator}
        dotsContainerStyles={dotsContainerStyles}
        customRowStyles={customRowStyles}
        tileWidth={tileWidth}
        tileHeight={tileHeight}
        onCurrentIndexChange={onCurrentIndexChange}
        scollAnimationDuration={scrollAnimationDuration}
        allowAutoScroll={allowAutoScroll}
      />
    </View>
  )
}
 
PocOffersWrapper.propTypes = {
  fromPage: string.isRequired,
  showItemSeparator: bool,
  onPaymentCategoryPress: func,
  offerStyle: object,
  dotsContainerStyles: object,
  customRowStyles: object,
  tileWidth: number,
  tileHeight: number,
  onCurrentIndexChange: func,
  scrollAnimationDuration: number,
  allowAutoScroll: bool
}
 
PocOffersWrapper.defaultProps = {
  showItemSeparator: false,
  onPaymentCategoryPress: EMPTY_STATE.METHOD,
  offerStyle: EMPTY_STATE.OBJECT,
  dotsContainerStyles: EMPTY_STATE.OBJECT,
  customRowStyles: EMPTY_STATE.OBJECT,
  tileWidth: WINDOW_WIDTH,
  tileHeight: WINDOW_HEIGHT,
  onCurrentIndexChange: EMPTY_STATE.METHOD,
  scrollAnimationDuration: 2000,
  allowAutoScroll: false
}
 
export default memo(PocOffersWrapper)
