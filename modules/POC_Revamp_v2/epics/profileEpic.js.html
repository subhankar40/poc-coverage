import { combineEpics, ofType } from 'redux-observable'
import { catchError, mergeMap } from 'rxjs'
import { showNewToast } from 'actions/toastActions'
import { TOAST_MESSAGES } from 'constants/StringConstants'
import { EdgeApis } from 'config/edgeAPIsv2'
import isEmpty from 'utils/isEmpty'
import {
  defaultHeaders, getRequestOptions, headersWithAuthToken, requestTypes
} from 'utils/RequestHelpers'
import { PocProfileActionTypes, PocProfileActions } from 'modules/POC_Revamp_v2/actions/profileActions'
import { PocProfileSelectors } from 'modules/POC_Revamp_v2/selectors/profileSelector'
import { UserSelectors } from 'selectors/user'
import { GlobalPersistActions } from 'actions/globalPersists'
import { PocUtils } from '../utils/pocUtils'
 
const parseAllAvatarImages = data => JSON.parse(data?.[0]?.value) || []
 
const getUploadImageBody = (imageData, profileUuid, formData) => {
  const {
    APP_NAME, RESOURCE_NAME, ATTR_NAME, ML, ATTR_ID, UPLOAD_SOURCE, REUPLOAD,
    STATUS, TAG
  } = PocUtils.pocImageConstants || {}
  formData.append('appName', APP_NAME)
  formData.append('resourceName', RESOURCE_NAME)
  formData.append('attrName', ATTR_NAME)
  formData.append('ml', ML)
  formData.append('uploadSource', UPLOAD_SOURCE)
  formData.append('reUpload', REUPLOAD)
 
  const fileMap = {
    attr_id: ATTR_ID,
    status: STATUS,
    tag: TAG
  }
  const { md5, uri: imageUri, type: fileType } = imageData || {}
  // This is done in order to add Support for PNG image As currently API is not supporting PNG Image
  fileMap.extension = 'jpg'
  fileMap.name = md5
  fileMap.obj_id = profileUuid
  formData.append('md5Hash', md5)
  formData.append('image', {
    uri: imageUri,
    type: `image/${fileType}`,
    name: `profile.${fileType}`
  })
  formData.append('files', JSON.stringify([fileMap]))
}
 
const fetchAllAvatars = ($action, store, { ajax }) => (
  $action.pipe(ofType(PocProfileActionTypes.FETCH_ALL_AVATARS_REQ),
    mergeMap(() => {
      const api = EdgeApis.fetchPocProfileAllAvatars()
 
      const requestOptions = getRequestOptions(api, requestTypes.GET,
        defaultHeaders)
 
      return ajax(requestOptions)
        .pipe(mergeMap(({ response }) => {
          if (isEmpty(response) || isEmpty(response?.data)) {
            return []
          }
          const allAvatarImages = parseAllAvatarImages(response?.data)
          return [PocProfileActions.fetchAllAvatarsSuccess(allAvatarImages)]
        }),
        catchError(err => [
          showNewToast(TOAST_MESSAGES.SOMETHING_WENT_WRONG),
          PocProfileActions.fetchAllAvatarsFailure(err)]))
    }))
)
 
const updateProfile = ($action, store, { ajax }) => (
  $action.pipe(ofType(PocProfileActionTypes.UPDATE_PROFILE_REQ),
    mergeMap((action) => {
      const reduxState = store.value
      const newPicture = PocProfileSelectors.getNewAvatarImage(reduxState)
      const currentPicture = PocProfileSelectors.getSelectedAvatarImage(reduxState)
      const currentName = PocProfileSelectors.getTenantName(reduxState)
      const profileUuid = UserSelectors.getProfileUuid(reduxState)
      const loginAuthToken = UserSelectors.getLoginAuthToken(reduxState)
      const imageData = PocProfileSelectors.getImageData(reduxState)
      const { newName, cb } = action?.payload || {}
      const api = EdgeApis.updatePocProfile()
      const formData = new FormData()
 
      const headers = {
        'Content-Type': 'multipart/form-data',
        Cookie: `login_auth_token=${loginAuthToken}`
      }
 
      let updateName = currentName
      if (!isEmpty(newName) && newName !== currentName) {
        updateName = newName
      }
      let avatarImage = currentPicture
      if (!isEmpty(newPicture) && newPicture !== currentPicture) {
        avatarImage = newPicture
      }
      if (!isEmpty(imageData)) {
        getUploadImageBody(imageData, profileUuid, formData)
      } else {
        formData.append('avatarImage', avatarImage)
      }
      formData.append('tenantName', updateName)
 
      const requestOptions = getRequestOptions(api, requestTypes.PUT,
        headers, formData)
 
      return ajax(requestOptions)
        .pipe(mergeMap(({ response }) => {
          if (isEmpty(response) || isEmpty(response?.data)) {
            cb?.(null, null)
            return [PocProfileActions.updateProfileFailure()]
          }
          cb?.(null, response?.data)
          /* If user upload an image (already uploaded in past), set the current image */
          if (isEmpty(response?.data?.imageUrl)) {
            return [PocProfileActions.updateProfileSuccess({
              ...response?.data,
              imageUrl: avatarImage
            })]
          }
          return [PocProfileActions.updateProfileSuccess(response?.data)]
        }),
        catchError((err) => {
          cb?.(err)
          return [
            showNewToast(TOAST_MESSAGES.SOMETHING_WENT_WRONG),
            PocProfileActions.updateProfileFailure(err)]
        }))
    }))
)
 
export const fetchClickAndSeenEvents = ($action, store, { ajax }) => (
  $action.pipe(ofType(PocProfileActionTypes.FETCH_CLICK_SEEN_EVENTS_REQ),
    mergeMap(() => {
      const url = EdgeApis.userLevelClickAndSeenEvents()
      const requestOptions = getRequestOptions(url, requestTypes.GET, headersWithAuthToken(store))
      return ajax(requestOptions).pipe(
        mergeMap(({ response }) => {
          if (!isEmpty(response?.data)) {
            const { userLevelEvents } = PocUtils
            const actions = []
            actions.push(PocProfileActions.fetchClickAndSeenEventsResolved(response?.data))
            response?.data?.forEach((e) => {
              // Add your cases here -
              if (e.eventName === userLevelEvents.educationFeesCount) {
                actions.push(GlobalPersistActions.resetEducationFeeCount(e.clickCount))
              }
            })
            return actions
          }
          return []
        }),
        catchError(() => [])
      )
    }))
)
 
export const updateClickAndSeenEvents = ($action, store, { ajax }) => (
  $action.pipe(ofType(PocProfileActionTypes.UPDATE_CLICK_SEEN_EVENTS_REQ),
    mergeMap(({ payload }) => {
      const url = EdgeApis.userLevelClickAndSeenEvents()
      const requestOptions = getRequestOptions(url, requestTypes.POST,
        headersWithAuthToken(store), payload)
      return ajax(requestOptions).pipe(
        mergeMap(({ response }) => {
          if (!isEmpty(response?.data)) {
            return [PocProfileActions.fetchClickAndSeenEventsResolved(response?.data)]
          }
          return []
        }),
        catchError(() => [])
      )
    }))
)
