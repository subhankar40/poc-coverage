export const isPastMMYY = (date) => {
  const today = new Date()
  const todayYY = Number(today.getFullYear().toString().substring(2))
  const todayMM = Number(today.getMonth()) + 1
  const [MM, YY] = date?.split('/')
  return todayYY > Number(YY) || (todayYY === Number(YY) && Number(MM) < todayMM)
}
