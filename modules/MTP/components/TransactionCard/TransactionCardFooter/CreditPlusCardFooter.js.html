import { View, StyleSheet } from 'react-native'
import React, { memo, useMemo } from 'react'
 
import ButtonWrapper from 'common/components/ButtonWrapperV2'
import { MTPStrings } from 'constants/mtpStrings'
import colors from 'utils/colors'
import { func, number, object } from 'prop-types'
import { buttonsKey, getSubscriptionValidityObject, mtpButtons } from 'modules/MTP/utils/helper'
import { moderateScale } from 'common/utils/scalingFunctions'
import { TYPES, creditPlusStrings } from 'constants/creditPlus'
import { useDispatch, useSelector } from 'react-redux'
import { CreditPlusSelectors } from 'views/HousingEdge/CreditPlus/selectors'
import { openUrl } from 'views/RentPay/RentPayHelpers'
import { MTPActions } from 'modules/MTP/actions'
import { CreditPlusActions } from 'views/HousingEdge/CreditPlus/actions'
import { EMPTY_STATE } from 'common/constants/AppConstants'
import { orderStatus } from 'constants/EdgeConstants'
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection: 'row',
    justifyContent: 'flex-end',
    marginTop: 18
  },
  leftButtonContainer: {
    width: '50%',
    paddingRight: moderateScale(4)
  },
  rightButtonContainer: {
    width: '50%',
    paddingLeft: moderateScale(4)
  },
  buttonContainerStyle: {
    backgroundColor: colors.transparent2,
    borderWidth: 1,
    borderColor: colors.black37
  }
})
 
const CreditPlusCardFooter = ({ data, closeDetails, index }) => {
  const dispatch = useDispatch()
  const { order } = data
  const {
    subscriptionEndDate, subscriptionExpectedEndDate, createdAt, validity,
    status
  } = order || {}
 
  const isFailed = status === orderStatus.PAYMENT_FAILED
 
  const { renewInAdvance, subscribed } = useSelector(CreditPlusSelectors.getPocSubscriptionDetails)
  const shouldShowCp = useSelector(CreditPlusSelectors.shouldShowCp)
 
  const subscriptionExpiryDate = useMemo(() => (
    subscriptionEndDate || subscriptionExpectedEndDate
  ), [subscriptionEndDate, subscriptionExpectedEndDate])
  // calculation for credit plus validity
  const validityDetails = useMemo(() => (
    getSubscriptionValidityObject(createdAt, validity, subscriptionExpiryDate, status)
  ), [validity, createdAt, subscriptionExpiryDate])
 
  const primaryButtonTitle = useMemo(() => {
    if (subscribed && !renewInAdvance) {
      return ''
    }
    if (!subscribed && isFailed) {
      return creditPlusStrings.subscribeNow
    }
    if (!subscribed && index === 0 && shouldShowCp) {
      return creditPlusStrings.renew
    }
    if (renewInAdvance && validityDetails.valid && shouldShowCp) {
      return creditPlusStrings.renewAdvance
    }
    return ''
  }, [renewInAdvance, subscribed, validityDetails, index, shouldShowCp, isFailed])
 
  const downloadReceipt = (url) => {
    openUrl(url)
  }
 
  const onPressRenew = () => {
    closeDetails?.()
    dispatch(CreditPlusActions.openSubscriptionModal({
      type: TYPES.onlySubscription
    }))
  }
 
  const onDownloadSubscriptonReceipt = () => {
    if (order?.subscriptionReceiptUrl) {
      downloadReceipt(order?.subscriptionReceiptUrl)
      return
    }
    const subscription = buttonsKey.SUBSCRIPTION_FEE_INVOICE
    const docTypeEnum = mtpButtons[subscription].docTypeEnum
    const docUrl = mtpButtons[subscription].url
    dispatch(MTPActions.fetchMtpReceipt(order?.id, docTypeEnum, docUrl, downloadReceipt))
  }
 
  return (
    <View style={styles.container}>
      {!isFailed && (
        <View style={styles.leftButtonContainer}>
          <ButtonWrapper
            onPress={onDownloadSubscriptonReceipt}
            title={MTPStrings.DOWNLOAD_INVOICE}
            containerStyle={styles.buttonContainerStyle}
          />
        </View>
      )}
      {!!primaryButtonTitle
      && (
        <View style={styles.rightButtonContainer}>
          <ButtonWrapper
            onPress={onPressRenew}
            title={primaryButtonTitle}
            titleColor={colors.purple10}
            backgroundColor={colors.white}
          />
        </View>
      )}
    </View>
  )
}
 
CreditPlusCardFooter.propTypes = {
  data: object.isRequired,
  closeDetails: func,
  index: number.isRequired
}
 
CreditPlusCardFooter.defaultProps = {
  closeDetails: EMPTY_STATE.METHOD
}
 
export default memo(CreditPlusCardFooter)
